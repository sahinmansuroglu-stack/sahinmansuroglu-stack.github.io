Soru 1 Abstract Sınıf Tanımlama (25 Puan)

Aşağıda Verilenlere Göre Ogrenci adında abstract bir sınıf tanımlayınız.

  • ad, soyad, ogrenciNo, dogumTarihi ve başlamaYili olmak üzere 5 tane private field bulunacak.
  • Herbir field için birer tane property oluşturarak encapsulation yapılacak.
  • Validasyonlar: Ad ve Soyad boş ise hata fırlat. OgrenciNo 3 basamaklı değilse hata fırlat. Tarihler 1990-2023 arasında değilse hata fırlat.
  • Yas adında readonly property tanımlanarak dogumTarihinden yaş hesaplanacak.
  • Değer ataması kurucu metot (constructor) içerisinde yapılacak.
  • enOnemliOlayi() adında abstract metot tanımlanacak.
  • bilgileriEkranaYaz() adında virtual metot tanımlanacak.
Ogrenci.cs 
using System;

namespace SinavUygulamasi
{
    public abstract class Ogrenci
    {
        // Private Fields
        private string ad;
        private string soyad;
        private int ogrenciNo;
        private int dogumTarihi;
        private int baslamaYili;

        // Constructor
        public Ogrenci(string ad, string soyad, int ogrenciNo, int dogumTarihi, int baslamaYili)
        {
            this.Ad = ad;
            this.Soyad = soyad;
            this.OgrenciNo = ogrenciNo;
            this.DogumTarihi = dogumTarihi;
            this.BaslamaYili = baslamaYili;
        }

        // Encapsulation & Validations
        public string Ad
        {
            get { return ad; }
            set
            {
                if (string.IsNullOrWhiteSpace(value))
                    throw new Exception("Ad boş olamaz!");
                ad = value;
            }
        }

        public string Soyad
        {
            get { return soyad; }
            set
            {
                if (string.IsNullOrWhiteSpace(value))
                    throw new Exception("Soyad boş olamaz!");
                soyad = value;
            }
        }

        public int OgrenciNo
        {
            get { return ogrenciNo; }
            set
            {
                if (value < 100 || value > 999)
                    throw new Exception("Öğrenci No 3 basamaklı olmalıdır!");
                ogrenciNo = value;
            }
        }

        public int DogumTarihi
        {
            get { return dogumTarihi; }
            set
            {
                if (value < 1990 || value > 2023)
                    throw new Exception("Doğum Tarihi 1990-2023 arasında olmalıdır!");
                dogumTarihi = value;
            }
        }

        public int BaslamaYili
        {
            get { return baslamaYili; }
            set
            {
                if (value < 1990 || value > 2023)
                    throw new Exception("Başlama Yılı 1990-2023 arasında olmalıdır!");
                baslamaYili = value;
            }
        }

        // Readonly Property
        public int Yas
        {
            get { return DateTime.Now.Year - this.dogumTarihi; }
        }

        // Abstract Metot
        public abstract string enOnemliOlayi();

        // Virtual Metot
        public virtual void bilgileriEkranaYaz()
        {
            Console.WriteLine("--------------------------------");
            Console.WriteLine($"Ad Soyad: {Ad} {Soyad}");
            Console.WriteLine($"Numara: {OgrenciNo}");
            Console.WriteLine($"Doğum Yılı: {DogumTarihi} (Yaş: {Yas})");
            Console.WriteLine($"Başlama Yılı: {BaslamaYili}");
        }
    }
}

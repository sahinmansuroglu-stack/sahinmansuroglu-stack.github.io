Soru 4 OrtaOkulOgrencisi Sınıfı (15 Puan)

Ogrenci class'ından ve IOgrenciIslem interface'inden türetiniz.

  • Constructor tanımlanacak ve Base kullanılarak parent sınıfın kurucusu çağrılacak.
  • enOnemliOlayi() -> "LGS Sınavı" döndürecek.
  • matematikDersIcerigi() -> "Temel Matematik" döndürecek.
  • mezunOlacagiYasiHesapla() -> Yaş + 4.
  • bilgileriEkranaYaz() override edilecek, base metod çağrılacak.
  • toString() override edilecek (Ad, Soyad ve OgrenciNo).
  • gidebileceğiOkullariEkranaYaz() -> "Gidebileceği Okullar: MTAL, Anadolu Lisesi, Fen Lisesi".
OrtaOkulOgrencisi.cs 
using System;

namespace SinavUygulamasi
{
    public class OrtaOkulOgrencisi : Ogrenci, IOgrenciIslem
    {
        // Constructor ve Base kullanımı
        public OrtaOkulOgrencisi(string ad, string soyad, int ogrenciNo, int dogumTarihi, int baslamaYili) 
            : base(ad, soyad, ogrenciNo, dogumTarihi, baslamaYili)
        {
        }

        // Abstract Metot Uygulanması
        public override string enOnemliOlayi()
        {
            return "LGS Sınavı";
        }

        // Interface Metotları
        public string matematikDersIcerigi()
        {
            return "Temel Matematik";
        }

        public int mezunOlacagiYasiHesapla()
        {
            return Yas + 4;
        }

        // Override İşlemleri
        public override void bilgileriEkranaYaz()
        {
            base.bilgileriEkranaYaz();
            Console.WriteLine($"Ders: {matematikDersIcerigi()}");
            Console.WriteLine($"Olay: {enOnemliOlayi()}");
            Console.WriteLine($"Mezuniyet Yaşı: {mezunOlacagiYasiHesapla()}");
        }

        public override string ToString()
        {
            return $"{Ad} {Soyad} - {OgrenciNo}";
        }

        // Sınıfa Özel Metot
        public void gidebileceğiOkullariEkranaYaz()
        {
            Console.WriteLine("Gidebileceği Okullar: MTAL, Anadolu Lisesi, Fen Lisesi");
        }
    }
}

Soru 5 Liste Tanımlama ve Veri Ekleme (15 Puan)

İçerisinde ilkokul ve ortaokul öğrencilerinin bulunabileceği bir liste tanımlayın. Bu Listeye aşağıda belirtilen 3 Adet İlkokul öğrencisi ve 3 adet Ortaokul öğrencisi ekleyin.

  • OrtaOkulOgrencisi("Zehra", "ACI", 354, 2009, 2019)
  • IlkOkulOgrencisi("Esra", "ZEYBEK", 985, 2014, 2000)
  • OrtaOkulOgrencisi("Ali", "AKIN", 654, 2010, 2020)
  • IlkOkulOgrencisi("Ahmet", "EKER", 387, 2015, 2000)
  • OrtaOkulOgrencisi("Ahmet", "ÇELİK", 658, 2011, 2022)
  • IlkOkulOgrencisi("Arda", "AYDIN", 498, 2013, 2008)
Program.cs (Main Metodu Başı) 
using System;
using System.Collections.Generic;

namespace SinavUygulamasi
{
    class Program
    {
        static void Main(string[] args)
        {
            // Soru 5 Çözümü
            // Polymorphism: Listeyi ana sınıf (Ogrenci) tipinde oluşturuyoruz.
            
            List<Ogrenci> ogrenciler = new List<Ogrenci>();

            try
            {
                ogrenciler.Add(new OrtaOkulOgrencisi("Zehra", "ACI", 354, 2009, 2019));
                ogrenciler.Add(new IlkOkulOgrencisi("Esra", "ZEYBEK", 985, 2014, 2000));
                ogrenciler.Add(new OrtaOkulOgrencisi("Ali", "AKIN", 654, 2010, 2020));
                ogrenciler.Add(new IlkOkulOgrencisi("Ahmet", "EKER", 387, 2015, 2000));
                ogrenciler.Add(new OrtaOkulOgrencisi("Ahmet", "ÇELİK", 658, 2011, 2022));
                ogrenciler.Add(new IlkOkulOgrencisi("Arda", "AYDIN", 498, 2013, 2008));
            }
            catch (Exception ex)
            {
                Console.WriteLine("Hata: " + ex.Message);
            }
            
            // ... Programın devamı ...
        }
    }
}

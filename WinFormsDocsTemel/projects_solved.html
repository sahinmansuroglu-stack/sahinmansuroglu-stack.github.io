Proje 5: Okul Yönetim Sistemi (Tek Sınıf)

Amaç: Tek bir Ogrenci sınıfı kullanarak Ortaokul ve Lise öğrencilerini yönetmek.

Kullanılan Sınıf:

public class Ogrenci
{
    private string ad;
    private string soyad;
    private int ogrenciNo;
    private int dogumTarihi;
    private int baslamaYili;
    private string egitimDuzeyi; 

    public Ogrenci(string ad, string soyad, int ogrenciNo, int dogumTarihi, int baslamaYili, string egitimDuzeyi)
    {
        this.Ad = ad;
        this.Soyad = soyad;
        this.OgrenciNo = ogrenciNo;
        this.DogumTarihi = dogumTarihi;
        this.BaslamaYili = baslamaYili;
        this.EgitimDuzeyi = egitimDuzeyi;
    }

    public string Ad
    {
        get { return ad; }
        set
        {
            if (string.IsNullOrWhiteSpace(value))
                throw new Exception("Ad boş olamaz!");
            ad = value;
        }
    }

    public string Soyad
    {
        get { return soyad; }
        set
        {
            if (string.IsNullOrWhiteSpace(value))
                throw new Exception("Soyad boş olamaz!");
            soyad = value;
        }
    }

    public int OgrenciNo
    {
        get { return ogrenciNo; }
        set
        {
            if (value < 100 || value > 999)
                throw new Exception("Öğrenci No 3 basamaklı olmalıdır!");
            ogrenciNo = value;
        }
    }

    public int DogumTarihi
    {
        get { return dogumTarihi; }
        set
        {
            if (value < 1990 || value > 2023)
                throw new Exception("Doğum Tarihi 1990-2023 arasında olmalıdır!");
            dogumTarihi = value;
        }
    }

    public int BaslamaYili
    {
        get { return baslamaYili; }
        set
        {
            if (value < 1990 || value > 2023)
                throw new Exception("Başlama Yılı 1990-2023 arasında olmalıdır!");
            baslamaYili = value;
        }
    }

    public string EgitimDuzeyi
    {
        get { return egitimDuzeyi; }
        set { egitimDuzeyi = value; }
    }

    public int Yas
    {
        get { return DateTime.Now.Year - this.dogumTarihi; }
    }

    public string enOnemliOlayi()
    {
        if (this.EgitimDuzeyi == "Ortaokul")
            return "LGS Sınavı";
        else if (this.EgitimDuzeyi == "Lise")
            return "Üniversite Sınavı";
        else
            return "Bilinmiyor";
    }

    public void bilgileriEkranaYaz()
    {
        
    }
}

Proje 6: Döviz Çevirici

Amaç: Girilen TL miktarını seçilen para birimine (USD, EUR) çevirmek.

Kullanılan Kontroller: 1 TextBox (Miktar), 1 ComboBox (Para Birimi), 1 Button (Çevir), 1 Label (Sonuç).

Kodlar:

private void btnCevir_Click(object sender, EventArgs e)
{
    try
    {
        double miktar = Convert.ToDouble(txtMiktar.Text);
        double kur = 0;
        string birim = cmbParaBirimi.SelectedItem.ToString();

        if (birim == "USD") kur = 32.50; // Örnek kur
        else if (birim == "EUR") kur = 35.20; // Örnek kur

        if (kur > 0)
        {
            double sonuc = miktar / kur;
            lblSonuc.Text = string.Format("{0:0.00} {1}", sonuc, birim);
        }
        else
        {
            MessageBox.Show("Lütfen para birimi seçiniz.");
        }
    }
    catch
    {
        MessageBox.Show("Geçerli bir miktar giriniz.");
    }
}

Proje 7: Kütüphane Kitap Takip (Class Kullanımı)

Amaç: Kitap sınıfını kullanarak kitap bilgilerini listeye eklemek.

Kullanılan Sınıf:

public class Kitap
{
    public string Ad { get; set; }
    public string Yazar { get; set; }
    public int SayfaSayisi { get; set; }

    public Kitap(string ad, string yazar, int sayfa)
    {
        Ad = ad;
        Yazar = yazar;
        SayfaSayisi = sayfa;
    }

    public string BilgiGetir()
    {
        return $"{Ad} - {Yazar} ({SayfaSayisi} Sayfa)";
    }
}

Form Kodları:

private void btnKitapEkle_Click(object sender, EventArgs e)
{
    string ad = txtKitapAd.Text;
    string yazar = txtYazar.Text;
    int sayfa = int.Parse(txtSayfa.Text);

    Kitap yeniKitap = new Kitap(ad, yazar, sayfa);

    listBoxKitaplar.Items.Add(yeniKitap.BilgiGetir());
}

Proje 8: Sayı Tahmin Oyunu

Amaç: Bilgisayarın tuttuğu 1-100 arası sayıyı tahmin etmek.

Kullanılan Kontroller: 1 TextBox (Tahmin), 1 Button (Tahmin Et), 1 Label (Yönlendirme).

Kodlar:

// Form Load veya Başlat butonu ile sayı tutulur
Random rnd = new Random();
int hedefSayi = rnd.Next(1, 101);
int hak = 5;

private void btnTahmin_Click(object sender, EventArgs e)
{
    int tahmin = int.Parse(txtTahmin.Text);
    hak--;

    if (tahmin == hedefSayi)
    {
        MessageBox.Show("Tebrikler! Doğru Bildiniz.");
        btnTahmin.Enabled = false;
    }
    else if (hak == 0)
    {
        MessageBox.Show("Hakkınız bitti. Sayı: " + hedefSayi);
        btnTahmin.Enabled = false;
    }
    else if (tahmin < hedefSayi)
    {
        lblYonlendirme.Text = "Daha büyük bir sayı girin. Kalan Hak: " + hak;
    }
    else
    {
        lblYonlendirme.Text = "Daha küçük bir sayı girin. Kalan Hak: " + hak;
    }
}

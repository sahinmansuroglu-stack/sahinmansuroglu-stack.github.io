ADIM 1

Giriş Formu (Login Form)

Kullanıcının adı ve şifresiyle sisteme girdiği ilk ekrandır.

🔑 Giriş Mantığı

Servis katmanında SELECT * FROM Users WHERE KullaniciAdi=@ad AND Sifre=@sifre sorgusunu çalıştırarak dönen değerin boş olup olmadığını kontrol ederiz.

// UserService.cs içine eklenecek metot
public async Task<User> LoginAsync(string ad, string sifre) 
{
    using var db = new MySqlConnection(_conn);
    string sql = "SELECT * FROM Users WHERE KullaniciAdi = @ad AND Sifre = @sifre";
    return await db.QueryFirstOrDefaultAsync<User>(sql, new { ad, sifre });
}

// FormLogin.cs - Buton Click Olayı
private async void btnGiris_Click(object sender, EventArgs e) 
{
    var user = await _userService.LoginAsync(txtAd.Text, txtSifre.Text);
    if (user != null) {
        FormAnaSayfa anaSayfa = new FormAnaSayfa(user); // Kullanıcıyı aktar
        anaSayfa.Show();
        this.Hide();
    } else {
        MessageBox.Show("Hatalı Giriş!");
    }
}
ADIM 2

Ortalama Puanlar Listesi

Login formunun altında veya ana sayfada, filmlerin genel başarı ortalamasını gösteren listedir.

-- SQL: Filmlerin Ortalama Puanlarını Getir
SELECT 
    M.FilmAdi, 
    AVG(W.VerilenPuan) as OrtalamaPuan,
    COUNT(W.KayitID) as ToplamIzlenme
FROM Movies M
LEFT JOIN Watchlist W ON M.FilmID = W.FilmID
GROUP BY M.FilmID, M.FilmAdi;
// ReportService.cs içine eklenecek
public class FilmBasariRaporu {
    public string FilmAdi { get; set; }
    public double OrtalamaPuan { get; set; }
    public int ToplamIzlenme { get; set; }
}

public async Task<List<FilmBasariRaporu>> GetAverageRatingsAsync() {
    using var db = new MySqlConnection(_conn);
    string sql = "SELECT M.FilmAdi, AVG(W.VerilenPuan) as OrtalamaPuan, COUNT(W.KayitID) as ToplamIzlenme FROM Movies M LEFT JOIN Watchlist W ON M.FilmID = W.FilmID GROUP BY M.FilmID, M.FilmAdi";
    var result = await db.QueryAsync<FilmBasariRaporu>(sql);
    return result.ToList();
}
ADIM 3

İzleme Akışı ve Dinamik Puanlama

Gerçekçi bir deneyim için kullanıcı önce izlemeye başlar, bitirdiğinde ise puanını verir.

🔄 İki Aşamalı Mantık

1. İzlemeye Başla: Veritabanında durumu 'İzleniyor' olarak bir kayıt açar.
2. Bitir ve Puanla: Mevcut kaydı güncelleyerek durumu 'Tamamlandı' yapar ve puanı işler.

1. İzlemeye Başla (Start)

// FormAnaSayfa.cs
private async void btnIzlemeyeBasla_Click(object sender, EventArgs e) 
{
    var izlemeKaydi = new Watchlist {
        KullaniciID = _girisYapanUser.KullaniciID,
        FilmID = _seciliFilmId,
        Durum = "İzleniyor",
        VerilenPuan = 0 // Henüz puan verilmedi
    };

    await _watchlistService.StartWatchingAsync(izlemeKaydi);
    btnIzlemeyeBasla.Enabled = false; // Tekrar basılmasın
    btnBitirVePuanla.Visible = true;  // Bitir butonu görünsün
}

2. Bitir ve Puanla (Finish & Rate)

// FormAnaSayfa.cs
private async void btnBitirVePuanla_Click(object sender, EventArgs e) 
{
    // Mevcut 'İzleniyor' kaydını 'Tamamlandı' olarak güncelle
    var guncelleme = new Watchlist {
        KullaniciID = _girisYapanUser.KullaniciID,
        FilmID = _seciliFilmId,
        VerilenPuan = int.Parse(txtPuan.Text),
        Durum = "Tamamlandı"
    };

    await _watchlistService.UpdateStatusAndRateAsync(guncelleme);
    MessageBox.Show("Harika! Film listenize eklendi ve puanlandı.");
    
    btnBitirVePuanla.Visible = false;
    btnIzlemeyeBasla.Enabled = true;
    await OrtalamaPuanlariGuncelle(); // İstatistikleri yenile
}

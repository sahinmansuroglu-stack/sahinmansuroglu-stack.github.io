🍿 3. İzleme Listesi Modülü (Watchlist Module)

Kapsam:

Kullanıcılar ve filmler arasındaki ilişkiyi (puan ve durum) yöneten köprü modüldür.

Watchlist.cs MODEL
namespace NetflixApp.Models
{
    public class Watchlist 
    {
        public int KayitID { get; set; }
        public int KullaniciID { get; set; } // FK
        public string FilmID { get; set; }    // FK
        public int VerilenPuan { get; set; }
        public string Durum { get; set; }
    }
}
WatchlistService.cs SERVICE
using Dapper;
using MySql.Data.MySqlClient;
using System.Data;
using System.Threading.Tasks;
using NetflixApp.Models;

public class WatchlistService 
{
    private readonly string _conn = "host=localhost;port=3306;user id=root;password=mtal123;database=filmdb;";

    public async Task<int> StartWatchingAsync(Watchlist w) 
    {
        using var db = new MySqlConnection(_conn);
        string sql = "INSERT INTO Watchlist (KullaniciID, FilmID, VerilenPuan, Durum) VALUES (@KullaniciID, @FilmID, 0, 'İzleniyor')";
        return await db.ExecuteAsync(sql, w);
    }

    public async Task<int> UpdateStatusAndRateAsync(Watchlist w) 
    {
        using var db = new MySqlConnection(_conn);
        string sql = "UPDATE Watchlist SET VerilenPuan = @VerilenPuan, Durum = 'Tamamlandı' WHERE KullaniciID = @KullaniciID AND FilmID = @FilmID AND Durum = 'İzleniyor'";
        return await db.ExecuteAsync(sql, w);
    }
}

📈 4. Raporlama Modülü (Reporting Module)

Kapsam:

Tabloları birleştirerek (JOIN) kullanıcıya anlamlı raporlar sunar.

ReportService.cs REPORT SERVICE
using Dapper;
using MySql.Data.MySqlClient;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

public class UserMovieReport 
{
    public string KullaniciAdi { get; set; }
    public string FilmAdi { get; set; }
    public int VerilenPuan { get; set; }
}

public class ReportService 
{
    private readonly string _conn = "host=localhost;port=3306;user id=root;password=mtal123;database=filmdb;";

    public async Task<List<UserMovieReport>> GetFullReportAsync() 
    {
        using var db = new MySqlConnection(_conn);
        string sql = @"SELECT U.KullaniciAdi, M.FilmAdi, W.VerilenPuan 
                       FROM Watchlist W
                       INNER JOIN Users U ON W.KullaniciID = U.KullaniciID
                       INNER JOIN Movies M ON W.FilmID = M.FilmID";
        
        var report = await db.QueryAsync<UserMovieReport>(sql);
        return report.ToList();
    }
}

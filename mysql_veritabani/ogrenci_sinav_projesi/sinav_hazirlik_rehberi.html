🗄️ Veritabanı Hazırlığı

CREATE TABLE ogrenciler (

    Soyad VARCHAR(50) NOT NULL,
    Numara VARCHAR(10) NOT NULL,

2 Fluent Validation (Doğrulama)

Kullanıcının girdiği verilerin kurallara uyup uymadığını kontrol eder.

OgrenciValidator.cs 
using FluentValidation;

public class OgrenciValidator : AbstractValidator<Ogrenci>
{
    public OgrenciValidator()
    {
        RuleFor(x => x.Ad).NotEmpty().WithMessage("Ad boş geçilemez.");
        RuleFor(x => x.Soyad).NotEmpty().WithMessage("Soyad boş geçilemez.");
        RuleFor(x => x.Numara).Length(1, 10).WithMessage("Numara 1-10 hane olmalı.");
        RuleFor(x => x.Puan).InclusiveBetween(0, 100).WithMessage("Puan 0-100 arası olmalı.");
    }
}

3 Veritabanı Servisi (Dapper)

OgrenciService.cs 
using Dapper;
using MySql.Data.MySqlClient;
using System.Collections.Generic;
using System.Data;
using System.Linq;
using System.Threading.Tasks;

public class OgrenciService
{
    private readonly string _conn = "host=localhost;port=3306;user id=root;password=mtal123;database=okuldb;";
    private IDbConnection GetConn() => new MySqlConnection(_conn);

    public async Task<List<Ogrenci>> GetListAsync() {
        using var db = GetConn();
        return (await db.QueryAsync<Ogrenci>("SELECT * FROM ogrenciler")).ToList();
    }

    public async Task<int> AddAsync(Ogrenci o) {
        using var db = GetConn();
        string sql = "INSERT INTO ogrenciler (Ad, Soyad, Numara, Sinif, Puan) VALUES (@Ad, @Soyad, @Numara, @Sinif, @Puan)";
        return await db.ExecuteAsync(sql, o);
    }

    public async Task<int> UpdateAsync(Ogrenci o) {
        using var db = GetConn();
        string sql = "UPDATE ogrenciler SET Ad=@Ad, Soyad=@Soyad, Numara=@Numara, Sinif=@Sinif, Puan=@Puan WHERE OgrenciID=@OgrenciID";
        return await db.ExecuteAsync(sql, o);
    }

    public async Task<int> DeleteAsync(int id) {
        using var db = GetConn();
        return await db.ExecuteAsync("DELETE FROM ogrenciler WHERE OgrenciID=@id", new { id });
    }

    public async Task<List<Ogrenci>> SearchAsync(string text) {
        using var db = GetConn();
        string sql = "SELECT * FROM ogrenciler WHERE Ad LIKE @t OR Soyad LIKE @t OR Numara LIKE @t";
        return (await db.QueryAsync<Ogrenci>(sql, new { t = "%" + text + "%" })).ToList();
    }
}

4 Form Olayları (UI Logic)

Button Click ve Validation kullanımı.

OgrenciService _servis = new OgrenciService();
int _seciliId = 0;

// EKLEME BUTONU
private async void btnEkle_Click(object sender, EventArgs e) {
    var ogrenci = new Ogrenci {
        Ad = txtAd.Text, Soyad = txtSoyad.Text, 
        Numara = txtNumara.Text, Sinif = txtSinif.Text, 
        Puan = int.Parse(txtPuan.Text)
    };

    var validator = new OgrenciValidator();
    var sonuc = validator.Validate(ogrenci);

    if (sonuc.IsValid) {
        await _servis.AddAsync(ogrenci);
        MessageBox.Show("Öğrenci başarıyla eklendi!");
        await Listele();
    } else {
        string hatalar = string.Join("\n", sonuc.Errors.Select(x => x.ErrorMessage));
        MessageBox.Show(hatalar, "Hata");
    }
}

// GÜNCELLEME BUTONU
private async void btnGuncelle_Click(object sender, EventArgs e) {
    if (_seciliId == 0) { MessageBox.Show("Önce listeden bir öğrenci seçin!"); return; }

    var ogrenci = new Ogrenci {
        OgrenciID = _seciliId,
        Ad = txtAd.Text, Soyad = txtSoyad.Text, 
        Numara = txtNumara.Text, Sinif = txtSinif.Text, 
        Puan = int.Parse(txtPuan.Text)
    };

    var validator = new OgrenciValidator();
    if (validator.Validate(ogrenci).IsValid) {
        await _servis.UpdateAsync(ogrenci);
        MessageBox.Show("Bilgiler güncellendi!");
        await Listele();
    }
}

// SİLME BUTONU
private async void btnSil_Click(object sender, EventArgs e) {
    if (_seciliId == 0) return;

    var cevap = MessageBox.Show("Seçili öğrenciyi silmek istediğinize emin misiniz?", "Onay", MessageBoxButtons.YesNo);
    if (cevap == DialogResult.Yes) {
        await _servis.DeleteAsync(_seciliId);
        await Listele();
        _seciliId = 0; // Seçimi temizle
    }
}

// ARAMA OLAYI (TextChanged)
private async void txtAra_TextChanged(object sender, EventArgs e) {
    dgvOgrenciler.DataSource = await _servis.SearchAsync(txtAra.Text);
}

// DATAGRID TIKLAMA (ID Seçme ve Kutuları Doldurma)
private void dgvOgrenciler_CellClick(object sender, DataGridViewCellEventArgs e) {
    if (e.RowIndex >= 0) {
        var satir = dgvOgrenciler.Rows[e.RowIndex];
        _seciliId = (int)satir.Cells["OgrenciID"].Value;
        
        txtAd.Text = satir.Cells["Ad"].Value.ToString();
        txtSoyad.Text = satir.Cells["Soyad"].Value.ToString();
        txtNumara.Text = satir.Cells["Numara"].Value.ToString();
        txtSinif.Text = satir.Cells["Sinif"].Value.ToString();
        txtPuan.Text = satir.Cells["Puan"].Value.ToString();
    }
}

using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;

public class OrderViewModel
{
    [Required(ErrorMessage = "Ürün seçimi zorunlu")]
    [Display(Name = "Ürün")]
    public int ProductId { get; set; }
    
    [Range(1, 100, ErrorMessage = "Adet 1-100 arası olmalı")]
    [Display(Name = "Adet")]
    public int Quantity { get; set; } = 1;
    
    [Required]
    [Display(Name = "Teslimat Tipi")]
    public string DeliveryType { get; set; }
    
    [Display(Name = "Ek Hizmetler")]
    public List<string> ExtraServices { get; set; } = new();
    
    [Required]
    [Display(Name = "Teslimat Adresi")]
    public string Address { get; set; }
    
    [Display(Name = "Ödeme Yöntemi")]
    public string PaymentMethod { get; set; }
    
    [Display(Name = "Not")]
    public string? Note { get; set; }
}

Controller

Controllers/OrderController.cs
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Rendering;
using System.Collections.Generic;
using System.Linq;

public class OrderController : Controller
{
    private static List<Product> _products = new()
    {
        new Product { Id = 1, Name = "Laptop", Price = 15000 },
        new Product { Id = 2, Name = "Telefon", Price = 8000 },
        new Product { Id = 3, Name = "Tablet", Price = 5000 }
    };
    
    [HttpGet]
    public IActionResult Create()
    {
        // Ürün dropdown için SelectList
        ViewBag.Products = new SelectList(_products, "Id", "Name");
        
        // Teslimat tipleri
        ViewBag.DeliveryTypes = new List<SelectListItem>
        {
            new() { Value = "standard", Text = "Standart (3-5 gün)" },
            new() { Value = "express", Text = "Hızlı (1-2 gün) +50₺" },
            new() { Value = "same_day", Text = "Aynı Gün +100₺" }
        };
        
        // Ödeme yöntemleri
        ViewBag.PaymentMethods = new[] { "Kredi Kartı", "Havale/EFT", "Kapıda Ödeme" };
        
        return View();
    }
    
    [HttpPost]
    public IActionResult Create(OrderViewModel model)
    {
        if (ModelState.IsValid)
        {
            // Seçilen ürünü bul
            var product = _products.FirstOrDefault(p => p.Id == model.ProductId);
            
            // Toplam hesapla
            decimal total = product.Price * model.Quantity;
            
            // Kargo ücreti
            if (model.DeliveryType == "express") total += 50;
            if (model.DeliveryType == "same_day") total += 100;
            
            // Ek hizmetler
            if (model.ExtraServices.Contains("insurance")) total += 30;
            if (model.ExtraServices.Contains("gift_wrap")) total += 20;
            
            ViewBag.Product = product;
            ViewBag.Total = total;
            
            return View("Confirmation", model);
        }
        
        // Hata varsa dropdown'ları tekrar doldur
        ViewBag.Products = new SelectList(_products, "Id", "Name", model.ProductId);
        return View(model);
    }
}

View - Sipariş Formu

Views/Order/Create.cshtml
@model OrderViewModel

<h1>🛒 Sipariş Ver</h1>

<form asp-action="Create" method="post">
    <div asp-validation-summary="ModelOnly" class="alert alert-danger"></div>
    
    <!-- Ürün Seçimi (Dropdown) -->
    <div class="mb-3">
        <label asp-for="ProductId"></label>
        <select asp-for="ProductId" asp-items="ViewBag.Products" class="form-select">
            <option value="">-- Ürün Seçin --</option>
        </select>
        <span asp-validation-for="ProductId" class="text-danger"></span>
    </div>
    
    <!-- Adet -->
    <div class="mb-3">
        <label asp-for="Quantity"></label>
        <input asp-for="Quantity" type="number" min="1" max="100" class="form-control" />
        <span asp-validation-for="Quantity" class="text-danger"></span>
    </div>
    
    <!-- Teslimat Tipi (Radio) -->
    <div class="mb-3">
        <label class="form-label">Teslimat Tipi</label>
        @foreach (var type in ViewBag.DeliveryTypes as List<SelectListItem>)
        {
            <div class="form-check">
                <input type="radio" asp-for="DeliveryType" value="@type.Value" class="form-check-input" />
                <label class="form-check-label">@type.Text</label>
            </div>
        }
        <span asp-validation-for="DeliveryType" class="text-danger"></span>
    </div>
    
    <!-- Ek Hizmetler (Checkbox) -->
    <div class="mb-3">
        <label class="form-label">Ek Hizmetler</label>
        <div class="form-check">
            <input type="checkbox" name="ExtraServices" value="insurance" class="form-check-input" />
            <label class="form-check-label">Kargo Sigortası (+30₺)</label>
        </div>
        <div class="form-check">
            <input type="checkbox" name="ExtraServices" value="gift_wrap" class="form-check-input" />
            <label class="form-check-label">Hediye Paketi (+20₺)</label>
        </div>
    </div>
    
    <!-- Ödeme Yöntemi (Select) -->
    <div class="mb-3">
        <label asp-for="PaymentMethod"></label>
        <select asp-for="PaymentMethod" class="form-select">
            <option value="">-- Seçin --</option>
            @foreach (var method in ViewBag.PaymentMethods as string[])
            {
                <option value="@method">@method</option>
            }
        </select>
    </div>
    
    <!-- Adres (Textarea) -->
    <div class="mb-3">
        <label asp-for="Address"></label>
        <textarea asp-for="Address" class="form-control" rows="3"></textarea>
        <span asp-validation-for="Address" class="text-danger"></span>
    </div>
    
    <button type="submit" class="btn btn-success btn-lg">Siparişi Tamamla</button>
</form>

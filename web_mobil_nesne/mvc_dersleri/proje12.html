Models

Models/Student.cs
public class Student
{
    public int Id { get; set; }
    public string StudentNumber { get; set; }
    public string FirstName { get; set; }
    public string LastName { get; set; }
    
    // One-to-Many: Öğrenci -> Bölüm
    public int DepartmentId { get; set; }
    public Department Department { get; set; }
    
    // Many-to-Many: Öğrenci <-> Ders
    public ICollection<StudentCourse> StudentCourses { get; set; }
    
    public string FullName => $"{FirstName} {LastName}";
    public int TotalCredits => StudentCourses?.Sum(sc => sc.Course?.Credit ?? 0) ?? 0;
}

public class Course
{
    public int Id { get; set; }
    public string Code { get; set; }  // BIL101, MAT201
    public string Name { get; set; }
    public int Credit { get; set; }
    public int Capacity { get; set; }
    
    // One-to-Many: Ders -> Öğretim Üyesi
    public int InstructorId { get; set; }
    public Instructor Instructor { get; set; }
    
    // Many-to-Many
    public ICollection<StudentCourse> StudentCourses { get; set; }
    
    public int EnrolledCount => StudentCourses?.Count ?? 0;
    public int RemainingCapacity => Capacity - EnrolledCount;
}
Models/StudentCourse.cs (Ara Tablo)
public class StudentCourse
{
    // Composite Key
    public int StudentId { get; set; }
    public Student Student { get; set; }
    
    public int CourseId { get; set; }
    public Course Course { get; set; }
    
    // Ek bilgiler
    public DateTime EnrollmentDate { get; set; } = DateTime.Now;
    public decimal? MidtermGrade { get; set; }
    public decimal? FinalGrade { get; set; }
    public decimal? AverageGrade => (MidtermGrade * 0.4m) + (FinalGrade * 0.6m);
    public string LetterGrade => AverageGrade switch
    {
        >= 90 => "AA",
        >= 85 => "BA",
        >= 80 => "BB",
        >= 75 => "CB",
        >= 70 => "CC",
        >= 60 => "DC",
        >= 50 => "DD",
        _ => "FF"
    };
}

DbContext

Data/UniversityDbContext.cs
protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    // Composite Primary Key
    modelBuilder.Entity<StudentCourse>()
        .HasKey(sc => new { sc.StudentId, sc.CourseId });
    
    // Many-to-Many ilişkiler
    modelBuilder.Entity<StudentCourse>()
        .HasOne(sc => sc.Student)
        .WithMany(s => s.StudentCourses)
        .HasForeignKey(sc => sc.StudentId);
    
    modelBuilder.Entity<StudentCourse>()
        .HasOne(sc => sc.Course)
        .WithMany(c => c.StudentCourses)
        .HasForeignKey(sc => sc.CourseId);
}

Servis

Services/EnrollmentService.cs
public class EnrollmentService : IEnrollmentService
{
    private readonly UniversityDbContext _context;
    private const int MaxCredits = 30;
    
    public async Task<EnrollmentResult> EnrollAsync(int studentId, int courseId)
    {
        // Öğrenciyi derslerle birlikte getir
        var student = await _context.Students
            .Include(s => s.StudentCourses)
                .ThenInclude(sc => sc.Course)
            .FirstOrDefaultAsync(s => s.Id == studentId);
        
        var course = await _context.Courses
            .Include(c => c.StudentCourses)
            .FirstOrDefaultAsync(c => c.Id == courseId);
        
        // Kontrol: Zaten kayıtlı mı?
        if (student.StudentCourses.Any(sc => sc.CourseId == courseId))
        {
            return new EnrollmentResult 
            { 
                Success = false, 
                Message = "Bu derse zaten kayıtlısınız!" 
            };
        }
        
        // Kontrol: Kontenjan dolu mu?
        if (course.RemainingCapacity <= 0)
        {
            return new EnrollmentResult 
            { 
                Success = false, 
                Message = "Ders kontenjanı dolu!" 
            };
        }
        
        // Kontrol: Kredi limiti aşılıyor mu?
        if (student.TotalCredits + course.Credit > MaxCredits)
        {
            return new EnrollmentResult 
            { 
                Success = false, 
                Message = $"Maksimum {MaxCredits} kredi alabilirsiniz!" 
            };
        }
        
        // Kayıt yap
        var enrollment = new StudentCourse
        {
            StudentId = studentId,
            CourseId = courseId,
            EnrollmentDate = DateTime.Now
        };
        
        _context.StudentCourses.Add(enrollment);
        await _context.SaveChangesAsync();
        
        return new EnrollmentResult 
        { 
            Success = true, 
            Message = $"{course.Name} dersine kayıt yapıldı!" 
        };
    }
    
    public async Task<List<Course>> GetAvailableCoursesAsync(int studentId)
    {
        var enrolledCourseIds = await _context.StudentCourses
            .Where(sc => sc.StudentId == studentId)
            .Select(sc => sc.CourseId)
            .ToListAsync();
        
        return await _context.Courses
            .Include(c => c.Instructor)
            .Include(c => c.StudentCourses)
            .Where(c => !enrolledCourseIds.Contains(c.Id))
            .Where(c => c.StudentCourses.Count < c.Capacity)
            .ToListAsync();
    }
}

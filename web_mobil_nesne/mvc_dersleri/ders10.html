using System.ComponentModel.DataAnnotations;

public class Student
{
    public int Id { get; set; }
    
    [Required(ErrorMessage = "Ad zorunludur")]
    [StringLength(50)]
    [Display(Name = "Ad")]
    public string FirstName { get; set; }
    
    [Required(ErrorMessage = "Soyad zorunludur")]
    [StringLength(50)]
    [Display(Name = "Soyad")]
    public string LastName { get; set; }
    
    [Required]
    [EmailAddress(ErrorMessage = "Geçerli e-posta girin")]
    [Display(Name = "E-posta")]
    public string Email { get; set; }
    
    [DataType(DataType.Date)]
    [Display(Name = "Doğum Tarihi")]
    public DateTime? BirthDate { get; set; }
    
    [Phone]
    [Display(Name = "Telefon")]
    public string Phone { get; set; }
    
    [Display(Name = "Aktif")]
    public bool IsActive { get; set; } = true;
    
    // Foreign Key
    [Required(ErrorMessage = "Sınıf seçimi zorunludur")]
    [Display(Name = "Sınıf")]
    public int ClassroomId { get; set; }
    public Classroom Classroom { get; set; }
    
    // Hesaplanmış özellik
    public string FullName => $"{FirstName} {LastName}";
}

2. DbContext

Data/SchoolDbContext.cs
using Microsoft.EntityFrameworkCore;

public class SchoolDbContext : DbContext
{
    public SchoolDbContext(DbContextOptions<SchoolDbContext> options)
        : base(options) { }
    
    public DbSet<Student> Students { get; set; }
    public DbSet<Classroom> Classrooms { get; set; }
    
    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        // Seed Data
        modelBuilder.Entity<Classroom>().HasData(
            new Classroom { Id = 1, Name = "9-A", Capacity = 30, TeacherName = "Ahmet Hoca" },
            new Classroom { Id = 2, Name = "10-A", Capacity = 28, TeacherName = "Fatma Hoca" },
            new Classroom { Id = 3, Name = "11-A", Capacity = 32, TeacherName = "Mehmet Hoca" },
            new Classroom { Id = 4, Name = "12-A", Capacity = 25, TeacherName = "Ayşe Hoca" }
        );
    }
}

3. StudentController (Tam)

Controllers/StudentController.cs
using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Rendering;
using Microsoft.EntityFrameworkCore;

public class StudentController : Controller
{
    private readonly SchoolDbContext _context;
    private const int PageSize = 10;

    public StudentController(SchoolDbContext context)
    {
        _context = context;
    }

    // GET: /Student?search=ali&classroomId=3&page=1
    public async Task<IActionResult> Index(
        string search, 
        int? classroomId, 
        int page = 1)
    {
        var query = _context.Students
            .Include(s => s.Classroom)
            .AsQueryable();

        // Arama
        if (!string.IsNullOrEmpty(search))
        {
            query = query.Where(s => 
                s.FirstName.Contains(search) || 
                s.LastName.Contains(search) ||
                s.Email.Contains(search));
        }

        // Sınıf filtresi
        if (classroomId.HasValue)
        {
            query = query.Where(s => s.ClassroomId == classroomId);
        }

        // Sayfalama
        var totalCount = await query.CountAsync();
        var students = await query
            .OrderBy(s => s.LastName)
            .Skip((page - 1) * PageSize)
            .Take(PageSize)
            .ToListAsync();

        // ViewBag değerleri
        ViewBag.Search = search;
        ViewBag.ClassroomId = classroomId;
        ViewBag.CurrentPage = page;
        ViewBag.TotalPages = (int)Math.Ceiling(totalCount / (double)PageSize);
        ViewBag.Classrooms = new SelectList(
            await _context.Classrooms.ToListAsync(), 
            "Id", "Name", classroomId);

        return View(students);
    }

    // GET: /Student/Create
    public async Task<IActionResult> Create()
    {
        ViewBag.Classrooms = new SelectList(
            await _context.Classrooms.ToListAsync(), 
            "Id", "Name");
        return View();
    }

    // POST: /Student/Create
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Create(Student student)
    {
        if (ModelState.IsValid)
        {
            _context.Add(student);
            await _context.SaveChangesAsync();
            TempData["Success"] = $"{student.FullName} başarıyla eklendi!";
            return RedirectToAction(nameof(Index));
        }
        
        ViewBag.Classrooms = new SelectList(
            await _context.Classrooms.ToListAsync(), 
            "Id", "Name", student.ClassroomId);
        return View(student);
    }

    // GET: /Student/Edit/5
    public async Task<IActionResult> Edit(int? id)
    {
        if (id == null) return NotFound();

        var student = await _context.Students.FindAsync(id);
        if (student == null) return NotFound();

        ViewBag.Classrooms = new SelectList(
            await _context.Classrooms.ToListAsync(), 
            "Id", "Name", student.ClassroomId);
        return View(student);
    }

    // POST: /Student/Edit/5
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Edit(int id, Student student)
    {
        if (id != student.Id) return NotFound();

        if (ModelState.IsValid)
        {
            _context.Update(student);
            await _context.SaveChangesAsync();
            TempData["Success"] = "Öğrenci güncellendi!";
            return RedirectToAction(nameof(Index));
        }
        
        ViewBag.Classrooms = new SelectList(
            await _context.Classrooms.ToListAsync(), 
            "Id", "Name", student.ClassroomId);
        return View(student);
    }

    // POST: /Student/Delete/5
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Delete(int id)
    {
        var student = await _context.Students.FindAsync(id);
        if (student != null)
        {
            _context.Students.Remove(student);
            await _context.SaveChangesAsync();
            TempData["Success"] = "Öğrenci silindi!";
        }
        return RedirectToAction(nameof(Index));
    }
}

public IActionResult Index()
{
    ViewBag.StudentCount = _context.Students.Count();
    ViewBag.ClassroomCount = _context.Classrooms.Count();
    ViewBag.ActiveStudents = _context.Students.Count(s => s.IsActive);
    
    var recentStudents = _context.Students
        .OrderByDescending(s => s.Id)
        .Take(5)
        .Include(s => s.Classroom)
        .ToList();
        
    return View(recentStudents);
}

public async Task<IActionResult> Index(string search, int? categoryId)
{

        .AsQueryable();
    
    if (!string.IsNullOrEmpty(search))
    {
        query = query.Where(b => 
            b.Title.Contains(search) || 
            b.Author.Contains(search));
    }
    
    if (categoryId.HasValue)
    {
        query = query.Where(b => b.CategoryId == categoryId);

    ViewBag.Categories = new SelectList(

    ViewBag.Search = search;
    
    return View(await query.ToListAsync());
}

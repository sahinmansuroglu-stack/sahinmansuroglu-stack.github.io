Model ve Data Annotations

Veri Doğrulama ve Kısıtlamalar

Data Annotations Nedir?

Data Annotations, Model özelliklerine kurallar ekleyerek veri doğrulama (validation) yapmamızı sağlar. System.ComponentModel.DataAnnotations namespace'ini kullanır.

💡 Neden Kullanırız?
Kullanıcıdan gelen verilerin geçerli olup olmadığını kontrol etmek için. Örneğin: boş bırakılamaz, e-posta formatı, minimum uzunluk.

Temel Annotations

Models/Student.cs
using System.ComponentModel.DataAnnotations;

public class Student
{
    public int Id { get; set; }
    
    [Required(ErrorMessage = "Ad alanı zorunludur")]
    [StringLength(50, MinimumLength = 2, 
        ErrorMessage = "Ad 2-50 karakter arasında olmalı")]
    [Display(Name = "Öğrenci Adı")]
    public string Name { get; set; }
    
    [Required(ErrorMessage = "E-posta zorunludur")]
    [EmailAddress(ErrorMessage = "Geçerli bir e-posta girin")]
    public string Email { get; set; }
    
    [Range(1, 12, ErrorMessage = "Sınıf 1-12 arası olmalı")]
    [Display(Name = "Sınıf")]
    public int Grade { get; set; }
    
    [Phone(ErrorMessage = "Geçerli telefon numarası girin")]
    public string Phone { get; set; }
}
Attribute Açıklama Örnek
[Required] Zorunlu alan Boş bırakılamaz
[StringLength] Metin uzunluğu Min/Max karakter
[Range] Sayı aralığı 1-100 arası
[EmailAddress] E-posta formatı xxx@xxx.com
[Phone] Telefon formatı Rakam kontrolü
[Display] Görünen isim Label metni
[Compare] Alan karşılaştırma Şifre onayı

Örnek: Ürün Modeli

Models/Product.cs
using System.ComponentModel.DataAnnotations;
using System.ComponentModel.DataAnnotations.Schema;

public class Product
{
    public int Id { get; set; }
    
    [Required(ErrorMessage = "Ürün adı gereklidir")]
    [StringLength(100)]
    [Display(Name = "Ürün Adı")]
    public string Name { get; set; }
    
    [Required]
    [Range(0.01, 99999.99, ErrorMessage = "Fiyat 0.01-99999.99 arasında olmalı")]
    [Column(TypeName = "decimal(18,2)")]
    [Display(Name = "Fiyat (TL)")]
    public decimal Price { get; set; }
    
    [StringLength(500)]
    [Display(Name = "Açıklama")]
    public string Description { get; set; }
    
    [Required]
    [Range(0, 10000, ErrorMessage = "Stok miktarını kontrol edin")]
    [Display(Name = "Stok Miktarı")]
    public int Stock { get; set; }
    
    [Display(Name = "Aktif mi?")]
    public bool IsActive { get; set; } = true;
    
    [DataType(DataType.Date)]
    [Display(Name = "Eklenme Tarihi")]
    public DateTime CreatedAt { get; set; } = DateTime.Now;
}

Örnek: Kullanıcı Kayıt Modeli

Models/RegisterViewModel.cs
public class RegisterViewModel
{
    [Required(ErrorMessage = "Kullanıcı adı gerekli")]
    [StringLength(30, MinimumLength = 3)]
    [Display(Name = "Kullanıcı Adı")]
    public string Username { get; set; }
    
    [Required(ErrorMessage = "E-posta gerekli")]
    [EmailAddress]
    public string Email { get; set; }
    
    [Required(ErrorMessage = "Şifre gerekli")]
    [StringLength(100, MinimumLength = 6, 
        ErrorMessage = "Şifre en az 6 karakter olmalı")]
    [DataType(DataType.Password)]
    [Display(Name = "Şifre")]
    public string Password { get; set; }
    
    [Required(ErrorMessage = "Şifre onayı gerekli")]
    [DataType(DataType.Password)]
    [Compare("Password", ErrorMessage = "Şifreler eşleşmiyor")]
    [Display(Name = "Şifre Onayı")]
    public string ConfirmPassword { get; set; }
    
    [Range(typeof(bool), "true", "true", 
        ErrorMessage = "Şartları kabul etmelisiniz")]
    [Display(Name = "Kullanım şartlarını kabul ediyorum")]
    public bool AcceptTerms { get; set; }
}

Özel Validation Attribute

Yaş Kontrolü - Custom Attribute
public class MinimumAgeAttribute : ValidationAttribute
{
    private readonly int _minimumAge;
    
    public MinimumAgeAttribute(int minimumAge)
    {
        _minimumAge = minimumAge;
    }
    
    protected override ValidationResult IsValid(
        object value, ValidationContext context)
    {
        if (value is DateTime birthDate)
        {
            var age = DateTime.Today.Year - birthDate.Year;
            if (age >= _minimumAge)
            {
                return ValidationResult.Success;
            }
        }
        return new ValidationResult(
            $"Yaşınız en az {_minimumAge} olmalıdır");
    }
}

// Kullanımı:
public class User
{
    [MinimumAge(18)]
    [DataType(DataType.Date)]
    public DateTime BirthDate { get; set; }
}

1 "Book" modeli için: Title (zorunlu, 5-100 karakter), Author (zorunlu), Price (0.01-9999), ISBN yazın. 
public class Book
{
    public int Id { get; set; }
    
    [Required(ErrorMessage = "Kitap adı zorunludur")]
    [StringLength(100, MinimumLength = 5)]
    [Display(Name = "Kitap Adı")]
    public string Title { get; set; }
    
    [Required(ErrorMessage = "Yazar adı zorunludur")]
    [Display(Name = "Yazar")]
    public string Author { get; set; }
    
    [Range(0.01, 9999)]
    [Display(Name = "Fiyat")]
    public decimal Price { get; set; }
    
    [StringLength(13)]
    public string ISBN { get; set; }
}
2 Şifre ve şifre onayı alanları olan LoginViewModel yazın (Compare kullanarak). 
public class ChangePasswordViewModel
{
    [Required(ErrorMessage = "Mevcut şifre gerekli")]
    [DataType(DataType.Password)]
    [Display(Name = "Mevcut Şifre")]
    public string CurrentPassword { get; set; }
    
    [Required]
    [StringLength(100, MinimumLength = 6)]
    [DataType(DataType.Password)]
    [Display(Name = "Yeni Şifre")]
    public string NewPassword { get; set; }
    
    [DataType(DataType.Password)]
    [Compare("NewPassword", ErrorMessage = "Şifreler eşleşmiyor")]
    [Display(Name = "Şifre Onayı")]
    public string ConfirmPassword { get; set; }
}
3 ModelState.IsValid kontrolü yapan Create action'ı yazın. 
[HttpPost]
public IActionResult Create(Book book)
{
    if (ModelState.IsValid)
    {
        // Kaydet
        _context.Books.Add(book);
        _context.SaveChanges();
        
        TempData["Success"] = "Kitap başarıyla eklendi!";
        return RedirectToAction("Index");
    }
    
    // Validation hataları varsa formu göster
    return View(book);
}

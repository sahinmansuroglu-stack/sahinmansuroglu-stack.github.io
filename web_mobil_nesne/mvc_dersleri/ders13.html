Arama ve Filtreleme

Controller - Arama ve Filtre
public async Task<IActionResult> Index(
    string search, 
    int? classroomId,
    bool? isActive,
    string sortBy = "name")
{
    var query = _context.Students
        .Include(s => s.Classroom)
        .AsQueryable();
    
    // Metin araması
    if (!string.IsNullOrEmpty(search))
    {
        search = search.ToLower();
        query = query.Where(s => 
            s.FirstName.ToLower().Contains(search) || 
            s.LastName.ToLower().Contains(search) ||
            s.Email.ToLower().Contains(search));
    }
    
    // Sınıf filtresi
    if (classroomId.HasValue)
    {
        query = query.Where(s => s.ClassroomId == classroomId);
    }
    
    // Aktiflik filtresi
    if (isActive.HasValue)
    {
        query = query.Where(s => s.IsActive == isActive);
    }
    
    // Sıralama
    query = sortBy switch
    {
        "name" => query.OrderBy(s => s.LastName).ThenBy(s => s.FirstName),
        "name_desc" => query.OrderByDescending(s => s.LastName),
        "classroom" => query.OrderBy(s => s.Classroom.Name),
        "date" => query.OrderByDescending(s => s.Id),
        _ => query.OrderBy(s => s.LastName)
    };
    
    // ViewBag değerleri
    ViewBag.Search = search;
    ViewBag.ClassroomId = classroomId;
    ViewBag.IsActive = isActive;
    ViewBag.SortBy = sortBy;
    ViewBag.Classrooms = new SelectList(
        await _context.Classrooms.ToListAsync(), 
        "Id", "Name", classroomId);
    
    return View(await query.ToListAsync());
}

Sayfalama (Pagination)

ViewModel/PagedResult.cs
public class PagedResult<T>
{
    public List<T> Items { get; set; }
    public int CurrentPage { get; set; }
    public int TotalPages { get; set; }
    public int PageSize { get; set; }
    public int TotalCount { get; set; }
    
    public bool HasPrevious => CurrentPage > 1;
    public bool HasNext => CurrentPage < TotalPages;
}
Controller - Sayfalama
private const int PageSize = 10;

public async Task<IActionResult> Index(
    string search, 
    int? classroomId,
    int page = 1)
{
    var query = _context.Students
        .Include(s => s.Classroom)
        .AsQueryable();
    
    // Filtreler...
    if (!string.IsNullOrEmpty(search))
    {
        query = query.Where(s => s.FirstName.Contains(search) || 
                                 s.LastName.Contains(search));
    }
    
    // Toplam kayıt sayısı
    var totalCount = await query.CountAsync();
    var totalPages = (int)Math.Ceiling(totalCount / (double)PageSize);
    
    // Sayfalama uygula
    var students = await query
        .OrderBy(s => s.LastName)
        .Skip((page - 1) * PageSize)
        .Take(PageSize)
        .ToListAsync();
    
    var result = new PagedResult<Student>
    {
        Items = students,
        CurrentPage = page,
        TotalPages = totalPages,
        PageSize = PageSize,
        TotalCount = totalCount
    };
    
    ViewBag.Search = search;
    ViewBag.ClassroomId = classroomId;
    
    return View(result);
}

Sayfalama Bileşeni (View)

Pagination Component
@model PagedResult<Student>

<!-- Sonuç bilgisi -->
<p class="text-muted">
    Toplam @Model.TotalCount kayıt, Sayfa @Model.CurrentPage / @Model.TotalPages
</p>

<!-- Sayfalama -->
@if (Model.TotalPages > 1)
{
    <nav>
        <ul class="pagination justify-content-center">
            <!-- Önceki -->
            <li class="page-item @(!Model.HasPrevious ? "disabled" : "")">
                <a class="page-link" 
                   href="?page=@(Model.CurrentPage - 1)&search=@ViewBag.Search&classroomId=@ViewBag.ClassroomId">
                    « Önceki
                </a>
            </li>
            
            <!-- Sayfa numaraları -->
            @for (int i = 1; i <= Model.TotalPages; i++)
            {
                <li class="page-item @(i == Model.CurrentPage ? "active" : "")">
                    <a class="page-link" 
                       href="?page=@i&search=@ViewBag.Search&classroomId=@ViewBag.ClassroomId">
                        @i
                    </a>
                </li>
            }
            
            <!-- Sonraki -->
            <li class="page-item @(!Model.HasNext ? "disabled" : "")">
                <a class="page-link" 
                   href="?page=@(Model.CurrentPage + 1)&search=@ViewBag.Search&classroomId=@ViewBag.ClassroomId">
                    Sonraki »
                </a>
            </li>
        </ul>
    </nav>
}

Models

Models/Employee.cs
public class Employee
{
    public int Id { get; set; }
    
    [Required, Display(Name = "Ad")]
    public string FirstName { get; set; }
    
    [Required, Display(Name = "Soyad")]
    public string LastName { get; set; }
    
    [EmailAddress, Display(Name = "E-posta")]
    public string Email { get; set; }
    
    [Display(Name = "Telefon")]
    public string? Phone { get; set; }
    
    [Display(Name = "İşe Başlama")]
    public DateTime HireDate { get; set; }
    
    [Display(Name = "Maaş")]
    [Range(0, 1000000)]
    public decimal Salary { get; set; }
    
    public bool IsActive { get; set; } = true;
    
    // Navigation
    public int DepartmentId { get; set; }
    public Department Department { get; set; }
    
    public int? ManagerId { get; set; }
    public Employee? Manager { get; set; }
    
    // Computed
    public string FullName => $"{FirstName} {LastName}";
}

public class Department
{
    public int Id { get; set; }
    public string Name { get; set; }
    public ICollection<Employee> Employees { get; set; }
}

Controller

Controllers/EmployeeController.cs
public class EmployeeController : Controller
{
    private readonly HRDbContext _context;
    
    public EmployeeController(HRDbContext context)
    {
        _context = context;
    }
    
    public async Task<IActionResult> Index(string search, int? departmentId)
    {
        var query = _context.Employees
            .Include(e => e.Department)
            .Where(e => e.IsActive)
            .AsQueryable();
        
        if (!string.IsNullOrEmpty(search))
        {
            query = query.Where(e => 
                e.FirstName.Contains(search) || 
                e.LastName.Contains(search));
        }
        
        if (departmentId.HasValue)
        {
            query = query.Where(e => e.DepartmentId == departmentId);
        }
        
        ViewBag.Departments = new SelectList(
            await _context.Departments.ToListAsync(), 
            "Id", "Name", departmentId);
        ViewBag.Search = search;
        ViewBag.DepartmentId = departmentId;
        
        return View(await query.ToListAsync());
    }
    
    public async Task<IActionResult> Details(int id)
    {
        var employee = await _context.Employees
            .Include(e => e.Department)
            .Include(e => e.Manager)
            .FirstOrDefaultAsync(e => e.Id == id);
        
        if (employee == null) return NotFound();
        
        // Ekip üyeleri
        ViewBag.TeamMembers = await _context.Employees
            .Where(e => e.ManagerId == id && e.IsActive)
            .ToListAsync();
        
        return View(employee);
    }
    
    [HttpGet]
    public async Task<IActionResult> Create()
    {
        await PopulateDropdownsAsync();
        return View();
    }
    
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Create(Employee employee)
    {
        if (ModelState.IsValid)
        {
            _context.Employees.Add(employee);
            await _context.SaveChangesAsync();
            
            TempData["Success"] = $"{employee.FullName} personel olarak eklendi!";
            return RedirectToAction(nameof(Index));
        }
        
        await PopulateDropdownsAsync(employee);
        return View(employee);
    }
    
    [HttpGet]
    public async Task<IActionResult> Edit(int id)
    {
        var employee = await _context.Employees.FindAsync(id);
        if (employee == null) return NotFound();
        
        await PopulateDropdownsAsync(employee);
        return View(employee);
    }
    
    [HttpPost]
    [ValidateAntiForgeryToken]
    public async Task<IActionResult> Edit(int id, Employee employee)
    {
        if (id != employee.Id) return NotFound();
        
        if (ModelState.IsValid)
        {
            _context.Update(employee);
            await _context.SaveChangesAsync();
            
            TempData["Success"] = "Personel bilgileri güncellendi!";
            return RedirectToAction(nameof(Index));
        }
        
        await PopulateDropdownsAsync(employee);
        return View(employee);
    }
    
    private async Task PopulateDropdownsAsync(Employee? employee = null)
    {
        ViewBag.Departments = new SelectList(
            await _context.Departments.ToListAsync(),
            "Id", "Name", employee?.DepartmentId);
        
        ViewBag.Managers = new SelectList(
            await _context.Employees
                .Where(e => e.IsActive && e.Id != employee?.Id)
                .Select(e => new { e.Id, Name = e.FirstName + " " + e.LastName })
                .ToListAsync(),
            "Id", "Name", employee?.ManagerId);
    }
}

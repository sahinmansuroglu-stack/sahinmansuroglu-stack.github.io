Servis

Services/MovieService.cs
public class MovieService : IMovieService
{
    private readonly MovieDbContext _context;
    
    public async Task<PagedResult<Movie>> SearchAsync(MovieFilterViewModel filter)
    {
        var query = _context.Movies
            .Include(m => m.Genre)
            .Include(m => m.Director)
            .AsQueryable();
        
        // Metin araması
        if (!string.IsNullOrEmpty(filter.Search))
        {
            var searchLower = filter.Search.ToLower();
            query = query.Where(m => 
                m.Title.ToLower().Contains(searchLower) ||
                m.OriginalTitle.ToLower().Contains(searchLower) ||
                m.Director.Name.ToLower().Contains(searchLower));
        }
        
        // Tür filtresi
        if (filter.GenreId.HasValue)
        {
            query = query.Where(m => m.GenreId == filter.GenreId);
        }
        
        // Yıl aralığı
        if (filter.YearFrom.HasValue)
        {
            query = query.Where(m => m.Year >= filter.YearFrom);
        }
        if (filter.YearTo.HasValue)
        {
            query = query.Where(m => m.Year <= filter.YearTo);
        }
        
        // Minimum puan
        if (filter.MinRating.HasValue)
        {
            query = query.Where(m => m.Rating >= filter.MinRating);
        }
        
        // Sıralama
        query = filter.SortBy switch
        {
            "title" => query.OrderBy(m => m.Title),
            "title_desc" => query.OrderByDescending(m => m.Title),
            "year" => query.OrderBy(m => m.Year),
            "year_desc" => query.OrderByDescending(m => m.Year),
            "rating" => query.OrderBy(m => m.Rating),
            "rating_desc" => query.OrderByDescending(m => m.Rating),
            _ => query.OrderByDescending(m => m.Rating)
        };
        
        // Toplam sayı
        var totalCount = await query.CountAsync();
        
        // Sayfalama
        var movies = await query
            .Skip((filter.Page - 1) * filter.PageSize)
            .Take(filter.PageSize)
            .ToListAsync();
        
        return new PagedResult<Movie>
        {
            Items = movies,
            CurrentPage = filter.Page,
            PageSize = filter.PageSize,
            TotalCount = totalCount,
            TotalPages = (int)Math.Ceiling(totalCount / (double)filter.PageSize)
        };
    }
}

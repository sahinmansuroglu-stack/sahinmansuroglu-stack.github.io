ProductController

Controllers/ProductController.cs
using Microsoft.AspNetCore.Mvc;

public class ProductController : Controller
{
    // Statik ürün listesi (demo amaçlı)
    private static List<Product> _products = new()
    {
        new Product { Id = 1, Name = "Laptop", Category = "Elektronik", Price = 15000 },
        new Product { Id = 2, Name = "Telefon", Category = "Elektronik", Price = 8000 },
        new Product { Id = 3, Name = "Kulaklık", Category = "Aksesuar", Price = 500 },
        new Product { Id = 4, Name = "Mouse", Category = "Aksesuar", Price = 200 }
    };
    
    // GET: /Product
    public IActionResult Index()
    {
        return View(_products);
    }
    
    // GET: /Product/Details/3
    public IActionResult Details(int id)
    {
        var product = _products.FirstOrDefault(p => p.Id == id);
        
        if (product == null)
        {
            return NotFound();
        }
        
        return View(product);
    }
    
    // GET: /Product/Category/Elektronik
    [Route("Product/Category/{categoryName}")]
    public IActionResult ByCategory(string categoryName)
    {
        var filtered = _products
            .Where(p => p.Category.ToLower() == categoryName.ToLower())
            .ToList();
        
        ViewBag.Category = categoryName;
        return View(filtered);
    }
    
    // GET: /Product/Search?q=laptop&minPrice=1000
    public IActionResult Search(string q, decimal? minPrice, decimal? maxPrice)
    {
        var query = _products.AsQueryable();
        
        if (!string.IsNullOrEmpty(q))
        {
            query = query.Where(p => p.Name.ToLower().Contains(q.ToLower()));
        }
        
        if (minPrice.HasValue)
        {
            query = query.Where(p => p.Price >= minPrice);
        }
        
        if (maxPrice.HasValue)
        {
            query = query.Where(p => p.Price <= maxPrice);
        }
        
        ViewBag.SearchTerm = q;
        ViewBag.MinPrice = minPrice;
        ViewBag.MaxPrice = maxPrice;
        
        return View(query.ToList());
    }
}

ViewComponent

ViewComponents/StatCardViewComponent.cs
public class StatCardViewComponent : ViewComponent
{
    private readonly ShopDbContext _context;
    
    public StatCardViewComponent(ShopDbContext context)
    {
        _context = context;
    }
    
    public async Task<IViewComponentResult> InvokeAsync(string type)
    {
        var model = new StatCardModel();
        
        switch (type)
        {
            case "orders":
                model.Title = "Toplam Sipariş";
                model.Value = (await _context.Orders.CountAsync()).ToString();
                model.Icon = "🧾";
                model.Color = "primary";
                break;
                
            case "revenue":
                var revenue = await _context.Orders
                    .Where(o => o.Status == "Completed")
                    .SumAsync(o => o.TotalAmount);
                model.Title = "Toplam Gelir";
                model.Value = revenue.ToString("C");
                model.Icon = "💰";
                model.Color = "success";
                break;
                
            case "products":
                model.Title = "Ürün Sayısı";
                model.Value = (await _context.Products.CountAsync()).ToString();
                model.Icon = "📦";
                model.Color = "info";
                break;
                
            case "customers":
                model.Title = "Müşteri Sayısı";
                model.Value = (await _context.Customers.CountAsync()).ToString();
                model.Icon = "👥";
                model.Color = "warning";
                break;
        }
        
        return View(model);
    }
}

Partial View

Views/Shared/_RecentOrders.cshtml
@model List<Order>

<table class="table table-sm">
    <thead>
        <tr>
            <th>#</th>
            <th>Müşteri</th>
            <th>Tutar</th>
            <th>Durum</th>
        </tr>
    </thead>
    <tbody>
        @foreach (var order in Model)
        {
            <tr>
                <td>#@order.Id</td>
                <td>@order.Customer.Name</td>
                <td>@order.TotalAmount.ToString("C")</td>
                <td><span class="badge bg-@GetStatusColor(order.Status)">@order.Status</span></td>
            </tr>
        }
    </tbody>
</table>

@functions {
    string GetStatusColor(string status) => status switch
    {
        "Pending" => "warning",
        "Processing" => "info",
        "Completed" => "success",
        "Cancelled" => "danger",
        _ => "secondary"
    };
}

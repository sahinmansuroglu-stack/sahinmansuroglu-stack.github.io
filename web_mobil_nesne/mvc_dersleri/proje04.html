📊 Dinamik Anket Sayfası

Ders 1-4 Konularını Kapsayan Proje

Proje Hakkında

Razor syntax ile dinamik anket formu ve sonuç gösterimi.

📚 Kapsanan Konular:
• Razor syntax (@if, @foreach, @for)
• Tag Helpers (asp-for, asp-action)
• Koşullu render
• Strongly-typed View

Model

Models/Survey.cs
public class Survey
{
    public int Id { get; set; }
    public string Title { get; set; }
    public List<Question> Questions { get; set; } = new();
}

public class Question
{
    public int Id { get; set; }
    public string Text { get; set; }
    public QuestionType Type { get; set; }
    public List<string> Options { get; set; } = new();
}

public enum QuestionType
{
    Text,
    SingleChoice,
    MultipleChoice
}

Controller

Controllers/SurveyController.cs
public class SurveyController : Controller
{
    private static Survey _survey = new()
    {
        Id = 1,
        Title = "Yazılım Tercihleri Anketi",
        Questions = new List<Question>
        {
            new() { Id = 1, Text = "Adınız", Type = QuestionType.Text },
            new() { 
                Id = 2, 
                Text = "Favori programlama diliniz?", 
                Type = QuestionType.SingleChoice,
                Options = new() { "C#", "Python", "JavaScript", "Java" }
            },
            new() { 
                Id = 3, 
                Text = "Hangi teknolojileri kullanıyorsunuz?", 
                Type = QuestionType.MultipleChoice,
                Options = new() { "ASP.NET", "React", "Angular", "Vue" }
            }
        }
    };
    
    public IActionResult Index()
    {
        return View(_survey);
    }
    
    [HttpPost]
    public IActionResult Submit(Dictionary<string, string> answers)
    {
        ViewBag.Answers = answers;
        ViewBag.Survey = _survey;
        return View("Result");
    }
}

View - Anket Formu

Views/Survey/Index.cshtml
@model Survey

<h1>📊 @Model.Title</h1>

<form asp-action="Submit" method="post">
    @for (int i = 0; i < Model.Questions.Count; i++)
    {
        var question = Model.Questions[i];
        
        <div class="card mb-3">
            <div class="card-body">
                <h5>Soru @(i + 1): @question.Text</h5>
                
                @switch (question.Type)
                {
                    case QuestionType.Text:
                        <input type="text" name="answers[@question.Id]" 
                               class="form-control" />
                        break;
                        
                    case QuestionType.SingleChoice:
                        @foreach (var option in question.Options)
                        {
                            <div class="form-check">
                                <input type="radio" name="answers[@question.Id]" 
                                       value="@option" class="form-check-input" />
                                <label class="form-check-label">@option</label>
                            </div>
                        }
                        break;
                        
                    case QuestionType.MultipleChoice:
                        @foreach (var option in question.Options)
                        {
                            <div class="form-check">
                                <input type="checkbox" name="answers[@question.Id]" 
                                       value="@option" class="form-check-input" />
                                <label class="form-check-label">@option</label>
                            </div>
                        }
                        break;
                }
            </div>
        </div>
    }
    
    <button type="submit" class="btn btn-success btn-lg">Gönder</button>
</form>

View - Sonuç

Views/Survey/Result.cshtml
@{
    var answers = ViewBag.Answers as Dictionary<string, string>;
    var survey = ViewBag.Survey as Survey;
}

<h1>✅ Teşekkürler!</h1>
<p>Anket yanıtlarınız başarıyla kaydedildi.</p>

<div class="card">
    <div class="card-header">Yanıtlarınız</div>
    <div class="card-body">
        @foreach (var question in survey.Questions)
        {
            <p>
                <strong>@question.Text:</strong>
                @if (answers.ContainsKey(question.Id.ToString()))
                {
                    <span class="text-success">@answers[question.Id.ToString()]</span>
                }
                else
                {
                    <span class="text-muted">Cevaplanmadı</span>
                }
            </p>
        }
    </div>
</div>

İlişkisel Tablolar

One-to-Many ve Many-to-Many İlişkiler

Tablo İlişkileri

📊 İlişki Türleri:
• One-to-Many (1:N): Bir sınıfta birçok öğrenci
• Many-to-Many (N:M): Bir öğrenci birçok ders alabilir, bir dersi birçok öğrenci alabilir
• One-to-One (1:1): Bir öğrencinin bir profili var

One-to-Many İlişki

Models/Classroom.cs (Parent)
public class Classroom
{
    public int Id { get; set; }
    public string Name { get; set; }
    
    // Navigation Property - Bir sınıfta birçok öğrenci
    public ICollection<Student> Students { get; set; }
}
Models/Student.cs (Child)
public class Student
{
    public int Id { get; set; }
    public string FirstName { get; set; }
    public string LastName { get; set; }
    
    // Foreign Key
    public int ClassroomId { get; set; }
    
    // Navigation Property
    public Classroom Classroom { get; set; }
}

Many-to-Many İlişki

Models/Course.cs
public class Course
{
    public int Id { get; set; }
    
    [Required]
    public string Name { get; set; }
    
    public int Credit { get; set; }
    
    // Many-to-Many Navigation
    public ICollection<StudentCourse> StudentCourses { get; set; }
}
Models/StudentCourse.cs (Ara Tablo)
public class StudentCourse
{
    public int StudentId { get; set; }
    public Student Student { get; set; }
    
    public int CourseId { get; set; }
    public Course Course { get; set; }
    
    // Ek alanlar
    public DateTime EnrollmentDate { get; set; }
    public int? Grade { get; set; }
}
DbContext - OnModelCreating
protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    // Composite Primary Key
    modelBuilder.Entity<StudentCourse>()
        .HasKey(sc => new { sc.StudentId, sc.CourseId });
    
    // İlişkileri tanımla
    modelBuilder.Entity<StudentCourse>()
        .HasOne(sc => sc.Student)
        .WithMany(s => s.StudentCourses)
        .HasForeignKey(sc => sc.StudentId);
    
    modelBuilder.Entity<StudentCourse>()
        .HasOne(sc => sc.Course)
        .WithMany(c => c.StudentCourses)
        .HasForeignKey(sc => sc.CourseId);
}

Ders Kayıt Formu

EnrollController.cs
public class EnrollController : Controller
{
    private readonly SchoolDbContext _context;
    
    [HttpGet]
    public async Task<IActionResult> Create(int studentId)
    {
        ViewBag.StudentId = studentId;
        ViewBag.Courses = new SelectList(
            await _context.Courses.ToListAsync(), 
            "Id", "Name");
        return View();
    }
    
    [HttpPost]
    public async Task<IActionResult> Create(int studentId, int courseId)
    {
        // Zaten kayıtlı mı kontrol et
        var exists = await _context.StudentCourses
            .AnyAsync(sc => sc.StudentId == studentId && sc.CourseId == courseId);
        
        if (exists)
        {
            TempData["Error"] = "Öğrenci bu derse zaten kayıtlı!";
            return RedirectToAction("Details", "Student", new { id = studentId });
        }
        
        var enrollment = new StudentCourse
        {
            StudentId = studentId,
            CourseId = courseId,
            EnrollmentDate = DateTime.Now
        };
        
        _context.StudentCourses.Add(enrollment);
        await _context.SaveChangesAsync();
        
        TempData["Success"] = "Ders kaydı yapıldı!";
        return RedirectToAction("Details", "Student", new { id = studentId });
    }
}

2 Ders silme işleminde ilişkili kayıtları da silin. 
[HttpPost]
public async Task<IActionResult> DeleteCourse(int id)
{
    var course = await _context.Courses
        .Include(c => c.StudentCourses)
        .FirstOrDefaultAsync(c => c.Id == id);
    
    if (course != null)
    {
        // Önce ilişkili kayıtları sil
        _context.StudentCourses.RemoveRange(course.StudentCourses);
        // Sonra dersi sil
        _context.Courses.Remove(course);
        await _context.SaveChangesAsync();
    }
    return RedirectToAction(nameof(Index));
}

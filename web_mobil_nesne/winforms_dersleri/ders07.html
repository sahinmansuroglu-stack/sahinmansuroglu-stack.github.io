MessageBox Nedir?

MessageBox, kullanıcıya bilgi, uyarı, hata mesajları göstermek veya onay almak için kullanılan modal dialog kutusudur.

📌 Modal Dialog: Kullanıcı mesaj kutusunu kapatmadan uygulamaya devam edemez.

1. Basit Mesaj

En Basit Kullanım
// Sadece mesaj
MessageBox.Show("İşlem başarılı!");

// Mesaj ve başlık
MessageBox.Show("Dosya kaydedildi.", "Bilgi");

2. Butonlar (MessageBoxButtons)

Değer Butonlar
OK Tamam
OKCancel Tamam, İptal
YesNo Evet, Hayır
YesNoCancel Evet, Hayır, İptal
RetryCancel Yeniden Dene, İptal
AbortRetryIgnore Durdur, Yeniden Dene, Yoksay
Butonlu Mesajlar
// Tamam / İptal
MessageBox.Show("Devam edilsin mi?", "Onay", 
    MessageBoxButtons.OKCancel);

// Evet / Hayır
MessageBox.Show("Silmek istiyor musunuz?", "Silme", 
    MessageBoxButtons.YesNo);

// Evet / Hayır / İptal
MessageBox.Show("Değişiklikleri kaydet?", "Kaydet", 
    MessageBoxButtons.YesNoCancel);

3. İkonlar (MessageBoxIcon)

Değer Görünüm Kullanım
Information ℹ️ Mavi bilgi Bilgilendirme
Warning ⚠️ Sarı uyarı Dikkat gerektiren durumlar
Error ❌ Kırmızı hata Hatalar
Question ❓ Soru işareti Kullanıcıya soru
None İkonsuz Basit mesajlar
İkonlu Mesajlar
// Bilgi mesajı
MessageBox.Show("Kayıt başarılı!", "Bilgi", 
    MessageBoxButtons.OK, MessageBoxIcon.Information);

// Uyarı mesajı
MessageBox.Show("Bu işlem geri alınamaz!", "Uyarı", 
    MessageBoxButtons.OK, MessageBoxIcon.Warning);

// Hata mesajı
MessageBox.Show("Dosya bulunamadı!", "Hata", 
    MessageBoxButtons.OK, MessageBoxIcon.Error);

// Soru
MessageBox.Show("Devam etmek istiyor musunuz?", "Onay", 
    MessageBoxButtons.YesNo, MessageBoxIcon.Question);

4. Cevabı Kontrol Etme (DialogResult)

Kullanıcının hangi butona tıkladığını DialogResult ile öğrenebilirsiniz.

Değer Açıklama
DialogResult.OK Tamam tıklandı
DialogResult.Cancel İptal tıklandı
DialogResult.Yes Evet tıklandı
DialogResult.No Hayır tıklandı
DialogResult.Retry Yeniden Dene tıklandı
Cevaba Göre İşlem
// Silme onayı
DialogResult sonuc = MessageBox.Show(
    "Bu kaydı silmek istediğinizden emin misiniz?",
    "Silme Onayı",
    MessageBoxButtons.YesNo,
    MessageBoxIcon.Question);

if (sonuc == DialogResult.Yes)
{
    // Silme işlemi
    listBox1.Items.RemoveAt(listBox1.SelectedIndex);
    MessageBox.Show("Kayıt silindi.", "Bilgi");
}
else
{
    MessageBox.Show("Silme işlemi iptal edildi.", "İptal");
}
Kaydetme - Üç Seçenek
private void Form1_FormClosing(object sender, FormClosingEventArgs e)
{
    if (degisiklikVar)
    {
        DialogResult sonuc = MessageBox.Show(
            "Değişiklikler kaydedilsin mi?",
            "Kaydet",
            MessageBoxButtons.YesNoCancel,
            MessageBoxIcon.Question);
        
        switch (sonuc)
        {
            case DialogResult.Yes:
                Kaydet();
                break;
            case DialogResult.No:
                // Kaydetmeden çık
                break;
            case DialogResult.Cancel:
                e.Cancel = true; // Kapanmayı iptal et
                break;
        }
    }
}

5. Varsayılan Buton

Enter'a basıldığında hangi butonun seçileceğini belirler.

Varsayılan Buton Ayarlama
// "Hayır" varsayılan olsun (yanlışlıkla silmeyi engeller)
DialogResult sonuc = MessageBox.Show(
    "Dosyayı silmek istiyor musunuz?",
    "Silme",
    MessageBoxButtons.YesNo,
    MessageBoxIcon.Warning,
    MessageBoxDefaultButton.Button2); // "Hayır" butonu

// Button1 = Evet, Button2 = Hayır, Button3 = İptal

Pratik Örnekler

Giriş Doğrulama
private void btnGiris_Click(object sender, EventArgs e)
{
    string kullanici = txtKullanici.Text;
    string sifre = txtSifre.Text;
    
    if (string.IsNullOrWhiteSpace(kullanici))
    {
        MessageBox.Show("Kullanıcı adı boş olamaz!", "Hata",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
        txtKullanici.Focus();
        return;
    }
    
    if (sifre.Length < 6)
    {
        MessageBox.Show("Şifre en az 6 karakter olmalı!", "Hata",
            MessageBoxButtons.OK, MessageBoxIcon.Warning);
        txtSifre.Focus();
        return;
    }
    
    // Giriş kontrolü
    if (kullanici == "admin" && sifre == "123456")
    {
        MessageBox.Show("Giriş başarılı!", "Hoş Geldiniz",
            MessageBoxButtons.OK, MessageBoxIcon.Information);
    }
    else
    {
        MessageBox.Show("Kullanıcı adı veya şifre hatalı!", "Hata",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
    }
}
Hata Yönetimi ile Birlikte
private void btnHesapla_Click(object sender, EventArgs e)
{
    try
    {
        int sayi1 = int.Parse(txtSayi1.Text);
        int sayi2 = int.Parse(txtSayi2.Text);
        
        if (sayi2 == 0)
        {
            MessageBox.Show("Sıfıra bölme yapılamaz!", "Hata",
                MessageBoxButtons.OK, MessageBoxIcon.Error);
            return;
        }
        
        double sonuc = (double)sayi1 / sayi2;
        lblSonuc.Text = sonuc.ToString("F2");
    }
    catch (FormatException)
    {
        MessageBox.Show("Lütfen geçerli sayılar girin!", "Format Hatası",
            MessageBoxButtons.OK, MessageBoxIcon.Warning);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Beklenmeyen hata: " + ex.Message, "Hata",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
    }
}

{
    MessageBox.Show("Hoş geldiniz!", "Karşılama",
        MessageBoxButtons.OK, MessageBoxIcon.Information);
}
2 Silme işlemi için Evet/Hayır onayı isteyen ve sonuca göre işlem yapan kod yazın. 
private void btnSil_Click(object sender, EventArgs e)
{
    DialogResult sonuc = MessageBox.Show(
        "Silmek istiyor musunuz?", "Onay",
        MessageBoxButtons.YesNo, MessageBoxIcon.Question);
    
    if (sonuc == DialogResult.Yes)
    {
        // Silme işlemi
        MessageBox.Show("Silindi!");
    }
}
3 Form kapanırken kaydetme hatırlatması yapan uygulama yazın. 
private void Form1_FormClosing(object sender, FormClosingEventArgs e)
{
    DialogResult sonuc = MessageBox.Show(
        "Çıkmadan önce kaydetmek ister misiniz?",
        "Çıkış",
        MessageBoxButtons.YesNoCancel,
        MessageBoxIcon.Question);
    
    if (sonuc == DialogResult.Yes)
        Kaydet();
    else if (sonuc == DialogResult.Cancel)
        e.Cancel = true;
}

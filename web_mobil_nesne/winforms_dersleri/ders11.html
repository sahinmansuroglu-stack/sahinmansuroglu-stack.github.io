PictureBox Nedir?

PictureBox, resimleri görüntülemek için kullanılan kontroldür. JPG, PNG, BMP, GIF gibi formatları destekler.

🖼️ Resim Alanı
Özellik Açıklama
Image Gösterilecek resim
SizeMode Resmin boyutlandırma modu
ImageLocation Resim dosya yolu veya URL
BackColor Arka plan rengi
BorderStyle Kenarlık stili
ErrorImage Hata durumunda gösterilecek resim
InitialImage Yüklenirken gösterilecek resim

SizeMode Ayarları

Değer Açıklama
Normal Resim orijinal boyutunda, sol üst köşeden başlar
StretchImage Resim PictureBox boyutuna esnetilir (oran bozulabilir)
AutoSize PictureBox resim boyutuna göre ayarlanır
CenterImage Resim ortada gösterilir
Zoom Oranı koruyarak sığdırır ✓ En çok kullanılan
SizeMode Kullanımı
// Zoom - Oranı koruyarak sığdır (önerilen)
pictureBox1.SizeMode = PictureBoxSizeMode.Zoom;

// StretchImage - Tam doldur (oran bozulabilir)
pictureBox1.SizeMode = PictureBoxSizeMode.StretchImage;

// CenterImage - Ortala
pictureBox1.SizeMode = PictureBoxSizeMode.CenterImage;

// AutoSize - PictureBox resim boyutuna uyar
pictureBox1.SizeMode = PictureBoxSizeMode.AutoSize;

Resim Yükleme Yöntemleri

Dosyadan Resim Yükleme
// Doğrudan dosya yolu
pictureBox1.Image = Image.FromFile(@"C:\Resimler\foto.jpg");

// OpenFileDialog ile
private void btnResimSec_Click(object sender, EventArgs e)
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Title = "Resim Seç";
    dialog.Filter = "Resim Dosyaları|*.jpg;*.jpeg;*.png;*.gif;*.bmp|Tüm Dosyalar|*.*";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        pictureBox1.Image = Image.FromFile(dialog.FileName);
        pictureBox1.SizeMode = PictureBoxSizeMode.Zoom;
    }
}
URL'den Resim Yükleme
// ImageLocation özelliği ile URL'den
pictureBox1.ImageLocation = "https://example.com/resim.jpg";
pictureBox1.Load(); // Asenkron yükle

// Veya LoadAsync metodu ile
pictureBox1.LoadAsync("https://example.com/resim.jpg");

// Yükleme tamamlandığında
private void pictureBox1_LoadCompleted(object sender, AsyncCompletedEventArgs e)
{
    if (e.Error != null)
    {
        MessageBox.Show("Resim yüklenemedi: " + e.Error.Message);
    }
    else
    {
        lblDurum.Text = "Resim yüklendi!";
    }
}
Resources'tan Resim Yükleme
// Proje kaynaklarından (Properties > Resources.resx)
pictureBox1.Image = Properties.Resources.logo;

// Embedded Resource olarak
pictureBox1.Image = Image.FromStream(
    Assembly.GetExecutingAssembly().GetManifestResourceStream("MyApp.Images.logo.png"));

private void btnTemizle_Click(object sender, EventArgs e)
{
    if (pictureBox1.Image != null)
    {
        pictureBox1.Image.Dispose(); // Belleği serbest bırak
        pictureBox1.Image = null;
    }

// Resmi değiştir (öncekini dispose et)
private void ResimYukle(string dosyaYolu)
{
    if (pictureBox1.Image != null)
    {
        pictureBox1.Image.Dispose();
    }
    
    pictureBox1.Image = Image.FromFile(dosyaYolu);
}

// Resim var mı kontrolü
if (pictureBox1.Image != null)
{
    MessageBox.Show("Resim yüklü");
}

Resim Kaydetme

Resmi Farklı Formatlarda Kaydetme
private void btnKaydet_Click(object sender, EventArgs e)
{
    if (pictureBox1.Image == null)
    {
        MessageBox.Show("Kaydedilecek resim yok!");
        return;
    }
    
    SaveFileDialog dialog = new SaveFileDialog();
    dialog.Filter = "PNG|*.png|JPEG|*.jpg|Bitmap|*.bmp|GIF|*.gif";
    dialog.DefaultExt = "png";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        string uzanti = Path.GetExtension(dialog.FileName).ToLower();
        ImageFormat format = ImageFormat.Png; // Varsayılan
        
        switch (uzanti)
        {
            case ".jpg":
            case ".jpeg":
                format = ImageFormat.Jpeg;
                break;
            case ".bmp":
                format = ImageFormat.Bmp;
                break;
            case ".gif":
                format = ImageFormat.Gif;
                break;
        }
        
        pictureBox1.Image.Save(dialog.FileName, format);
        MessageBox.Show("Resim kaydedildi!");
    }
}

PictureBox Olayları

Click ve DoubleClick Olayları
// Tıklandığında resim seç
private void pictureBox1_Click(object sender, EventArgs e)
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Filter = "Resimler|*.jpg;*.png;*.bmp";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        pictureBox1.Image = Image.FromFile(dialog.FileName);
    }
}

// Çift tıklandığında büyüt
private void pictureBox1_DoubleClick(object sender, EventArgs e)
{
    if (pictureBox1.Image != null)
    {
        Form buyukForm = new Form();
        buyukForm.Text = "Resim Önizleme";
        buyukForm.WindowState = FormWindowState.Maximized;
        
        PictureBox buyukPic = new PictureBox();
        buyukPic.Dock = DockStyle.Fill;
        buyukPic.SizeMode = PictureBoxSizeMode.Zoom;
        buyukPic.Image = pictureBox1.Image;
        buyukPic.Click += (s, args) => buyukForm.Close();
        
        buyukForm.Controls.Add(buyukPic);
        buyukForm.ShowDialog();
    }
}

Resim Galerisi Örneği

Basit Resim Galerisi
private string[] resimler;
private int mevcutIndex = 0;

// Klasör seç ve resimleri yükle
private void btnKlasorSec_Click(object sender, EventArgs e)
{
    FolderBrowserDialog dialog = new FolderBrowserDialog();
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        resimler = Directory.GetFiles(dialog.SelectedPath)
            .Where(f => f.EndsWith(".jpg") || f.EndsWith(".png") || f.EndsWith(".bmp"))
            .ToArray();
        
        if (resimler.Length > 0)
        {
            mevcutIndex = 0;
            ResimGoster();
        }
        else
        {
            MessageBox.Show("Klasörde resim bulunamadı!");
        }
    }
}

private void ResimGoster()
{
    if (resimler != null && resimler.Length > 0)
    {
        pictureBox1.Image = Image.FromFile(resimler[mevcutIndex]);
        lblDurum.Text = $"{mevcutIndex + 1} / {resimler.Length}";
    }
}

// Önceki resim
private void btnOnceki_Click(object sender, EventArgs e)
{
    if (resimler != null && mevcutIndex > 0)
    {
        mevcutIndex--;
        ResimGoster();
    }
}

// Sonraki resim
private void btnSonraki_Click(object sender, EventArgs e)
{
    if (resimler != null && mevcutIndex < resimler.Length - 1)
    {
        mevcutIndex++;
        ResimGoster();
    }
}

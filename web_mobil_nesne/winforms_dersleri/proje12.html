🍅 Pomodoro Zamanlayıcı

Proje Hakkında

Pomodoro tekniği ile çalışma/mola döngüsü yöneten zamanlayıcı. 25 dk çalışma, 5 dk mola.

📚 Yeni Konular (Ders 12): Timer, Interval, Tick olayı, Start/Stop

Kod Çözümü

Değişkenler ve Başlangıç
private int calismaSSure = 25 * 60; // 25 dakika (saniye)
private int molaSure = 5 * 60;      // 5 dakika
private int kalanSure;
private bool calismaModu = true;
private int tamamlananPomodoro = 0;

private void Form1_Load(object sender, EventArgs e)
{
    timer1.Interval = 1000;
    kalanSure = calismaSSure;
    SureyiGoster();
    
    // NumericUpDown ile süre ayarlama
    numCalisma.Value = 25;
    numMola.Value = 5;
}

private void SureyiGoster()
{
    int dakika = kalanSure / 60;
    int saniye = kalanSure % 60;
    lblSure.Text = $"{dakika:D2}:{saniye:D2}";
    
    // ProgressBar güncelle
    int toplamSure = calismaModu ? calismaSSure : molaSure;
    progressBar1.Value = (int)((double)(toplamSure - kalanSure) / toplamSure * 100);
}
Timer ve Kontroller
private void btnBaslat_Click(object sender, EventArgs e)
{
    timer1.Start();
    btnBaslat.Enabled = false;
    btnDuraklat.Enabled = true;
    lblDurum.Text = calismaModu ? "🎯 Çalışma Zamanı" : "☕ Mola Zamanı";
}

private void btnDuraklat_Click(object sender, EventArgs e)
{
    timer1.Stop();
    btnBaslat.Enabled = true;
    btnDuraklat.Enabled = false;
    lblDurum.Text = "⏸️ Duraklatıldı";
}

private void btnSifirla_Click(object sender, EventArgs e)
{
    timer1.Stop();
    calismaModu = true;
    kalanSure = calismaSSure;
    SureyiGoster();
    progressBar1.Value = 0;
    btnBaslat.Enabled = true;
    btnDuraklat.Enabled = false;
    lblDurum.Text = "⏹️ Hazır";
}

private void timer1_Tick(object sender, EventArgs e)
{
    kalanSure--;
    SureyiGoster();
    
    if (kalanSure <= 0)
    {
        timer1.Stop();
        
        if (calismaModu)
        {
            tamamlananPomodoro++;
            lblPomodoro.Text = $"🍅 x {tamamlananPomodoro}";
            
            MessageBox.Show("Çalışma tamamlandı!\nMola zamanı.", "Pomodoro",
                MessageBoxButtons.OK, MessageBoxIcon.Information);
            
            calismaModu = false;
            kalanSure = molaSure;
        }
        else
        {
            MessageBox.Show("Mola bitti!\nÇalışmaya devam.", "Pomodoro",
                MessageBoxButtons.OK, MessageBoxIcon.Information);
            
            calismaModu = true;
            kalanSure = calismaSSure;
        }
        
        SureyiGoster();
        progressBar1.Value = 0;
        btnBaslat.Enabled = true;
        btnDuraklat.Enabled = false;
    }
}
Süre Ayarlama
private void numCalisma_ValueChanged(object sender, EventArgs e)
{
    calismaSSure = (int)numCalisma.Value * 60;
    if (calismaModu && !timer1.Enabled)
    {
        kalanSure = calismaSSure;
        SureyiGoster();
    }
}

private void numMola_ValueChanged(object sender, EventArgs e)
{
    molaSure = (int)numMola.Value * 60;
}

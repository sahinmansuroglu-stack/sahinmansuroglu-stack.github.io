Kod Çözümü

Soru Yapısı ve Veriler
class Soru
{
    public string Metin { get; set; }
    public string[] Secenekler { get; set; }
    public int DogruIndex { get; set; }
}

private List<Soru> sorular = new List<Soru>
{
    new Soru { 
        Metin = "C#'ta string tanımlamak için hangi keyword kullanılır?",
        Secenekler = new[] { "int", "string", "bool", "char" },
        DogruIndex = 1
    },
    new Soru { 
        Metin = "Windows Forms'ta buton tıklaması hangi olayla yakalanır?",
        Secenekler = new[] { "TextChanged", "Click", "Load", "Shown" },
        DogruIndex = 1
    },
    new Soru { 
        Metin = "Hangi kontrol listeden tek seçim yapmayı sağlar?",
        Secenekler = new[] { "CheckBox", "TextBox", "RadioButton", "Label" },
        DogruIndex = 2
    }
};

private int mevcutIndex = 0;
private int puan = 0;
private int dogru = 0, yanlis = 0;
Soru Gösterme
private void Form1_Load(object sender, EventArgs e)
{
    progressBar1.Maximum = sorular.Count;
    SoruGoster();
}

private void SoruGoster()
{
    Soru s = sorular[mevcutIndex];
    
    lblSoruNo.Text = $"Soru {mevcutIndex + 1}/{sorular.Count}";
    lblSoru.Text = s.Metin;
    
    rbA.Text = "A) " + s.Secenekler[0];
    rbB.Text = "B) " + s.Secenekler[1];
    rbC.Text = "C) " + s.Secenekler[2];
    rbD.Text = "D) " + s.Secenekler[3];
    
    // Seçimleri temizle
    rbA.Checked = rbB.Checked = rbC.Checked = rbD.Checked = false;
    progressBar1.Value = mevcutIndex;
}
Cevap Kontrolü - MessageBox Kullanımı
private void btnCevapla_Click(object sender, EventArgs e)
{
    // Seçim kontrolü
    if (!rbA.Checked && !rbB.Checked && !rbC.Checked && !rbD.Checked)
    {
        MessageBox.Show("Lütfen bir seçenek işaretleyin!", "Uyarı",
            MessageBoxButtons.OK, MessageBoxIcon.Warning);
        return;
    }
    
    // Seçilen cevabı bul
    int secilen = rbA.Checked ? 0 : rbB.Checked ? 1 : rbC.Checked ? 2 : 3;
    Soru s = sorular[mevcutIndex];
    
    if (secilen == s.DogruIndex)
    {
        dogru++;
        puan += 10;
        MessageBox.Show("Doğru cevap! 🎉\n+10 puan", "Tebrikler",
            MessageBoxButtons.OK, MessageBoxIcon.Information);
    }
    else
    {
        yanlis++;
        string dogruCevap = s.Secenekler[s.DogruIndex];
        MessageBox.Show($"Yanlış! 😞\n\nDoğru cevap: {dogruCevap}", "Üzgünüm",
            MessageBoxButtons.OK, MessageBoxIcon.Error);
    }
    
    lblPuan.Text = "Puan: " + puan;
    
    // Sonraki soruya geç veya bitir
    mevcutIndex++;
    
    if (mevcutIndex >= sorular.Count)
    {
        QuizBitir();
    }
    else
    {
        SoruGoster();
    }
}

private void QuizBitir()
{
    progressBar1.Value = sorular.Count;
    
    double basari = (double)dogru / sorular.Count * 100;
    string mesaj = $@"
Quiz Tamamlandı!

Toplam Soru: {sorular.Count}
Doğru: {dogru}
Yanlış: {yanlis}
Puan: {puan}
Başarı: %{basari:N0}

Tekrar oynamak ister misiniz?";

    DialogResult sonuc = MessageBox.Show(mesaj, "Sonuçlar",
        MessageBoxButtons.YesNo, MessageBoxIcon.Question);
    
    if (sonuc == DialogResult.Yes)
    {
        // Sıfırla
        mevcutIndex = 0;
        puan = 0;
        dogru = 0;
        yanlis = 0;
        lblPuan.Text = "Puan: 0";
        SoruGoster();
    }
    else
    {
        this.Close();
    }
}

1. ColorDialog (Renk Seçme)

ColorDialog, kullanıcının renk paletinden renk seçmesini sağlar.

Özellik Açıklama
Color Seçilen renk
AllowFullOpen Özel renk seçimi aktif mi?
FullOpen Dialog açık/geniş görünümde başlasın mı?
AnyColor Herhangi bir renk seçilebilir mi?
SolidColorOnly Sadece katı renkler mi?
CustomColors Özel renk paleti
ColorDialog Kullanımı
// Form arka plan rengini değiştir
private void btnArkaPlanRenk_Click(object sender, EventArgs e)
{
    ColorDialog dialog = new ColorDialog();
    dialog.Color = this.BackColor; // Mevcut rengi göster
    dialog.AllowFullOpen = true;
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        this.BackColor = dialog.Color;
    }
}

// Label yazı rengini değiştir
private void btnYaziRenk_Click(object sender, EventArgs e)
{
    ColorDialog dialog = new ColorDialog();
    dialog.Color = lblMesaj.ForeColor;
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        lblMesaj.ForeColor = dialog.Color;
    }
}
Özel Renkler (Custom Colors)
// Özel renk paleti oluştur
ColorDialog dialog = new ColorDialog();

// Özel renkler (int dizisi olarak ARGB değerleri)
dialog.CustomColors = new int[] 
{
    ColorTranslator.ToOle(Color.FromArgb(255, 128, 0)),     // Turuncu
    ColorTranslator.ToOle(Color.FromArgb(128, 0, 128)),     // Mor
    ColorTranslator.ToOle(Color.FromArgb(0, 128, 128)),     // Deniz mavisi
};

if (dialog.ShowDialog() == DialogResult.OK)
{
    panel1.BackColor = dialog.Color;
    
    // RGB değerlerini göster
    lblRGB.Text = $"RGB: {dialog.Color.R}, {dialog.Color.G}, {dialog.Color.B}";
    lblHex.Text = $"HEX: #{dialog.Color.R:X2}{dialog.Color.G:X2}{dialog.Color.B:X2}";
}

2. FontDialog (Yazı Tipi Seçme)

FontDialog, kullanıcının yazı tipi, boyutu ve stilini seçmesini sağlar.

Arial 14pt  |  Times 16pt İtalik  |  Courier 12pt Kalın
Özellik Açıklama
Font Seçilen yazı tipi
Color Yazı rengi (ShowColor=true ise)
ShowColor Renk seçimi göster
ShowEffects Altı çizili, üstü çizili göster
MinSize / MaxSize Font boyutu sınırları
FontMustExist Sadece mevcut fontlar
FontDialog Kullanımı
// Label yazı tipini değiştir
private void btnYaziTipi_Click(object sender, EventArgs e)
{
    FontDialog dialog = new FontDialog();
    dialog.Font = lblMesaj.Font; // Mevcut fontu göster
    dialog.ShowColor = true;
    dialog.Color = lblMesaj.ForeColor;
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        lblMesaj.Font = dialog.Font;
        lblMesaj.ForeColor = dialog.Color;
    }
}

// TextBox için font değiştir
private void btnTxtFont_Click(object sender, EventArgs e)
{
    FontDialog dialog = new FontDialog();
    dialog.Font = txtIcerik.Font;
    dialog.MinSize = 8;
    dialog.MaxSize = 72;
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        txtIcerik.Font = dialog.Font;
        
        // Font bilgilerini göster
        lblFontBilgi.Text = $"{dialog.Font.Name}, {dialog.Font.Size}pt";
        
        if (dialog.Font.Bold) lblFontBilgi.Text += ", Kalın";
        if (dialog.Font.Italic) lblFontBilgi.Text += ", İtalik";
        if (dialog.Font.Underline) lblFontBilgi.Text += ", Altı Çizili";
    }
}

Renklerle çalışırken kullanılan yöntemler:

Color Kullanımı
// Hazır renkler
panel1.BackColor = Color.Red;

panel3.BackColor = Color.DarkGreen;

// RGB ile renk oluşturma
Color ozelRenk = Color.FromArgb(255, 128, 64); // R, G, B
panel1.BackColor = ozelRenk;

// ARGB ile (Alpha dahil - şeffaflık)
Color yariseffaf = Color.FromArgb(128, 255, 0, 0); // %50 şeffaf kırmızı

// HEX'ten renk
Color hexRenk = ColorTranslator.FromHtml("#FF5733");

// Renk bileşenlerini alma
int kirmizi = Color.Coral.R;
int yesil = Color.Coral.G;
int mavi = Color.Coral.B;

// Sistem renkleri
this.BackColor = SystemColors.Control;
label1.BackColor = SystemColors.Highlight;

Font Sınıfı

Font Kullanımı
// Basit font
label1.Font = new Font("Arial", 14);

// Font stili ile
label1.Font = new Font("Arial", 14, FontStyle.Bold);

// Birden fazla stil (| - pipe ile)
label1.Font = new Font("Arial", 14, FontStyle.Bold | FontStyle.Italic);

// Mevcut fontun boyutunu değiştirme
label1.Font = new Font(label1.Font.FontFamily, 20);

// Mevcut fontu kalın yapma
label1.Font = new Font(label1.Font, FontStyle.Bold);

// Font stilleri
// FontStyle.Regular - Normal
// FontStyle.Bold - Kalın
// FontStyle.Italic - İtalik
// FontStyle.Underline - Altı çizili
// FontStyle.Strikeout - Üstü çizili

Metin Editörü Biçimlendirme
// Kalın
private void btnKalin_Click(object sender, EventArgs e)
{
    if (txtIcerik.Font.Bold)
        txtIcerik.Font = new Font(txtIcerik.Font, txtIcerik.Font.Style ^ FontStyle.Bold);
    else
        txtIcerik.Font = new Font(txtIcerik.Font, txtIcerik.Font.Style | FontStyle.Bold);
}

private void btnItalik_Click(object sender, EventArgs e)

private void btnAltiCizili_Click(object sender, EventArgs e)
{
    txtIcerik.Font = new Font(txtIcerik.Font, txtIcerik.Font.Style ^ FontStyle.Underline);
}

// Font boyutu ComboBox'tan
private void cmbBoyut_SelectedIndexChanged(object sender, EventArgs e)

}

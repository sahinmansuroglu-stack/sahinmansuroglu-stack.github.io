Proje Hakkında

Otel için rezervasyon formu. Müşteriler tarih seçer, oda tipi belirler ve misafir sayısı girer.

📚 Kullanılan Konular: DateTimePicker, NumericUpDown, MaskedTextBox, RadioButton, CheckBox

Form Kontrolleri

Kontrol Name Özellik
MaskedTextBox mtbTelefon Mask: (000) 000 00 00
DateTimePicker dtpGiris MinDate: Today
DateTimePicker dtpCikis MinDate: Today + 1
NumericUpDown numYetiskin Min: 1, Max: 4
RadioButton rbStandart Standart (500₺/gece)
RadioButton rbDeluxe Deluxe (800₺/gece)

Kod Çözümü

Form_Load ve Fiyat Hesaplama
private void Form1_Load(object sender, EventArgs e)
{
    dtpGiris.MinDate = DateTime.Today;
    dtpCikis.MinDate = DateTime.Today.AddDays(1);
    numYetiskin.Minimum = 1;
    numYetiskin.Maximum = 4;
    rbStandart.Checked = true;
    FiyatHesapla();
}

private void dtpGiris_ValueChanged(object sender, EventArgs e)
{
    dtpCikis.MinDate = dtpGiris.Value.AddDays(1);
    FiyatHesapla();
}

private void FiyatHesapla()
{
    int gece = (dtpCikis.Value - dtpGiris.Value).Days;
    decimal odaFiyat = rbStandart.Checked ? 500 : 
                       rbDeluxe.Checked ? 800 : 1200;
    
    decimal toplam = odaFiyat * gece;
    
    if (chkKahvalti.Checked) toplam += 100 * gece;
    if (chkTransfer.Checked) toplam += 250;
    
    lblToplam.Text = $"Toplam: {toplam:N0}₺ ({gece} gece)";
}
Rezervasyon Onaylama
private void btnRezervasyon_Click(object sender, EventArgs e)
{
    if (string.IsNullOrWhiteSpace(txtAd.Text))
    {
        MessageBox.Show("Ad zorunludur!");
        return;
    }
    
    if (!mtbTelefon.MaskCompleted)
    {
        MessageBox.Show("Telefon eksiksiz girin!");
        return;
    }
    
    string rezervasyonNo = "RZ-" + DateTime.Now.ToString("yyyyMMddHHmmss");
    MessageBox.Show($"Rezervasyon Alındı!\nNo: {rezervasyonNo}");
}

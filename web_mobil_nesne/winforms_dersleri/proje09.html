Klasör seçerek dosyaları listeleyen, kopyalama ve taşıma işlemleri yapabilen dosya yöneticisi.

Kod Çözümü

Klasör Seçme ve Dosya Listeleme
private string mevcutKlasor = "";

private void btnKlasorSec_Click(object sender, EventArgs e)
{
    FolderBrowserDialog fbd = new FolderBrowserDialog();
    fbd.Description = "Klasör seçin";
    
    if (fbd.ShowDialog() == DialogResult.OK)
    {
        mevcutKlasor = fbd.SelectedPath;
        txtKlasorYolu.Text = mevcutKlasor;
        DosyalariListele();
    }
}

private void DosyalariListele()
{
    lstDosyalar.Items.Clear();
    
    try
    {
        // Klasörleri ekle
        foreach (string klasor in Directory.GetDirectories(mevcutKlasor))
        {
            lstDosyalar.Items.Add("📁 " + Path.GetFileName(klasor));
        }
        
        // Dosyaları ekle
        foreach (string dosya in Directory.GetFiles(mevcutKlasor))
        {
            string uzanti = Path.GetExtension(dosya).ToLower();
            string ikon = uzanti switch
            {
                ".txt" => "📄",
                ".jpg" or ".png" => "🖼️",
                ".mp3" => "🎵",
                ".mp4" => "🎬",
                _ => "📋"
            };
            lstDosyalar.Items.Add(ikon + " " + Path.GetFileName(dosya));
        }
        
        lblDurum.Text = $"{lstDosyalar.Items.Count} öğe";
    }
    catch (Exception ex)
    {
        MessageBox.Show("Hata: " + ex.Message);
    }
}
Dosya Bilgisi Gösterme
private void lstDosyalar_SelectedIndexChanged(object sender, EventArgs e)
{
    if (lstDosyalar.SelectedItem == null) return;
    
    string secili = lstDosyalar.SelectedItem.ToString().Substring(2).Trim();
    string tamYol = Path.Combine(mevcutKlasor, secili);
    
    if (File.Exists(tamYol))
    {
        FileInfo fi = new FileInfo(tamYol);
        lblDosyaBilgi.Text = $"Boyut: {fi.Length / 1024.0:N1} KB\n" +
                            $"Oluşturma: {fi.CreationTime:dd.MM.yyyy}\n" +
                            $"Değişiklik: {fi.LastWriteTime:dd.MM.yyyy}";
    }
}
Kopyala ve Taşı İşlemleri
private void btnKopyala_Click(object sender, EventArgs e)
{
    if (lstDosyalar.SelectedItem == null) return;
    
    string dosyaAdi = lstDosyalar.SelectedItem.ToString().Substring(2).Trim();
    string kaynakYol = Path.Combine(mevcutKlasor, dosyaAdi);
    
    if (!File.Exists(kaynakYol)) return;
    
    FolderBrowserDialog fbd = new FolderBrowserDialog();
    fbd.Description = "Hedef klasörü seçin";
    
    if (fbd.ShowDialog() == DialogResult.OK)
    {
        string hedefYol = Path.Combine(fbd.SelectedPath, dosyaAdi);
        
        try
        {
            File.Copy(kaynakYol, hedefYol, true);
            MessageBox.Show("Dosya kopyalandı!", "Başarılı");
        }
        catch (Exception ex)
        {
            MessageBox.Show("Hata: " + ex.Message);
        }
    }
}

private void btnTasi_Click(object sender, EventArgs e)
{
    if (lstDosyalar.SelectedItem == null) return;
    
    string dosyaAdi = lstDosyalar.SelectedItem.ToString().Substring(2).Trim();
    string kaynakYol = Path.Combine(mevcutKlasor, dosyaAdi);
    
    if (!File.Exists(kaynakYol)) return;
    
    FolderBrowserDialog fbd = new FolderBrowserDialog();
    
    if (fbd.ShowDialog() == DialogResult.OK)
    {
        string hedefYol = Path.Combine(fbd.SelectedPath, dosyaAdi);
        
        try
        {
            File.Move(kaynakYol, hedefYol);
            MessageBox.Show("Dosya taşındı!");
            DosyalariListele();
        }
        catch (Exception ex)
        {
            MessageBox.Show("Hata: " + ex.Message);
        }
    }
}

private void btnSil_Click(object sender, EventArgs e)
{
    if (lstDosyalar.SelectedItem == null) return;
    
    DialogResult onay = MessageBox.Show("Silmek istediğinize emin misiniz?",
        "Onay", MessageBoxButtons.YesNo, MessageBoxIcon.Warning);
    
    if (onay == DialogResult.Yes)
    {
        string dosyaAdi = lstDosyalar.SelectedItem.ToString().Substring(2).Trim();
        string tamYol = Path.Combine(mevcutKlasor, dosyaAdi);
        
        try
        {
            File.Delete(tamYol);
            DosyalariListele();
        }
        catch (Exception ex)
        {
            MessageBox.Show("Hata: " + ex.Message);
        }
    }
}

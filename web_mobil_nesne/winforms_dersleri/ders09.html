1. OpenFileDialog (Dosya Açma)

OpenFileDialog, kullanıcının bilgisayarından dosya seçmesini sağlayan Windows dialogudur.

Özellik Açıklama
FileName Seçilen dosyanın tam yolu
FileNames Çoklu seçimde tüm dosyalar
Filter Gösterilecek dosya türleri
FilterIndex Varsayılan filtre (1'den başlar)
Title Dialog başlığı
InitialDirectory Başlangıç klasörü
Multiselect Birden fazla dosya seçimi

Filter Formatı

Dosya filtreleri şu formatta yazılır: Açıklama|uzantı

Filtre Örnekleri
// Tek uzantı
dialog.Filter = "Metin Dosyaları|*.txt";

// Birden fazla filtre (| ile ayrılır)
dialog.Filter = "Metin Dosyaları|*.txt|Word Dosyaları|*.docx|Tüm Dosyalar|*.*";

// Aynı filtrede birden fazla uzantı (; ile ayrılır)
dialog.Filter = "Resim Dosyaları|*.jpg;*.jpeg;*.png;*.gif;*.bmp";

// Varsayılan filtre indeksi (1'den başlar)
dialog.FilterIndex = 2; // İkinci filtre seçili gelir

OpenFileDialog Kullanımı

Metin Dosyası Açma
private void btnDosyaAc_Click(object sender, EventArgs e)
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Title = "Dosya Seç";
    dialog.Filter = "Metin Dosyaları|*.txt|Tüm Dosyalar|*.*";
    dialog.InitialDirectory = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        string dosyaYolu = dialog.FileName;
        string icerik = File.ReadAllText(dosyaYolu);
        
        txtIcerik.Text = icerik;
        this.Text = Path.GetFileName(dosyaYolu);
        
        lblDurum.Text = "Dosya yüklendi: " + dosyaYolu;
    }
}
Resim Dosyası Açma
private void btnResimSec_Click(object sender, EventArgs e)
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Title = "Resim Seç";
    dialog.Filter = "Resim Dosyaları|*.jpg;*.jpeg;*.png;*.gif;*.bmp";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        pictureBox1.Image = Image.FromFile(dialog.FileName);
        pictureBox1.SizeMode = PictureBoxSizeMode.Zoom;
    }
}
Çoklu Dosya Seçimi
private void btnCokluSec_Click(object sender, EventArgs e)
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Multiselect = true; // Çoklu seçim aktif
    dialog.Filter = "Tüm Dosyalar|*.*";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        listBox1.Items.Clear();
        
        foreach (string dosya in dialog.FileNames)
        {
            listBox1.Items.Add(Path.GetFileName(dosya));
        }
        
        lblDurum.Text = $"Seçilen: {dialog.FileNames.Length} dosya";
    }
}

2. SaveFileDialog (Dosya Kaydetme)

SaveFileDialog, kullanıcının dosya kaydetme konumu ve adını belirlemesini sağlar.

Özellik Açıklama
FileName Kaydedilecek dosya adı
Filter Kaydetme formatları
DefaultExt Varsayılan uzantı
AddExtension Uzantı otomatik eklensin mi?
OverwritePrompt Üzerine yazma uyarısı
Title Dialog başlığı
Metin Dosyası Kaydetme
private void btnKaydet_Click(object sender, EventArgs e)
{
    SaveFileDialog dialog = new SaveFileDialog();
    dialog.Title = "Dosyayı Kaydet";
    dialog.Filter = "Metin Dosyası|*.txt|Tüm Dosyalar|*.*";
    dialog.DefaultExt = "txt";
    dialog.FileName = "YeniDosya"; // Varsayılan ad
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        File.WriteAllText(dialog.FileName, txtIcerik.Text);
        
        MessageBox.Show("Dosya kaydedildi!", "Bilgi",
            MessageBoxButtons.OK, MessageBoxIcon.Information);
        
        this.Text = Path.GetFileName(dialog.FileName);
    }
}
Resim Kaydetme
private void btnResimKaydet_Click(object sender, EventArgs e)
{
    if (pictureBox1.Image == null)
    {
        MessageBox.Show("Kaydedilecek resim yok!");
        return;
    }
    
    SaveFileDialog dialog = new SaveFileDialog();
    dialog.Filter = "PNG Resim|*.png|JPEG Resim|*.jpg|Bitmap|*.bmp";
    dialog.DefaultExt = "png";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        string uzanti = Path.GetExtension(dialog.FileName).ToLower();
        
        switch (uzanti)
        {
            case ".png":
                pictureBox1.Image.Save(dialog.FileName, ImageFormat.Png);
                break;
            case ".jpg":
            case ".jpeg":
                pictureBox1.Image.Save(dialog.FileName, ImageFormat.Jpeg);
                break;
            case ".bmp":
                pictureBox1.Image.Save(dialog.FileName, ImageFormat.Bmp);
                break;
        }
        
        MessageBox.Show("Resim kaydedildi!");
    }
}

3. FolderBrowserDialog (Klasör Seçme)

FolderBrowserDialog, kullanıcının klasör seçmesini sağlar.

Klasör Seçimi
private void btnKlasorSec_Click(object sender, EventArgs e)
{
    FolderBrowserDialog dialog = new FolderBrowserDialog();
    dialog.Description = "Hedef klasörü seçin";
    dialog.ShowNewFolderButton = true;
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        string klasorYolu = dialog.SelectedPath;
        txtKlasor.Text = klasorYolu;
        
        // Klasördeki dosyaları listele
        string[] dosyalar = Directory.GetFiles(klasorYolu);
        
        listBox1.Items.Clear();
        foreach (string dosya in dosyalar)
        {
            listBox1.Items.Add(Path.GetFileName(dosya));
        }
    }
}

Pratik Örnek: Basit Metin Editörü

Metin Editörü
private string mevcutDosya = "";
private bool degisiklikVar = false;

// Yeni dosya
private void Yeni()
{
    if (degisiklikVar)
    {
        DialogResult sonuc = MessageBox.Show(
            "Değişiklikleri kaydetmek ister misiniz?",
            "Kaydet", MessageBoxButtons.YesNoCancel);
        
        if (sonuc == DialogResult.Yes) Kaydet();
        else if (sonuc == DialogResult.Cancel) return;
    }
    
    txtIcerik.Clear();
    mevcutDosya = "";
    degisiklikVar = false;
    this.Text = "Yeni Belge - Metin Editörü";
}

// Dosya aç
private void Ac()
{
    OpenFileDialog dialog = new OpenFileDialog();
    dialog.Filter = "Metin Dosyaları|*.txt|Tüm Dosyalar|*.*";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        txtIcerik.Text = File.ReadAllText(dialog.FileName);
        mevcutDosya = dialog.FileName;
        degisiklikVar = false;
        this.Text = Path.GetFileName(dialog.FileName) + " - Metin Editörü";
    }
}

// Kaydet
private void Kaydet()
{
    if (string.IsNullOrEmpty(mevcutDosya))
    {
        FarkliKaydet();
    }
    else
    {
        File.WriteAllText(mevcutDosya, txtIcerik.Text);
        degisiklikVar = false;
    }
}

// Farklı kaydet
private void FarkliKaydet()
{
    SaveFileDialog dialog = new SaveFileDialog();
    dialog.Filter = "Metin Dosyası|*.txt";
    dialog.DefaultExt = "txt";
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        File.WriteAllText(dialog.FileName, txtIcerik.Text);
        mevcutDosya = dialog.FileName;
        degisiklikVar = false;
        this.Text = Path.GetFileName(dialog.FileName) + " - Metin Editörü";
    }
}

// Değişiklik takibi
private void txtIcerik_TextChanged(object sender, EventArgs e)
{
    degisiklikVar = true;
    if (!this.Text.EndsWith("*"))
        this.Text += "*";
}

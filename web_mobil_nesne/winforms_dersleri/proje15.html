Model Sınıfı

Kisi.cs
using System.Text.Json.Serialization;

public class Kisi
{
    public int Id { get; set; }
    public string Ad { get; set; }
    public string Soyad { get; set; }
    public string Telefon { get; set; }
    public string Email { get; set; }
    public string Adres { get; set; }
    public DateTime DogumTarihi { get; set; }
    public string Kategori { get; set; }
    public string FotoYolu { get; set; }
    public bool Favori { get; set; }
    
    [JsonIgnore]
    public string TamAd => $"{Ad} {Soyad}";
    
    [JsonIgnore]
    public int Yas => DateTime.Today.Year - DogumTarihi.Year;
}

Ana Form - Form1.cs

Değişkenler ve Başlangıç
private BindingList<Kisi> kisiler = new BindingList<Kisi>();
private string veriDosyasi;
private int sonId = 0;
private bool degisiklikVar = false;

private void Form1_Load(object sender, EventArgs e)
{
    // Veri dosyası yolu
    veriDosyasi = Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments),
        "Rehber", "kisiler.json");
    
    // Klasörü oluştur
    Directory.CreateDirectory(Path.GetDirectoryName(veriDosyasi));
    
    // Kategorileri doldur
    cmbKategori.Items.AddRange(new[] { "Aile", "Arkadaş", "İş", "Diğer" });
    cmbFiltre.Items.Add("Tümü");
    cmbFiltre.Items.AddRange(new[] { "Aile", "Arkadaş", "İş", "Diğer", "Favoriler" });
    cmbFiltre.SelectedIndex = 0;
    
    // DataGridView
    DataGridViewAyarla();
    
    // Verileri yükle
    VerileriYukle();
    
    // Otomatik kaydetme timer
    timerOtoKaydet.Interval = 60000; // 1 dakika
    timerOtoKaydet.Start();
}
DataGridView Ayarları
private void DataGridViewAyarla()
{
    dgvKisiler.AutoGenerateColumns = false;
    dgvKisiler.SelectionMode = DataGridViewSelectionMode.FullRowSelect;
    dgvKisiler.MultiSelect = false;
    
    dgvKisiler.Columns.Add(new DataGridViewTextBoxColumn 
        { HeaderText = "ID", DataPropertyName = "Id", Width = 40 });
    dgvKisiler.Columns.Add(new DataGridViewTextBoxColumn 
        { HeaderText = "Ad Soyad", DataPropertyName = "TamAd", Width = 150 });
    dgvKisiler.Columns.Add(new DataGridViewTextBoxColumn 
        { HeaderText = "Telefon", DataPropertyName = "Telefon", Width = 120 });
    dgvKisiler.Columns.Add(new DataGridViewTextBoxColumn 
        { HeaderText = "Kategori", DataPropertyName = "Kategori", Width = 80 });
    dgvKisiler.Columns.Add(new DataGridViewCheckBoxColumn 
        { HeaderText = "⭐", DataPropertyName = "Favori", Width = 40 });
    
    dgvKisiler.DataSource = kisiler;
}
Veri Yükleme ve Kaydetme
private void VerileriYukle()
{
    if (File.Exists(veriDosyasi))
    {
        try
        {
            string json = File.ReadAllText(veriDosyasi);
            var liste = JsonSerializer.Deserialize<List<Kisi>>(json);
            
            foreach (var k in liste)
            {
                kisiler.Add(k);
                if (k.Id > sonId) sonId = k.Id;
            }
            
            tslDurum.Text = $"{kisiler.Count} kişi yüklendi";
        }
        catch (Exception ex)
        {
            MessageBox.Show("Yükleme hatası: " + ex.Message);
        }
    }
}

private void VerileriKaydet()
{
    try
    {
        var options = new JsonSerializerOptions { WriteIndented = true };
        string json = JsonSerializer.Serialize(kisiler.ToList(), options);
        File.WriteAllText(veriDosyasi, json);
        
        degisiklikVar = false;
        tslDurum.Text = "Kaydedildi: " + DateTime.Now.ToString("HH:mm:ss");
    }
    catch (Exception ex)
    {
        MessageBox.Show("Kaydetme hatası: " + ex.Message);
    }
}

private void timerOtoKaydet_Tick(object sender, EventArgs e)
{
    if (degisiklikVar)
    {
        VerileriKaydet();
    }
}
CRUD İşlemleri
private void btnEkle_Click(object sender, EventArgs e)
{
    if (!Dogrula()) return;
    
    kisiler.Add(new Kisi
    {
        Id = ++sonId,
        Ad = txtAd.Text,
        Soyad = txtSoyad.Text,
        Telefon = mtbTelefon.Text,
        Email = txtEmail.Text,
        Adres = txtAdres.Text,
        DogumTarihi = dtpDogum.Value,
        Kategori = cmbKategori.Text,
        FotoYolu = picFoto.ImageLocation ?? "",
        Favori = chkFavori.Checked
    });
    
    degisiklikVar = true;
    FormuTemizle();
}

private bool Dogrula()
{
    if (string.IsNullOrWhiteSpace(txtAd.Text))
    {
        MessageBox.Show("Ad zorunludur!", "Hata", 
            MessageBoxButtons.OK, MessageBoxIcon.Error);
        txtAd.Focus();
        return false;
    }
    
    if (!mtbTelefon.MaskCompleted)
    {
        MessageBox.Show("Telefon numarasını eksiksiz girin!");
        mtbTelefon.Focus();
        return false;
    }
    
    return true;
}

private void dgvKisiler_SelectionChanged(object sender, EventArgs e)
{
    if (dgvKisiler.CurrentRow == null) return;
    
    Kisi k = kisiler[dgvKisiler.CurrentRow.Index];
    KisiyiFormaDoldur(k);
}

private void btnSil_Click(object sender, EventArgs e)
{
    if (dgvKisiler.CurrentRow == null) return;
    
    if (MessageBox.Show("Kişiyi silmek istiyor musunuz?", "Onay",
        MessageBoxButtons.YesNo, MessageBoxIcon.Question) == DialogResult.Yes)
    {
        kisiler.RemoveAt(dgvKisiler.CurrentRow.Index);
        degisiklikVar = true;
    }
}
Arama ve Filtreleme
private void Filtrele()
{
    string arama = txtArama.Text.ToLower();
    string kategori = cmbFiltre.SelectedItem?.ToString();
    
    var sonuc = kisiler.Where(k =>
        (k.Ad.ToLower().Contains(arama) || 
         k.Soyad.ToLower().Contains(arama) ||
         k.Telefon.Contains(arama)) &&
        (kategori == "Tümü" || 
         kategori == "Favoriler" && k.Favori ||
         k.Kategori == kategori)
    ).ToList();
    
    dgvKisiler.DataSource = new BindingList<Kisi>(sonuc);
    tslDurum.Text = $"{sonuc.Count} kişi bulundu";
}

private void txtArama_TextChanged(object sender, EventArgs e) => Filtrele();
private void cmbFiltre_SelectedIndexChanged(object sender, EventArgs e) => Filtrele();
Fotoğraf Seçme (PictureBox)
private void btnFotoSec_Click(object sender, EventArgs e)
{
    OpenFileDialog ofd = new OpenFileDialog();
    ofd.Filter = "Resim Dosyaları|*.jpg;*.png;*.bmp";
    
    if (ofd.ShowDialog() == DialogResult.OK)
    {
        picFoto.ImageLocation = ofd.FileName;
        picFoto.SizeMode = PictureBoxSizeMode.Zoom;
    }
}

// Dosya Menüsü
private void kaydetToolStripMenuItem_Click(object sender, EventArgs e)
{
    VerileriKaydet();

private void cikisToolStripMenuItem_Click(object sender, EventArgs e)
{
    if (degisiklikVar)
    {
        DialogResult dr = MessageBox.Show("Değişiklikleri kaydetmek ister misiniz?",
            "Çıkış", MessageBoxButtons.YesNoCancel);
        
        if (dr == DialogResult.Yes) VerileriKaydet();
        else if (dr == DialogResult.Cancel) return;
    }
    
    Application.Exit();
}

// Dışa Aktar - CSV
private void csvAktarToolStripMenuItem_Click(object sender, EventArgs e)
{
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "CSV Dosyası|*.csv";
    sfd.FileName = "Rehber_" + DateTime.Now.ToString("yyyyMMdd");
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        using (StreamWriter sw = new StreamWriter(sfd.FileName, false, Encoding.UTF8))
        {
            sw.WriteLine("Ad,Soyad,Telefon,Email,Kategori");
            
            foreach (Kisi k in kisiler)
            {
                sw.WriteLine($"\"{k.Ad}\",\"{k.Soyad}\",\"{k.Telefon}\",\"{k.Email}\",\"{k.Kategori}\"");
            }
        }
        
        MessageBox.Show("CSV dosyası oluşturuldu!");
    }
}

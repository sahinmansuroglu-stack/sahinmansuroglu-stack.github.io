Proje Hakkında

DataGridView ile CRUD işlemleri yapan ürün yönetim sistemi. Arama, filtreleme ve koşullu formatlama.

📚 Yeni Konular (Ders 13): DataGridView, DataSource, Columns, Rows, CellClick, CellFormatting

Kod Çözümü

Model ve Veri
public class Urun
{
    public int Id { get; set; }
    public string Ad { get; set; }
    public string Kategori { get; set; }
    public decimal Fiyat { get; set; }
    public int Stok { get; set; }
    
    public string StokDurumu => Stok == 0 ? "Tükendi" : 
                                Stok < 10 ? "Az" : "Yeterli";
}

private BindingList<Urun> urunler = new BindingList<Urun>();
private int sonId = 0;
Form Başlangıç
private void Form1_Load(object sender, EventArgs e)
{
    // Kategorileri doldur
    cmbKategori.Items.AddRange(new[] { "Elektronik", "Giyim", "Gıda", "Kitap" });
    cmbFiltre.Items.Add("Tümü");
    cmbFiltre.Items.AddRange(new[] { "Elektronik", "Giyim", "Gıda", "Kitap" });
    cmbFiltre.SelectedIndex = 0;
    
    // DataGridView ayarları
    dgvUrunler.AutoGenerateColumns = false;
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Id", HeaderText = "ID", DataPropertyName = "Id", Width = 50 });
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Ad", HeaderText = "Ürün Adı", DataPropertyName = "Ad", Width = 150 });
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Kategori", HeaderText = "Kategori", DataPropertyName = "Kategori", Width = 100 });
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Fiyat", HeaderText = "Fiyat", DataPropertyName = "Fiyat", Width = 80 });
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Stok", HeaderText = "Stok", DataPropertyName = "Stok", Width = 60 });
    dgvUrunler.Columns.Add(new DataGridViewTextBoxColumn { Name = "Durum", HeaderText = "Durum", DataPropertyName = "StokDurumu", Width = 70 });
    
    dgvUrunler.DataSource = urunler;
    
    // Örnek veri
    urunler.Add(new Urun { Id = ++sonId, Ad = "Laptop", Kategori = "Elektronik", Fiyat = 15000, Stok = 25 });
    urunler.Add(new Urun { Id = ++sonId, Ad = "T-Shirt", Kategori = "Giyim", Fiyat = 150, Stok = 5 });
}
CRUD İşlemleri
private void btnEkle_Click(object sender, EventArgs e)
{
    if (string.IsNullOrWhiteSpace(txtAd.Text) || cmbKategori.SelectedIndex == -1)
    {
        MessageBox.Show("Tüm alanları doldurun!");
        return;
    }
    
    urunler.Add(new Urun
    {
        Id = ++sonId,
        Ad = txtAd.Text,
        Kategori = cmbKategori.Text,
        Fiyat = numFiyat.Value,
        Stok = (int)numStok.Value
    });
    
    FormuTemizle();
}

private void dgvUrunler_CellClick(object sender, DataGridViewCellEventArgs e)
{
    if (e.RowIndex < 0) return;
    
    Urun u = urunler[e.RowIndex];
    txtAd.Text = u.Ad;
    cmbKategori.Text = u.Kategori;
    numFiyat.Value = u.Fiyat;
    numStok.Value = u.Stok;
}

private void btnGuncelle_Click(object sender, EventArgs e)
{
    if (dgvUrunler.CurrentRow == null) return;
    
    Urun u = urunler[dgvUrunler.CurrentRow.Index];
    u.Ad = txtAd.Text;
    u.Kategori = cmbKategori.Text;
    u.Fiyat = numFiyat.Value;
    u.Stok = (int)numStok.Value;
    
    dgvUrunler.Refresh();
}

private void btnSil_Click(object sender, EventArgs e)
{
    if (dgvUrunler.CurrentRow == null) return;
    
    if (MessageBox.Show("Silmek istiyor musunuz?", "Onay", 
        MessageBoxButtons.YesNo) == DialogResult.Yes)
    {
        urunler.RemoveAt(dgvUrunler.CurrentRow.Index);
    }
}
Arama ve Koşullu Formatlama
private void txtArama_TextChanged(object sender, EventArgs e)
{
    string arama = txtArama.Text.ToLower();
    string kategori = cmbFiltre.SelectedItem?.ToString();
    
    var sonuc = urunler.Where(u => 
        u.Ad.ToLower().Contains(arama) &&
        (kategori == "Tümü" || u.Kategori == kategori)
    ).ToList();
    
    dgvUrunler.DataSource = new BindingList<Urun>(sonuc);
}

// Stok durumuna göre renklendirme
private void dgvUrunler_CellFormatting(object sender, DataGridViewCellFormattingEventArgs e)
{
    if (e.ColumnIndex == dgvUrunler.Columns["Stok"].Index && e.Value != null)
    {
        int stok = (int)e.Value;
        
        if (stok == 0)
            e.CellStyle.BackColor = Color.LightCoral;
        else if (stok < 10)
            e.CellStyle.BackColor = Color.LightYellow;
        else
            e.CellStyle.BackColor = Color.LightGreen;
    }
}

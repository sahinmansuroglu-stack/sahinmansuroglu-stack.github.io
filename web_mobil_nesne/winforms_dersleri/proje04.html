Proje Hakkında

Bu projede, bir restoran için sipariş alma formu oluşturacaksınız. Müşteriler yemek seçebilir, ekstra malzemeler ekleyebilir ve siparişlerini onaylayabilir.

📚 Kullanılan Konular:
  • Ders 1: Form oluşturma, proje yapısı
  • Ders 2: Label, TextBox, Button
  • Ders 3: CheckBox, RadioButton, GroupBox
  • Ders 4: ListBox, ComboBox

Proje Gereksinimleri

📋 Form Tasarımı

  • Müşteri adı için TextBox
  • Masa numarası için ComboBox (1-20 arası)
  • Yemek kategorisi seçimi için RadioButton'lar (Pizza, Burger, Makarna)
  • Seçilen kategorideki yemekleri gösteren ListBox
  • Ekstra malzemeler için CheckBox'lar
  • Sipariş özeti için çok satırlı TextBox
  • Sipariş Ekle ve Siparişi Onayla butonları

🎯 İşlevsellik

  • Kategori değiştiğinde ListBox'taki yemekler güncellensin
  • Yemek seçildiğinde fiyat gösterilsin
  • Ekstra malzemeler ek ücret eklesin
  • Toplam tutar hesaplansın
  • Sipariş özeti TextBox'ta gösterilsin

Adım 2: Form Yüklendiğinde

Form1_Load
private void Form1_Load(object sender, EventArgs e)
{
    // Masa numaralarını ekle
    for (int i = 1; i <= 20; i++)
    {
        cmbMasa.Items.Add("Masa " + i);
    }
    cmbMasa.SelectedIndex = 0;
    
    // Varsayılan kategori
    rbPizza.Checked = true;
    
    // Pizza menüsünü yükle
    YemekleriYukle();
}

Adım 3: Kategori Değişimi

Kategori RadioButton CheckedChanged Olayları
// Yemek verileri (sınıf düzeyinde)
private Dictionary<string, decimal> pizzalar = new Dictionary<string, decimal>()
{
    { "Margarita", 80 },
    { "Pepperoni", 95 },
    { "Karışık", 110 },
    { "Sucuklu", 90 },
    { "Vejeteryan", 85 }
};

private Dictionary<string, decimal> burgerler = new Dictionary<string, decimal>()
{
    { "Klasik Burger", 60 },
    { "Cheese Burger", 70 },
    { "Double Burger", 90 },
    { "Tavuk Burger", 65 }
};

private Dictionary<string, decimal> makarnalar = new Dictionary<string, decimal>()
{
    { "Bolonez", 55 },
    { "Alfredo", 60 },
    { "Carbonara", 65 },
    { "Arabiata", 50 }
};

private void YemekleriYukle()
{
    lstYemekler.Items.Clear();
    
    Dictionary<string, decimal> seciliMenu = null;
    
    if (rbPizza.Checked)
        seciliMenu = pizzalar;
    else if (rbBurger.Checked)
        seciliMenu = burgerler;
    else if (rbMakarna.Checked)
        seciliMenu = makarnalar;
    
    if (seciliMenu != null)
    {
        foreach (var yemek in seciliMenu)
        {
            lstYemekler.Items.Add($"{yemek.Key} - {yemek.Value}₺");
        }
    }
}

// Her RadioButton için
private void rbPizza_CheckedChanged(object sender, EventArgs e)
{
    if (rbPizza.Checked) YemekleriYukle();
}

private void rbBurger_CheckedChanged(object sender, EventArgs e)
{
    if (rbBurger.Checked) YemekleriYukle();
}

private void rbMakarna_CheckedChanged(object sender, EventArgs e)
{
    if (rbMakarna.Checked) YemekleriYukle();
}

Adım 4: Fiyat Hesaplama

Toplam Fiyat Hesaplama
private decimal FiyatHesapla()
{
    decimal toplam = 0;
    
    // Seçili yemek fiyatı
    if (lstYemekler.SelectedItem != null)
    {
        string secili = lstYemekler.SelectedItem.ToString();
        string yemekAdi = secili.Split('-')[0].Trim();
        
        if (rbPizza.Checked && pizzalar.ContainsKey(yemekAdi))
            toplam = pizzalar[yemekAdi];
        else if (rbBurger.Checked && burgerler.ContainsKey(yemekAdi))
            toplam = burgerler[yemekAdi];
        else if (rbMakarna.Checked && makarnalar.ContainsKey(yemekAdi))
            toplam = makarnalar[yemekAdi];
    }
    
    // Ekstralari ekle
    if (chkPeynir.Checked) toplam += 5;
    if (chkSos.Checked) toplam += 3;
    if (chkIcecek.Checked) toplam += 10;
    
    return toplam;
}

// ListBox seçim değiştiğinde
private void lstYemekler_SelectedIndexChanged(object sender, EventArgs e)
{
    lblToplam.Text = "Toplam: " + FiyatHesapla() + "₺";
}

// CheckBox'lar değiştiğinde (hepsine aynı event bağla)
private void Ekstra_CheckedChanged(object sender, EventArgs e)
{
    lblToplam.Text = "Toplam: " + FiyatHesapla() + "₺";
}

Adım 5: Siparişi Ekleme ve Onaylama

Sipariş İşlemleri
private List<string> siparisler = new List<string>();
private decimal genelToplam = 0;

private void btnEkle_Click(object sender, EventArgs e)
{
    // Doğrulama
    if (string.IsNullOrWhiteSpace(txtMusteri.Text))
    {
        MessageBox.Show("Müşteri adını girin!", "Uyarı");
        txtMusteri.Focus();
        return;
    }
    
    if (lstYemekler.SelectedItem == null)
    {
        MessageBox.Show("Bir yemek seçin!", "Uyarı");
        return;
    }
    
    // Sipariş metnini oluştur
    string yemek = lstYemekler.SelectedItem.ToString().Split('-')[0].Trim();
    decimal fiyat = FiyatHesapla();
    
    string ekstralar = "";
    if (chkPeynir.Checked) ekstralar += "Ekstra Peynir, ";
    if (chkSos.Checked) ekstralar += "Özel Sos, ";
    if (chkIcecek.Checked) ekstralar += "İçecek, ";
    
    string siparis = $"• {yemek}";
    if (!string.IsNullOrEmpty(ekstralar))
        siparis += $" ({ekstralar.TrimEnd(',', ' ')})";
    siparis += $" - {fiyat}₺";
    
    siparisler.Add(siparis);
    genelToplam += fiyat;
    
    // Özeti güncelle
    OzetiGuncelle();
    
    // Seçimleri temizle
    lstYemekler.ClearSelected();
    chkPeynir.Checked = false;
    chkSos.Checked = false;
    chkIcecek.Checked = false;
    lblToplam.Text = "Toplam: 0₺";
}

private void OzetiGuncelle()
{
    txtOzet.Clear();
    txtOzet.AppendText($"=== SİPARİŞ FİŞİ ===\r\n");
    txtOzet.AppendText($"Müşteri: {txtMusteri.Text}\r\n");
    txtOzet.AppendText($"{cmbMasa.SelectedItem}\r\n");
    txtOzet.AppendText($"Tarih: {DateTime.Now:dd.MM.yyyy HH:mm}\r\n");
    txtOzet.AppendText("─────────────────────\r\n");
    
    foreach (string s in siparisler)
    {
        txtOzet.AppendText(s + "\r\n");
    }
    
    txtOzet.AppendText("─────────────────────\r\n");
    txtOzet.AppendText($"GENEL TOPLAM: {genelToplam}₺\r\n");
}

private void btnOnayla_Click(object sender, EventArgs e)
{
    if (siparisler.Count == 0)
    {
        MessageBox.Show("Önce sipariş ekleyin!", "Uyarı");
        return;
    }
    
    DialogResult sonuc = MessageBox.Show(
        $"Toplam tutar: {genelToplam}₺\nSiparişi onaylıyor musunuz?",
        "Sipariş Onayı",
        MessageBoxButtons.YesNo,
        MessageBoxIcon.Question);
    
    if (sonuc == DialogResult.Yes)
    {
        MessageBox.Show("Sipariş mutfağa gönderildi!\nAfiyet olsun!", 
            "Başarılı", MessageBoxButtons.OK, MessageBoxIcon.Information);
        
        // Formu sıfırla
        siparisler.Clear();
        genelToplam = 0;
        txtMusteri.Clear();
        txtOzet.Clear();
        cmbMasa.SelectedIndex = 0;
    }
}

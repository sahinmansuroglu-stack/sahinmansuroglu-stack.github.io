📚 Yeni Konular (Ders 10): ColorDialog, FontDialog, seçili metin formatlama

Kod Çözümü

Form Tasarımı
// ToolStrip butonları:
// Yeni, Aç, Kaydet | Yazı Tipi, Yazı Rengi, Arka Plan | Kalın, İtalik, Altı Çizili

// Ana kontrol: RichTextBox (rtbIcerik)
rtbIcerik.Dock = DockStyle.Fill;
Font ve Renk Dialogleri
private void btnYaziTipi_Click(object sender, EventArgs e)
{
    FontDialog fd = new FontDialog();
    fd.Font = rtbIcerik.SelectionFont ?? rtbIcerik.Font;
    fd.ShowColor = true;
    fd.Color = rtbIcerik.SelectionColor;
    
    if (fd.ShowDialog() == DialogResult.OK)
    {
        rtbIcerik.SelectionFont = fd.Font;
        rtbIcerik.SelectionColor = fd.Color;
    }
}

private void btnYaziRengi_Click(object sender, EventArgs e)
{
    ColorDialog cd = new ColorDialog();
    cd.Color = rtbIcerik.SelectionColor;
    cd.FullOpen = true;
    
    if (cd.ShowDialog() == DialogResult.OK)
    {
        rtbIcerik.SelectionColor = cd.Color;
    }
}

private void btnArkaPlan_Click(object sender, EventArgs e)
{
    ColorDialog cd = new ColorDialog();
    cd.Color = rtbIcerik.SelectionBackColor;
    
    if (cd.ShowDialog() == DialogResult.OK)
    {
        rtbIcerik.SelectionBackColor = cd.Color;
    }
}
Kalın, İtalik, Altı Çizili
private void btnKalin_Click(object sender, EventArgs e)
{
    if (rtbIcerik.SelectionFont != null)
    {
        Font mevcut = rtbIcerik.SelectionFont;
        FontStyle yeniStil = mevcut.Bold ? 
            mevcut.Style & ~FontStyle.Bold : 
            mevcut.Style | FontStyle.Bold;
        
        rtbIcerik.SelectionFont = new Font(mevcut, yeniStil);
    }
}

private void btnItalik_Click(object sender, EventArgs e)
{
    if (rtbIcerik.SelectionFont != null)
    {
        Font mevcut = rtbIcerik.SelectionFont;
        FontStyle yeniStil = mevcut.Italic ? 
            mevcut.Style & ~FontStyle.Italic : 
            mevcut.Style | FontStyle.Italic;
        
        rtbIcerik.SelectionFont = new Font(mevcut, yeniStil);
    }
}

private void btnAltiCizili_Click(object sender, EventArgs e)
{
    if (rtbIcerik.SelectionFont != null)
    {
        Font mevcut = rtbIcerik.SelectionFont;
        FontStyle yeniStil = mevcut.Underline ? 
            mevcut.Style & ~FontStyle.Underline : 
            mevcut.Style | FontStyle.Underline;
        
        rtbIcerik.SelectionFont = new Font(mevcut, yeniStil);
    }
}
RTF Dosya Kaydetme/Açma
private void btnKaydet_Click(object sender, EventArgs e)
{
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "RTF Dosyası|*.rtf|Metin Dosyası|*.txt";
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        if (sfd.FileName.EndsWith(".rtf"))
            rtbIcerik.SaveFile(sfd.FileName);
        else
            rtbIcerik.SaveFile(sfd.FileName, RichTextBoxStreamType.PlainText);
    }
}

private void btnAc_Click(object sender, EventArgs e)
{
    OpenFileDialog ofd = new OpenFileDialog();
    ofd.Filter = "RTF Dosyası|*.rtf|Metin Dosyası|*.txt";
    
    if (ofd.ShowDialog() == DialogResult.OK)
    {
        rtbIcerik.LoadFile(ofd.FileName, 
            ofd.FileName.EndsWith(".rtf") ? 
            RichTextBoxStreamType.RichText : 
            RichTextBoxStreamType.PlainText);
    }
}

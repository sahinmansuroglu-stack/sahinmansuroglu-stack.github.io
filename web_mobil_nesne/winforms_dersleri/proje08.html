Form ve Değişkenler
private string dosyaYolu = "";
private bool degisiklikVar = false;

private void Form1_Load(object sender, EventArgs e)
{
    txtIcerik.Dock = DockStyle.Fill;
    txtIcerik.Multiline = true;
    txtIcerik.ScrollBars = ScrollBars.Both;
    GuncelleBaslik();
}

private void GuncelleBaslik()
{
    string dosyaAdi = string.IsNullOrEmpty(dosyaYolu) ? "Adsız" : Path.GetFileName(dosyaYolu);
    string degisiklik = degisiklikVar ? "*" : "";
    this.Text = $"{degisiklik}{dosyaAdi} - Metin Editörü";
}

private void txtIcerik_TextChanged(object sender, EventArgs e)
{
    degisiklikVar = true;
    GuncelleBaslik();
    tslKarakterSayisi.Text = $"Karakter: {txtIcerik.Text.Length}";
}
Dosya Menüsü İşlemleri
private void yeniToolStripMenuItem_Click(object sender, EventArgs e)
{
    if (degisiklikVar)
    {
        DialogResult sonuc = MessageBox.Show("Değişiklikleri kaydetmek ister misiniz?",
            "Kaydet", MessageBoxButtons.YesNoCancel, MessageBoxIcon.Question);
        
        if (sonuc == DialogResult.Yes) Kaydet();
        else if (sonuc == DialogResult.Cancel) return;
    }
    
    txtIcerik.Clear();
    dosyaYolu = "";
    degisiklikVar = false;
    GuncelleBaslik();
}

private void acToolStripMenuItem_Click(object sender, EventArgs e)
{
    OpenFileDialog ofd = new OpenFileDialog();
    ofd.Filter = "Metin Dosyaları|*.txt|Tüm Dosyalar|*.*";
    
    if (ofd.ShowDialog() == DialogResult.OK)
    {
        txtIcerik.Text = File.ReadAllText(ofd.FileName);
        dosyaYolu = ofd.FileName;
        degisiklikVar = false;
        GuncelleBaslik();
    }
}

private void Kaydet()
{
    if (string.IsNullOrEmpty(dosyaYolu))
    {
        FarkliKaydet();
    }
    else
    {
        File.WriteAllText(dosyaYolu, txtIcerik.Text);
        degisiklikVar = false;
        GuncelleBaslik();
    }
}

private void FarkliKaydet()
{
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "Metin Dosyası|*.txt";
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        File.WriteAllText(sfd.FileName, txtIcerik.Text);
        dosyaYolu = sfd.FileName;
        degisiklikVar = false;
        GuncelleBaslik();
    }
}

private void txtIcerik_SelectionChanged(object sender, EventArgs e)
{
    int satir = txtIcerik.GetLineFromCharIndex(txtIcerik.SelectionStart) + 1;
    int sutun = txtIcerik.SelectionStart - txtIcerik.GetFirstCharIndexOfCurrentLine() + 1;
    
    tslKonum.Text = $"Satır: {satir}, Sütun: {sutun}";
}

Kod Çözümü

Değişkenler ve Klasör Seçimi
private string[] resimler;
private int mevcutIndex = 0;

private void btnKlasorSec_Click(object sender, EventArgs e)
{
    FolderBrowserDialog fbd = new FolderBrowserDialog();
    
    if (fbd.ShowDialog() == DialogResult.OK)
    {
        resimler = Directory.GetFiles(fbd.SelectedPath)
            .Where(f => f.EndsWith(".jpg") || f.EndsWith(".png") || 
                       f.EndsWith(".bmp") || f.EndsWith(".gif"))
            .ToArray();
        
        if (resimler.Length > 0)
        {
            mevcutIndex = 0;
            ResimGoster();
            btnOnceki.Enabled = true;
            btnSonraki.Enabled = true;
        }
        else
        {
            MessageBox.Show("Klasörde resim bulunamadı!");
        }
    }
}
Resim Gösterme ve Navigasyon
private void ResimGoster()
{
    if (resimler == null || resimler.Length == 0) return;
    
    // Önceki resmi temizle
    if (picResim.Image != null)
    {
        picResim.Image.Dispose();
    }
    
    picResim.Image = Image.FromFile(resimler[mevcutIndex]);
    lblDurum.Text = $"{mevcutIndex + 1} / {resimler.Length}";
    lblDosyaAdi.Text = Path.GetFileName(resimler[mevcutIndex]);
    
    // Resim bilgisi
    FileInfo fi = new FileInfo(resimler[mevcutIndex]);
    lblBoyut.Text = $"{fi.Length / 1024} KB | {picResim.Image.Width}x{picResim.Image.Height}";
}

private void btnOnceki_Click(object sender, EventArgs e)
{
    if (mevcutIndex > 0)
    {
        mevcutIndex--;
        ResimGoster();
    }
}

private void btnSonraki_Click(object sender, EventArgs e)
{
    if (mevcutIndex < resimler.Length - 1)
    {
        mevcutIndex++;
        ResimGoster();
    }
}

// Klavye ile gezinme
private void Form1_KeyDown(object sender, KeyEventArgs e)
{
    if (e.KeyCode == Keys.Left) btnOnceki_Click(null, null);
    if (e.KeyCode == Keys.Right) btnSonraki_Click(null, null);
}
Zoom ve Kaydetme
// Zoom kontrolü
private void trkZoom_Scroll(object sender, EventArgs e)
{
    double oran = trkZoom.Value / 100.0;
    
    int yeniGenislik = (int)(panelResim.Width * oran);
    int yeniYukseklik = (int)(panelResim.Height * oran);
    
    picResim.Width = yeniGenislik;
    picResim.Height = yeniYukseklik;
    
    lblZoom.Text = $"%{trkZoom.Value}";
}

// Resmi farklı kaydet
private void btnKaydet_Click(object sender, EventArgs e)
{
    if (picResim.Image == null) return;
    
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "PNG|*.png|JPEG|*.jpg|Bitmap|*.bmp";
    sfd.FileName = Path.GetFileNameWithoutExtension(resimler[mevcutIndex]);
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        ImageFormat format = sfd.FileName.EndsWith(".png") ? ImageFormat.Png :
                            sfd.FileName.EndsWith(".jpg") ? ImageFormat.Jpeg :
                            ImageFormat.Bmp;
        
        picResim.Image.Save(sfd.FileName, format);
        MessageBox.Show("Resim kaydedildi!");
    }
}

// Çift tıklayınca tam ekran
private void picResim_DoubleClick(object sender, EventArgs e)
{
    Form tamEkran = new Form();
    tamEkran.WindowState = FormWindowState.Maximized;
    tamEkran.BackColor = Color.Black;
    tamEkran.KeyPreview = true;
    tamEkran.KeyDown += (s, args) => { if (args.KeyCode == Keys.Escape) tamEkran.Close(); };
    
    PictureBox pb = new PictureBox();
    pb.Dock = DockStyle.Fill;
    pb.SizeMode = PictureBoxSizeMode.Zoom;
    pb.Image = picResim.Image;
    pb.Click += (s, args) => tamEkran.Close();
    
    tamEkran.Controls.Add(pb);
    tamEkran.ShowDialog();
}

DataGridView Nedir?

DataGridView, verileri tablo formatında görüntülemek ve düzenlemek için kullanılan güçlü bir kontroldür.

ID Ad Soyad Yaş

Özellik Açıklama
Columns Sütun koleksiyonu
Rows Satır koleksiyonu
DataSource Veri kaynağı (List, DataTable vb.)
SelectedRows Seçili satırlar
CurrentRow Aktif satır
AllowUserToAddRows Kullanıcı satır ekleyebilir mi?
AllowUserToDeleteRows Kullanıcı satır silebilir mi?
ReadOnly Sadece okunur mu?

DataSource Kullanımı (List ile)

List'ten Veri Bağlama
// Model sınıfı
public class Ogrenci
{
    public int Id { get; set; }
    public string Ad { get; set; }
    public string Soyad { get; set; }
    public int Yas { get; set; }
    public double Ortalama { get; set; }
}

// Liste oluştur ve bağla
private List<Ogrenci> ogrenciler = new List<Ogrenci>();

private void Form1_Load(object sender, EventArgs e)
{
    ogrenciler.Add(new Ogrenci { Id = 1, Ad = "Ali", Soyad = "Yılmaz", Yas = 20, Ortalama = 85.5 });
    ogrenciler.Add(new Ogrenci { Id = 2, Ad = "Ayşe", Soyad = "Kaya", Yas = 21, Ortalama = 92.3 });
    ogrenciler.Add(new Ogrenci { Id = 3, Ad = "Mehmet", Soyad = "Demir", Yas = 19, Ortalama = 78.8 });
    
    // BindingList kullanarak değişikliklerin yansımasını sağla
    dataGridView1.DataSource = new BindingList<Ogrenci>(ogrenciler);
}

// Yeni öğrenci ekle
private void btnEkle_Click(object sender, EventArgs e)
{
    ogrenciler.Add(new Ogrenci
    {
        Id = ogrenciler.Count + 1,
        Ad = txtAd.Text,
        Soyad = txtSoyad.Text,
        Yas = (int)numYas.Value,
        Ortalama = (double)numOrtalama.Value
    });
    
    dataGridView1.DataSource = null;
    dataGridView1.DataSource = new BindingList<Ogrenci>(ogrenciler);
}

Hücre ve Satır Olayları

DataGridView Olayları
// Satır seçildiğinde
private void dataGridView1_SelectionChanged(object sender, EventArgs e)
{
    if (dataGridView1.CurrentRow != null)
    {
        txtAd.Text = dataGridView1.CurrentRow.Cells["ad"].Value?.ToString();
        txtSoyad.Text = dataGridView1.CurrentRow.Cells["soyad"].Value?.ToString();
    }
}

// Hücreye tıklandığında
private void dataGridView1_CellClick(object sender, DataGridViewCellEventArgs e)
{
    if (e.RowIndex >= 0)
    {
        DataGridViewRow row = dataGridView1.Rows[e.RowIndex];
        lblSecili.Text = $"Seçili: {row.Cells["ad"].Value} {row.Cells["soyad"].Value}";
    }
}

// Button sütununa tıklandığında
private void dataGridView1_CellContentClick(object sender, DataGridViewCellEventArgs e)
{
    if (dataGridView1.Columns[e.ColumnIndex].Name == "sil" && e.RowIndex >= 0)
    {
        DialogResult sonuc = MessageBox.Show("Bu satırı silmek istiyor musunuz?", "Onay",
            MessageBoxButtons.YesNo, MessageBoxIcon.Question);
        
        if (sonuc == DialogResult.Yes)
        {
            dataGridView1.Rows.RemoveAt(e.RowIndex);
        }
    }
}

// Hücre düzenlendikten sonra
private void dataGridView1_CellEndEdit(object sender, DataGridViewCellEventArgs e)
{
    var deger = dataGridView1.Rows[e.RowIndex].Cells[e.ColumnIndex].Value;
    lblDurum.Text = $"Değiştirildi: {deger}";
}

private void dataGridView1_CellFormatting(object sender, DataGridViewCellFormattingEventArgs e)
{
    if (dataGridView1.Columns[e.ColumnIndex].Name == "ortalama")
    {
        double ort;
        if (double.TryParse(e.Value?.ToString(), out ort))
        {
            if (ort >= 85)
                e.CellStyle.BackColor = Color.LightGreen;
            else if (ort >= 70)
                e.CellStyle.BackColor = Color.LightYellow;
            else
                e.CellStyle.BackColor = Color.LightCoral;
        }
    }
}

Arama ve Filtreleme

Basit Arama
private void txtArama_TextChanged(object sender, EventArgs e)
{
    string arama = txtArama.Text.ToLower();
    
    foreach (DataGridViewRow row in dataGridView1.Rows)
    {
        if (row.IsNewRow) continue;
        
        bool goster = false;
        
        foreach (DataGridViewCell cell in row.Cells)
        {
            if (cell.Value != null && cell.Value.ToString().ToLower().Contains(arama))
            {
                goster = true;
                break;
            }
        }
        
        row.Visible = goster;
    }
}

Timer Nedir?

Timer, belirli zaman aralıklarında otomatik olarak kod çalıştıran kontroldür. Saatler, animasyonlar, otomatik kaydetme gibi işlemler için kullanılır.

Timer.Start()
→
Interval ms bekle
→
Tick olayı çalışır
↻
Özellik/Metot Açıklama
Interval Tick aralığı (milisaniye). 1000 = 1 saniye
Enabled Timer aktif mi?
Start() Timer'ı başlat
Stop() Timer'ı durdur
Tick (Event) Her interval'de tetiklenen olay

Dijital Saat
// Form_Load'da ayarla
private void Form1_Load(object sender, EventArgs e)
{
    timer1.Interval = 1000; // 1 saniye
    timer1.Start();
    
    // Başlangıçta saati göster
    lblSaat.Text = DateTime.Now.ToString("HH:mm:ss");

private void timer1_Tick(object sender, EventArgs e)
{
    lblSaat.Text = DateTime.Now.ToString("HH:mm:ss");
    lblTarih.Text = DateTime.Now.ToString("dd MMMM yyyy dddd");
}

Geri Sayım Sayacı

Countdown Timer
private int kalanSaniye;

private void btnBaslat_Click(object sender, EventArgs e)
{
    kalanSaniye = (int)numSure.Value; // NumericUpDown'dan al
    
    if (kalanSaniye <= 0)
    {
        MessageBox.Show("Geçerli bir süre girin!");
        return;
    }
    
    progressBar1.Maximum = kalanSaniye;
    progressBar1.Value = kalanSaniye;
    
    timer1.Interval = 1000;
    timer1.Start();
    
    btnBaslat.Enabled = false;
    btnDurdur.Enabled = true;
}

private void timer1_Tick(object sender, EventArgs e)
{
    kalanSaniye--;
    
    // Süreyi formatlı göster
    TimeSpan ts = TimeSpan.FromSeconds(kalanSaniye);
    lblSure.Text = ts.ToString(@"mm\:ss");
    
    progressBar1.Value = kalanSaniye;
    
    if (kalanSaniye <= 0)
    {
        timer1.Stop();
        MessageBox.Show("Süre doldu!", "Alarm", 
            MessageBoxButtons.OK, MessageBoxIcon.Information);
        
        btnBaslat.Enabled = true;
        btnDurdur.Enabled = false;
    }
}

private void btnDurdur_Click(object sender, EventArgs e)
{
    timer1.Stop();
    btnBaslat.Enabled = true;
    btnDurdur.Enabled = false;
}

Basit Animasyon

Hareket Eden Nesne
private int hiz = 5;
private int yonX = 1;
private int yonY = 1;

private void Form1_Load(object sender, EventArgs e)
{
    timer1.Interval = 20; // 50 FPS
    timer1.Start();
}

private void timer1_Tick(object sender, EventArgs e)
{
    // Yeni konum hesapla
    int yeniX = pictureBox1.Left + (hiz * yonX);
    int yeniY = pictureBox1.Top + (hiz * yonY);
    
    // Sınır kontrolü - X ekseni
    if (yeniX <= 0 || yeniX + pictureBox1.Width >= this.ClientSize.Width)
    {
        yonX = -yonX; // Yön değiştir
    }
    
    // Sınır kontrolü - Y ekseni
    if (yeniY <= 0 || yeniY + pictureBox1.Height >= this.ClientSize.Height)
    {
        yonY = -yonY; // Yön değiştir
    }
    
    // Pozisyonu güncelle
    pictureBox1.Left += hiz * yonX;
    pictureBox1.Top += hiz * yonY;
}

// Hız kontrolü
private void trackBar1_Scroll(object sender, EventArgs e)
{
    hiz = trackBar1.Value;
}

Otomatik Kaydetme

Auto-Save Özelliği
private DateTime sonKayit;
private bool degisiklikVar = false;

private void Form1_Load(object sender, EventArgs e)
{
    timerAutoSave.Interval = 60000; // 1 dakika
    timerAutoSave.Start();
}

private void txtIcerik_TextChanged(object sender, EventArgs e)
{
    degisiklikVar = true;
}

private void timerAutoSave_Tick(object sender, EventArgs e)
{
    if (degisiklikVar)
    {
        OtomatikKaydet();
    }
}

private void OtomatikKaydet()
{
    string yedekDosya = Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments),
        "autosave_" + DateTime.Now.ToString("yyyyMMdd_HHmmss") + ".txt");
    
    File.WriteAllText(yedekDosya, txtIcerik.Text);
    
    sonKayit = DateTime.Now;
    degisiklikVar = false;
    tslDurum.Text = "Otomatik kaydedildi: " + sonKayit.ToString("HH:mm:ss");
}

Slideshow

Otomatik Resim Geçişi
private string[] resimler;
private int index = 0;

private void btnKlasorSec_Click(object sender, EventArgs e)
{
    FolderBrowserDialog dialog = new FolderBrowserDialog();
    
    if (dialog.ShowDialog() == DialogResult.OK)
    {
        resimler = Directory.GetFiles(dialog.SelectedPath, "*.jpg");
        
        if (resimler.Length > 0)
        {
            index = 0;
            ResimGoster();
            
            timerSlide.Interval = 3000; // 3 saniye
            timerSlide.Start();
        }
    }
}

private void timerSlide_Tick(object sender, EventArgs e)
{
    index = (index + 1) % resimler.Length;
    ResimGoster();
}

private void ResimGoster()
{
    pictureBox1.Image = Image.FromFile(resimler[index]);
    lblDurum.Text = $"{index + 1} / {resimler.Length}";
}

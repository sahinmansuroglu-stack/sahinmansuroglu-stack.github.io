Proje Hakkında

Kullanıcıya rastgele kelimeler gösterilir, doğru yazıp Enter'a basması beklenir. Puan ve süre takibi yapılır.

📚 Yeni Konular (Ders 6): TextChanged, KeyPress, KeyDown olayları, Form_Activated, validasyon

Form Kontrolleri

Kontrol Name Açıklama
Label lblKelime Yazılacak kelime (büyük font)
TextBox txtYazi Kullanıcı girişi
Label lblPuan Skor gösterimi
Label lblSure Kalan süre
Label lblDurum Doğru/Yanlış bildirimi
Button btnBaslat Oyunu başlat
Timer timer1 Geri sayım için

Kod Çözümü

Değişkenler ve Başlangıç
private string[] kelimeler = { 
    "bilgisayar", "programlama", "klavye", "ekran", 
    "yazılım", "algoritma", "değişken", "metot",
    "döngü", "koşul", "dizi", "liste"
};
private Random rnd = new Random();
private int puan = 0;
private int kalanSure = 60;
private string mevcutKelime;

private void Form1_Load(object sender, EventArgs e)
{
    timer1.Interval = 1000;
    txtYazi.Enabled = false;
}

private void btnBaslat_Click(object sender, EventArgs e)
{
    puan = 0;
    kalanSure = 60;
    lblPuan.Text = "Puan: 0";
    lblSure.Text = "Süre: 60";
    
    txtYazi.Enabled = true;
    txtYazi.Clear();
    txtYazi.Focus();
    
    YeniKelime();
    timer1.Start();
    btnBaslat.Enabled = false;
}

private void YeniKelime()
{
    mevcutKelime = kelimeler[rnd.Next(kelimeler.Length)];
    lblKelime.Text = mevcutKelime;
    lblDurum.Text = "";
}
KeyDown Olayı - Enter Kontrolü
private void txtYazi_KeyDown(object sender, KeyEventArgs e)
{
    if (e.KeyCode == Keys.Enter)
    {
        e.SuppressKeyPress = true; // Ding sesini engelle
        
        if (txtYazi.Text.Trim().ToLower() == mevcutKelime.ToLower())
        {
            puan += 10;
            lblPuan.Text = "Puan: " + puan;
            lblDurum.Text = "✓ Doğru!";
            lblDurum.ForeColor = Color.Green;
        }
        else
        {
            puan -= 5;
            if (puan < 0) puan = 0;
            lblPuan.Text = "Puan: " + puan;
            lblDurum.Text = "✗ Yanlış!";
            lblDurum.ForeColor = Color.Red;
        }
        
        txtYazi.Clear();
        YeniKelime();
    }
}
TextChanged ve Timer
// Anlık karşılaştırma (opsiyonel)
private void txtYazi_TextChanged(object sender, EventArgs e)
{
    if (mevcutKelime.StartsWith(txtYazi.Text, StringComparison.OrdinalIgnoreCase))
    {
        txtYazi.BackColor = Color.LightGreen;
    }
    else
    {
        txtYazi.BackColor = Color.LightCoral;
    }
    
    if (string.IsNullOrEmpty(txtYazi.Text))
        txtYazi.BackColor = Color.White;
}

private void timer1_Tick(object sender, EventArgs e)
{
    kalanSure--;
    lblSure.Text = "Süre: " + kalanSure;
    
    if (kalanSure <= 0)
    {
        timer1.Stop();
        txtYazi.Enabled = false;
        btnBaslat.Enabled = true;
        
        MessageBox.Show($"Oyun Bitti!\n\nToplam Puan: {puan}", 
            "Sonuç", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }
}

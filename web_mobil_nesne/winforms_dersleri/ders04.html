1. ListBox (Liste Kutusu)

ListBox, alt alta sıralanmış verileri listelemek ve seçim yapmak için kullanılır.

Özellik Açıklama
Items Listedeki elemanlar koleksiyonu
SelectedItem Seçili eleman (object)
SelectedIndex Seçili elemanın indeksi (0'dan başlar)
SelectionMode Tekli/Çoklu seçim modu
Sorted Alfabetik sıralama
ListBox - Eleman Ekleme
// Tek eleman ekleme
listBox1.Items.Add("Elma");
listBox1.Items.Add("Armut");
listBox1.Items.Add("Muz");

// Birden fazla eleman ekleme
string[] meyveler = { "Çilek", "Karpuz", "Kavun" };
listBox1.Items.AddRange(meyveler);

// TextBox'tan ekleme
private void btnEkle_Click(object sender, EventArgs e)
{
    if (!string.IsNullOrWhiteSpace(txtEleman.Text))
    {
        listBox1.Items.Add(txtEleman.Text);
        txtEleman.Clear();
        txtEleman.Focus();
    }
}
ListBox - Eleman Silme
// Seçili elemanı sil
if (listBox1.SelectedIndex != -1)
{
    listBox1.Items.RemoveAt(listBox1.SelectedIndex);
}

// Belirli bir elemanı sil
listBox1.Items.Remove("Elma");

// İndeks ile sil
listBox1.Items.RemoveAt(0); // İlk elemanı sil

// Tüm elemanları temizle
listBox1.Items.Clear();
ListBox - Seçim İşlemleri
// Seçili elemanı alma
if (listBox1.SelectedItem != null)
{
    string secilen = listBox1.SelectedItem.ToString();
    MessageBox.Show("Seçilen: " + secilen);
}

// Seçim değiştiğinde (SelectedIndexChanged olayı)
private void listBox1_SelectedIndexChanged(object sender, EventArgs e)
{
    if (listBox1.SelectedIndex != -1)
    {
        lblSecim.Text = "Seçilen: " + listBox1.SelectedItem.ToString();
    }
}

// Eleman sayısı
lblToplam.Text = "Toplam: " + listBox1.Items.Count;

2. ComboBox (Açılır Liste)

ComboBox, ListBox'a benzer ama daha az yer kaplar. Açılır menü şeklinde çalışır.

Özellik Açıklama
Items Listedeki elemanlar
SelectedItem Seçili eleman
SelectedIndex Seçili indeks
Text Görünen/yazılan metin
DropDownStyle Simple, DropDown, DropDownList
📌 DropDownStyle Farkları:
  • DropDown: Kullanıcı yeni değer yazabilir
  • DropDownList: Sadece listeden seçim yapılabilir
  • Simple: Liste her zaman açık görünür
ComboBox Kullanımı
// Eleman ekleme
comboBox1.Items.Add("İstanbul");
comboBox1.Items.Add("Ankara");
comboBox1.Items.Add("İzmir");

// Dizi ile ekleme
string[] sehirler = { "Bursa", "Antalya", "Adana" };
comboBox1.Items.AddRange(sehirler);

// Varsayılan seçim
comboBox1.SelectedIndex = 0; // İlk elemanı seç

// Seçili elemanı alma
string sehir = comboBox1.SelectedItem.ToString();

// DropDownList modunda kullanım (sadece listeden seçim)
comboBox1.DropDownStyle = ComboBoxStyle.DropDownList;
ComboBox - Şehir/İlçe Bağlantısı
// İl seçildiğinde ilçeleri yükle
private void cmbIl_SelectedIndexChanged(object sender, EventArgs e)
{
    cmbIlce.Items.Clear();
    
    switch (cmbIl.SelectedItem.ToString())
    {
        case "İstanbul":
            cmbIlce.Items.AddRange(new[] { "Kadıköy", "Beşiktaş", "Üsküdar" });
            break;
        case "Ankara":
            cmbIlce.Items.AddRange(new[] { "Çankaya", "Keçiören", "Mamak" });
            break;
        case "İzmir":
            cmbIlce.Items.AddRange(new[] { "Konak", "Karşıyaka", "Bornova" });
            break;
    }
    
    if (cmbIlce.Items.Count > 0)
        cmbIlce.SelectedIndex = 0;
}

Pratik Örnek: Yapılacaklar Listesi

Todo Uygulaması
// Görev ekleme
private void btnEkle_Click(object sender, EventArgs e)
{
    if (!string.IsNullOrWhiteSpace(txtGorev.Text))
    {
        listGorevler.Items.Add(txtGorev.Text);
        txtGorev.Clear();
        GuncelleSayac();
    }
}

// Seçili görevi sil
private void btnSil_Click(object sender, EventArgs e)
{
    if (listGorevler.SelectedIndex != -1)
    {
        listGorevler.Items.RemoveAt(listGorevler.SelectedIndex);
        GuncelleSayac();
    }
    else
    {
        MessageBox.Show("Lütfen silinecek görevi seçin!");
    }
}

// Tamamlanan görevi taşı
private void btnTamamla_Click(object sender, EventArgs e)
{
    if (listGorevler.SelectedIndex != -1)
    {
        string gorev = listGorevler.SelectedItem.ToString();
        listTamamlanan.Items.Add("✓ " + gorev);
        listGorevler.Items.Remove(gorev);
        GuncelleSayac();
    }
}

private void GuncelleSayac()
{
    lblBekleyen.Text = $"Bekleyen: {listGorevler.Items.Count}";
    lblTamamlanan.Text = $"Tamamlanan: {listTamamlanan.Items.Count}";
}

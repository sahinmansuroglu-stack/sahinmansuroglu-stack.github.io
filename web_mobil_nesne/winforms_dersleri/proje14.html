Proje Hakkında

Günlük yazıları dosyaya kaydeden ve okuyan kişisel günlük uygulaması.

📚 Yeni Konular (Ders 14): File.ReadAllText, File.WriteAllText, StreamReader, StreamWriter, Directory işlemleri

Kod Çözümü

Değişkenler ve Başlangıç
private string gunlukKlasor;

private void Form1_Load(object sender, EventArgs e)
{
    // Belgelerim altında Günlük klasörü
    gunlukKlasor = Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments),
        "Gunluk");
    
    if (!Directory.Exists(gunlukKlasor))
    {
        Directory.CreateDirectory(gunlukKlasor);
    }
    
    dtpTarih.Value = DateTime.Today;
    GunlukleriListele();
    GunuYukle();
}
Günlükleri Listeleme
private void GunlukleriListele()
{
    lstGunlukler.Items.Clear();
    
    var dosyalar = Directory.GetFiles(gunlukKlasor, "*.txt")
        .OrderByDescending(f => f)
        .Take(30);
    
    foreach (string dosya in dosyalar)
    {
        string dosyaAdi = Path.GetFileNameWithoutExtension(dosya);
        
        // 2024-02-10 formatından tarihe çevir
        if (DateTime.TryParse(dosyaAdi, out DateTime tarih))
        {
            lstGunlukler.Items.Add(tarih.ToString("dd MMMM yyyy, dddd"));
        }
    }
}

private void lstGunlukler_SelectedIndexChanged(object sender, EventArgs e)
{
    if (lstGunlukler.SelectedItem == null) return;
    
    string secili = lstGunlukler.SelectedItem.ToString();
    DateTime tarih = DateTime.Parse(secili.Split(',')[0]);
    
    dtpTarih.Value = tarih;
    GunuYukle();
}
Yükleme ve Kaydetme
private string DosyaYolu(DateTime tarih)
{
    return Path.Combine(gunlukKlasor, tarih.ToString("yyyy-MM-dd") + ".txt");
}

private void GunuYukle()
{
    string yol = DosyaYolu(dtpTarih.Value);
    
    if (File.Exists(yol))
    {
        txtGunluk.Text = File.ReadAllText(yol);
        lblDurum.Text = "Mevcut günlük yüklendi";
    }
    else
    {
        txtGunluk.Clear();
        lblDurum.Text = "Yeni günlük";
    }
}

private void btnKaydet_Click(object sender, EventArgs e)
{
    if (string.IsNullOrWhiteSpace(txtGunluk.Text))
    {
        MessageBox.Show("Günlük boş olamaz!");
        return;
    }
    
    string yol = DosyaYolu(dtpTarih.Value);
    
    try
    {
        File.WriteAllText(yol, txtGunluk.Text);
        lblDurum.Text = "Kaydedildi: " + DateTime.Now.ToString("HH:mm");
        GunlukleriListele();
    }
    catch (Exception ex)
    {
        MessageBox.Show("Hata: " + ex.Message);
    }
}

private void dtpTarih_ValueChanged(object sender, EventArgs e)
{
    GunuYukle();
}
Silme ve Dışa Aktarma
private void btnSil_Click(object sender, EventArgs e)
{
    string yol = DosyaYolu(dtpTarih.Value);
    
    if (!File.Exists(yol))
    {
        MessageBox.Show("Silinecek günlük yok!");
        return;
    }
    
    if (MessageBox.Show("Bu günlüğü silmek istiyor musunuz?", "Onay",
        MessageBoxButtons.YesNo, MessageBoxIcon.Warning) == DialogResult.Yes)
    {
        File.Delete(yol);
        txtGunluk.Clear();
        GunlukleriListele();
        lblDurum.Text = "Günlük silindi";
    }
}

private void btnDisaAktar_Click(object sender, EventArgs e)
{
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "Metin Dosyası|*.txt";
    sfd.FileName = "Gunluk_Yedek_" + DateTime.Now.ToString("yyyyMMdd");
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        using (StreamWriter sw = new StreamWriter(sfd.FileName))
        {
            foreach (string dosya in Directory.GetFiles(gunlukKlasor, "*.txt"))
            {
                string tarih = Path.GetFileNameWithoutExtension(dosya);
                string icerik = File.ReadAllText(dosya);
                
                sw.WriteLine("=== " + tarih + " ===");
                sw.WriteLine(icerik);
                sw.WriteLine();
            }
        }
        
        MessageBox.Show("Tüm günlükler dışa aktarıldı!");
    }
}

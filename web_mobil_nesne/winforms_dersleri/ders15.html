🚀 Final Projesi: Kişi Yönetim Sistemi

Bu projede, tüm öğrendiklerinizi bir araya getireceksiniz.

Proje Gereksinimleri:

  • ✅ Kişi ekleme, düzenleme, silme
  • ✅ DataGridView ile listeleme
  • ✅ Dosyaya kaydetme ve okuma (JSON veya CSV)
  • ✅ Arama ve filtreleme
  • ✅ Menü ve ToolStrip
  • ✅ Resim ekleme (profil fotoğrafı)
Kisi Model Sınıfı
public class Kisi
{
    public int Id { get; set; }
    public string Ad { get; set; }
    public string Soyad { get; set; }
    public string Telefon { get; set; }
    public string Email { get; set; }
    public DateTime DogumTarihi { get; set; }
    public string Adres { get; set; }
    public string ResimYolu { get; set; }
    
    public string TamAd => $"{Ad} {Soyad}";
    
    public int Yas
    {
        get
        {
            int yas = DateTime.Now.Year - DogumTarihi.Year;
            if (DateTime.Now.DayOfYear < DogumTarihi.DayOfYear) yas--;
            return yas;
        }
    }
}
Form Yapısı
public partial class frmKisiYonetimi : Form
{
    private List<Kisi> kisiler = new List<Kisi>();
    private string dosyaYolu = "kisiler.json";
    
    public frmKisiYonetimi()
    {
        InitializeComponent();
    }
    
    private void frmKisiYonetimi_Load(object sender, EventArgs e)
    {
        // DataGridView ayarları
        SetupDataGridView();
        
        // Dosyadan yükle
        DosyadanYukle();
        
        // Listeyi göster
        ListeGuncelle();
    }
    
    private void SetupDataGridView()
    {
        dgvKisiler.AutoGenerateColumns = false;
        dgvKisiler.Columns.Add("Id", "ID");
        dgvKisiler.Columns.Add("TamAd", "Ad Soyad");
        dgvKisiler.Columns.Add("Telefon", "Telefon");
        dgvKisiler.Columns.Add("Email", "E-posta");
        dgvKisiler.Columns.Add("Yas", "Yaş");
        
        dgvKisiler.AutoSizeColumnsMode = DataGridViewAutoSizeColumnsMode.Fill;
    }
}
CRUD İşlemleri
// Kişi Ekle
private void btnEkle_Click(object sender, EventArgs e)
{
    if (!DogrulaForm()) return;
    
    Kisi yeniKisi = new Kisi
    {
        Id = kisiler.Count > 0 ? kisiler.Max(k => k.Id) + 1 : 1,
        Ad = txtAd.Text,
        Soyad = txtSoyad.Text,
        Telefon = mtbTelefon.Text,
        Email = txtEmail.Text,
        DogumTarihi = dtpDogum.Value,
        Adres = txtAdres.Text,
        ResimYolu = picProfil.Tag?.ToString()
    };
    
    kisiler.Add(yeniKisi);
    ListeGuncelle();
    FormuTemizle();
    DosyayaKaydet();
    
    MessageBox.Show("Kişi eklendi!", "Bilgi", MessageBoxButtons.OK, MessageBoxIcon.Information);
}

// Kişi Güncelle
private void btnGuncelle_Click(object sender, EventArgs e)
{
    if (dgvKisiler.CurrentRow == null) return;
    
    int id = (int)dgvKisiler.CurrentRow.Cells["Id"].Value;
    Kisi kisi = kisiler.FirstOrDefault(k => k.Id == id);
    
    if (kisi != null)
    {
        kisi.Ad = txtAd.Text;
        kisi.Soyad = txtSoyad.Text;
        kisi.Telefon = mtbTelefon.Text;
        kisi.Email = txtEmail.Text;
        kisi.DogumTarihi = dtpDogum.Value;
        kisi.Adres = txtAdres.Text;
        
        ListeGuncelle();
        DosyayaKaydet();
        MessageBox.Show("Kişi güncellendi!");
    }
}

// Kişi Sil
private void btnSil_Click(object sender, EventArgs e)
{
    if (dgvKisiler.CurrentRow == null) return;
    
    DialogResult sonuc = MessageBox.Show(
        "Bu kişiyi silmek istiyor musunuz?",
        "Silme Onayı",
        MessageBoxButtons.YesNo,
        MessageBoxIcon.Question,
        MessageBoxDefaultButton.Button2);
    
    if (sonuc == DialogResult.Yes)
    {
        int id = (int)dgvKisiler.CurrentRow.Cells["Id"].Value;
        kisiler.RemoveAll(k => k.Id == id);
        ListeGuncelle();
        DosyayaKaydet();
        FormuTemizle();
    }
}
JSON ile Kaydetme/Yükleme
// System.Text.Json kullanarak
using System.Text.Json;

private void DosyayaKaydet()
{
    try
    {
        var options = new JsonSerializerOptions { WriteIndented = true };
        string json = JsonSerializer.Serialize(kisiler, options);
        File.WriteAllText(dosyaYolu, json);
    }
    catch (Exception ex)
    {
        MessageBox.Show("Kayıt hatası: " + ex.Message);
    }
}

private void DosyadanYukle()
{
    try
    {
        if (File.Exists(dosyaYolu))
        {
            string json = File.ReadAllText(dosyaYolu);
            kisiler = JsonSerializer.Deserialize<List<Kisi>>(json) ?? new List<Kisi>();
        }
    }
    catch (Exception ex)
    {
        MessageBox.Show("Yükleme hatası: " + ex.Message);
        kisiler = new List<Kisi>();
    }
}
Arama Fonksiyonu
private void txtArama_TextChanged(object sender, EventArgs e)
{
    string arama = txtArama.Text.ToLower();
    
    var filtrelenmis = kisiler.Where(k =>
        k.Ad.ToLower().Contains(arama) ||
        k.Soyad.ToLower().Contains(arama) ||
        k.Telefon.Contains(arama) ||
        k.Email.ToLower().Contains(arama)
    ).ToList();
    
    dgvKisiler.DataSource = new BindingList<Kisi>(filtrelenmis);
}

❓ Çözülmemiş Problemler

Aşağıdaki projeleri kendiniz çözmeye çalışın. Her biri farklı konuları kapsar.

📝 Problem 1: Yapılacaklar Listesi (Todo App)

Zorluk: ⭐⭐

Konular: ListBox, CheckBox, Dosya İşlemleri

Gereksinimler:

  • Görev ekleme, silme
  • Görevleri tamamlandı olarak işaretleme
  • Görevleri .txt dosyasına kaydetme/yükleme
  • Tamamlanan görevleri gizleme/gösterme filtesi
  • Öncelik seçimi (Düşük, Normal, Yüksek)
💡 İpucu: CheckedListBox kontrolünü kullanabilirsiniz. Her satırı "görev|tamamlandı|öncelik" formatında kaydedin.

📊 Problem 2: Öğrenci Not Takip Sistemi

Zorluk: ⭐⭐⭐

Konular: DataGridView, Hesaplamalar, Raporlama

Gereksinimler:

  • Öğrenci ekleme (Ad, Numara)
  • Her öğrenci için Vize, Final, Ödev notları girişi
  • Ortalama hesaplama (Vize %30, Final %50, Ödev %20)
  • Geçti/Kaldı durumu (50 üstü geçer)
  • Sınıf ortalaması ve istatistikler
  • Verileri dosyaya kaydetme
💡 İpucu: NumericUpDown ile not girişi yapın. DataGridView'da koşullu formatlama ile kalan öğrencileri kırmızı gösterin.

🧮 Problem 3: Hesap Makinesi

Zorluk: ⭐⭐

Konular: Button, Event Handling, Hesaplama Mantığı

Gereksinimler:

  • 0-9 rakam butonları
  • +, -, *, / işlem butonları
  • = (eşittir) ve C (temizle) butonları
  • Ondalık sayı desteği
  • İşlem geçmişi (ListBox)
  • Klavye desteği (KeyPress olayı)
💡 İpucu: Tüm rakam butonları için tek bir event handler kullanın, sender nesnesinden hangi butona basıldığını belirleyin.

🖼️ Problem 4: Resim Düzenleyici

Zorluk: ⭐⭐⭐⭐

Konular: PictureBox, ColorDialog, Graphics

Gereksinimler:

  • Resim açma ve kaydetme
  • Parlaklık/Kontrast ayarı (TrackBar)
  • Resmi döndürme (90°, 180°, 270°)
  • Resmi yatay/dikey çevirme
  • Gri tonlama filtresi
  • Resim boyutlandırma
💡 İpucu: Bitmap sınıfını ve RotateFlip metodunu araştırın. Parlaklık için her pikselin RGB değerlerini değiştirin.

🎮 Problem 5: Yılan Oyunu

Zorluk: ⭐⭐⭐⭐⭐

Konular: Timer, KeyDown, Graphics, Oyun Mantığı

Gereksinimler:

  • Yön tuşları ile yılanı kontrol etme
  • Timer ile hareket
  • Yem yeme ve büyüme
  • Duvara veya kendine çarpınca oyun sonu
  • Skor takibi
  • Yeniden başlat butonu
💡 İpucu: Yılanı List<Point> olarak tutun. Her Tick'te listenin başına yeni kafa ekleyin, yem yenmediyse kuyruktan silin.

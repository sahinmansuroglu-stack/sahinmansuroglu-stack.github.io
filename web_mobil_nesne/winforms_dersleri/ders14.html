System.IO Namespace

Dosya işlemleri için System.IO namespace'ini kullanırız. En önemli sınıflar:

Sınıf Kullanım
File Dosya oluşturma, okuma, yazma, silme
Directory Klasör oluşturma, silme, listeleme
Path Dosya yolu işlemleri
FileInfo Dosya bilgileri
DirectoryInfo Klasör bilgileri
StreamReader Metin okuma (büyük dosyalar)
StreamWriter Metin yazma (büyük dosyalar)
📌 using System.IO; dosyanızın başına eklemeyi unutmayın!

1. Dosya Okuma

Dosya Okuma Yöntemleri
// Tüm içeriği string olarak oku
string icerik = File.ReadAllText(@"C:\dosya.txt");
txtIcerik.Text = icerik;

// Encoding belirterek oku (Türkçe karakterler için)
string icerik = File.ReadAllText(@"C:\dosya.txt", Encoding.UTF8);

// Satır satır oku (dizi olarak)
string[] satirlar = File.ReadAllLines(@"C:\dosya.txt");
foreach (string satir in satirlar)
{
    listBox1.Items.Add(satir);
}

// Byte olarak oku (binary dosyalar)
byte[] byteler = File.ReadAllBytes(@"C:\resim.jpg");
StreamReader ile Okuma (Büyük Dosyalar)
// Büyük dosyalar için StreamReader
using (StreamReader sr = new StreamReader(@"C:\buyukdosya.txt"))
{
    string satir;
    while ((satir = sr.ReadLine()) != null)
    {
        listBox1.Items.Add(satir);
    }
}

// Tüm içeriği oku
using (StreamReader sr = new StreamReader(@"C:\dosya.txt", Encoding.UTF8))
{
    txtIcerik.Text = sr.ReadToEnd();
}

2. Dosya Yazma

Dosya Yazma Yöntemleri
// Tüm içeriği yaz (dosya varsa üzerine yazar)
File.WriteAllText(@"C:\dosya.txt", txtIcerik.Text);

// Encoding ile yaz
File.WriteAllText(@"C:\dosya.txt", txtIcerik.Text, Encoding.UTF8);

// Satır dizisi yaz
string[] satirlar = { "Satır 1", "Satır 2", "Satır 3" };
File.WriteAllLines(@"C:\dosya.txt", satirlar);

// Dosyanın sonuna ekle (append)
File.AppendAllText(@"C:\log.txt", "Yeni satır\n");
File.AppendAllLines(@"C:\log.txt", new[] { "Satır 1", "Satır 2" });
StreamWriter ile Yazma
// Yeni dosya oluştur ve yaz
using (StreamWriter sw = new StreamWriter(@"C:\yeni.txt"))
{
    sw.WriteLine("Birinci satır");
    sw.WriteLine("İkinci satır");
    sw.Write("Satır sonu olmadan");
}

// Dosyanın sonuna ekle (append: true)
using (StreamWriter sw = new StreamWriter(@"C:\log.txt", true, Encoding.UTF8))
{
    sw.WriteLine(DateTime.Now + ": Log mesajı");
}

3. Dosya İşlemleri (Kontrol, Kopyala, Taşı, Sil)

Dosya Kontrolleri
string dosya = @"C:\test.txt";

// Dosya var mı?
if (File.Exists(dosya))
{
    MessageBox.Show("Dosya mevcut");
}

// Dosya bilgileri
FileInfo fi = new FileInfo(dosya);
lblDosyaAdi.Text = fi.Name;
lblBoyut.Text = fi.Length + " byte";
lblOlusturma.Text = fi.CreationTime.ToString();
lblSonDeğişiklik.Text = fi.LastWriteTime.ToString();
lblUzanti.Text = fi.Extension;
Kopyala, Taşı, Sil
// Dosya kopyala
File.Copy(@"C:\kaynak.txt", @"D:\hedef.txt");

// Üzerine yazarak kopyala
File.Copy(@"C:\kaynak.txt", @"D:\hedef.txt", true);

// Dosya taşı (aynı sürücüde hızlı)
File.Move(@"C:\eski.txt", @"C:\yeni.txt");

// Dosya sil
if (File.Exists(@"C:\silinecek.txt"))
{
    File.Delete(@"C:\silinecek.txt");
}

// Dosya adını değiştir
File.Move(@"C:\eski_ad.txt", @"C:\yeni_ad.txt");

4. Klasör İşlemleri

Directory Sınıfı
// Klasör var mı?
if (Directory.Exists(@"C:\Belgeler"))
{
    MessageBox.Show("Klasör mevcut");
}

// Klasör oluştur
Directory.CreateDirectory(@"C:\YeniKlasor");
Directory.CreateDirectory(@"C:\Klasor1\Klasor2\Klasor3"); // İç içe

// Klasör sil (boş olmalı)
Directory.Delete(@"C:\BosKlasor");

// Klasör sil (içindekilerle birlikte)
Directory.Delete(@"C:\DoluKlasor", true);

// Klasördeki dosyaları listele
string[] dosyalar = Directory.GetFiles(@"C:\Belgeler");
foreach (string dosya in dosyalar)
{
    listBox1.Items.Add(Path.GetFileName(dosya));
}

// Belirli uzantıdaki dosyalar
string[] txtDosyalar = Directory.GetFiles(@"C:\Belgeler", "*.txt");

// Alt klasörler dahil
string[] tumDosyalar = Directory.GetFiles(@"C:\Belgeler", "*.*", SearchOption.AllDirectories);

// Alt klasörleri listele
string[] altKlasorler = Directory.GetDirectories(@"C:\Belgeler");

5. Path Sınıfı

Yol İşlemleri
string dosyaYolu = @"C:\Belgeler\rapor.txt";

// Dosya adı (uzantıyla)
string dosyaAdi = Path.GetFileName(dosyaYolu); // "rapor.txt"

// Dosya adı (uzantısız)
string isim = Path.GetFileNameWithoutExtension(dosyaYolu); // "rapor"

// Uzantı
string uzanti = Path.GetExtension(dosyaYolu); // ".txt"

// Klasör yolu
string klasor = Path.GetDirectoryName(dosyaYolu); // "C:\Belgeler"

// Yolları birleştir
string tamYol = Path.Combine(@"C:\Belgeler", "alt", "dosya.txt");
// "C:\Belgeler\alt\dosya.txt"

// Geçici dosya yolu
string geciciDosya = Path.GetTempFileName();

// Geçici klasör
string geciciKlasor = Path.GetTempPath();

// Özel klasörler
string masaustu = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);
string belgeler = Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments);

Pratik Örnek: Not Defteri

Basit Not Defteri
private string mevcutDosya = "";

private void Yeni()
{
    txtIcerik.Clear();
    mevcutDosya = "";
    this.Text = "Yeni - Not Defteri";
}

private void Ac()
{
    OpenFileDialog ofd = new OpenFileDialog();
    ofd.Filter = "Metin Dosyaları|*.txt|Tüm Dosyalar|*.*";
    
    if (ofd.ShowDialog() == DialogResult.OK)
    {
        try
        {
            txtIcerik.Text = File.ReadAllText(ofd.FileName, Encoding.UTF8);
            mevcutDosya = ofd.FileName;
            this.Text = Path.GetFileName(ofd.FileName) + " - Not Defteri";
        }
        catch (Exception ex)
        {
            MessageBox.Show("Dosya açılamadı: " + ex.Message,
                "Hata", MessageBoxButtons.OK, MessageBoxIcon.Error);
        }
    }
}

private void Kaydet()
{
    if (string.IsNullOrEmpty(mevcutDosya))
    {
        FarkliKaydet();
    }
    else
    {
        try
        {
            File.WriteAllText(mevcutDosya, txtIcerik.Text, Encoding.UTF8);
            MessageBox.Show("Kaydedildi!");
        }
        catch (Exception ex)
        {
            MessageBox.Show("Kayıt hatası: " + ex.Message);
        }
    }
}

private void FarkliKaydet()
{
    SaveFileDialog sfd = new SaveFileDialog();
    sfd.Filter = "Metin Dosyası|*.txt";
    sfd.DefaultExt = "txt";
    
    if (sfd.ShowDialog() == DialogResult.OK)
    {
        File.WriteAllText(sfd.FileName, txtIcerik.Text, Encoding.UTF8);
        mevcutDosya = sfd.FileName;
        this.Text = Path.GetFileName(sfd.FileName) + " - Not Defteri";
    }
}

    if (fbd.ShowDialog() == DialogResult.OK)
    {
        listBox1.Items.Clear();

        }
    }
}

Olay (Event) Nedir?

Olaylar, kullanıcı etkileşimleri veya sistem değişiklikleri sonucu tetiklenen aksiyonlardır. Windows Forms'ta her kontrol çeşitli olaylar sunnar.

Kullanıcı Eylemi
(Tıklama, Yazma)
→
Olay Tetiklenir
(Click, KeyPress)
→
Event Handler
Çalışır
📌 Event Handler Oluşturma:
  • Properties panelinde ⚡ (Events) simgesine tıklayın
  • İstediğiniz olaya çift tıklayın
  • Kod dosyasında metot otomatik oluşturulur

1. Click Olayı

Kontrole tıklandığında tetiklenir. En çok kullanılan olaydır.

Click Olayı
private void button1_Click(object sender, EventArgs e)
{
    MessageBox.Show("Butona tıklandı!");
}

// Birden fazla buton için tek handler
private void Hesapla_Click(object sender, EventArgs e)
{
    Button tiklanilan = (Button)sender;
    
    switch (tiklanilan.Name)
    {
        case "btnTopla":
            lblSonuc.Text = (sayi1 + sayi2).ToString();
            break;
        case "btnCikar":
            lblSonuc.Text = (sayi1 - sayi2).ToString();
            break;
        case "btnCarp":
            lblSonuc.Text = (sayi1 * sayi2).ToString();
            break;
    }
}

2. TextChanged Olayı

TextBox içeriği değiştiğinde (her tuşa basıldığında) tetiklenir.

TextChanged - Anlık Arama
private void txtArama_TextChanged(object sender, EventArgs e)
{
    string arama = txtArama.Text.ToLower();
    
    listBox1.Items.Clear();
    
    foreach (string urun in tumUrunler)
    {
        if (urun.ToLower().Contains(arama))
        {
            listBox1.Items.Add(urun);
        }
    }
    
    lblSonuc.Text = $"Bulunan: {listBox1.Items.Count}";
}

// Karakter sayacı
private void txtMesaj_TextChanged(object sender, EventArgs e)
{
    int kalan = 280 - txtMesaj.Text.Length;
    lblKarakter.Text = $"Kalan: {kalan}";
    
    if (kalan < 0)
        lblKarakter.ForeColor = Color.Red;
    else
        lblKarakter.ForeColor = Color.Black;
}

3. KeyPress ve KeyDown Olayları

Klavye tuşlarına basıldığında tetiklenir.

Olay Açıklama Parametre
KeyPress Karakter tuşlarında e.KeyChar
KeyDown Tüm tuşlarda (basıldığında) e.KeyCode
KeyUp Tuş bırakıldığında e.KeyCode
KeyPress - Sadece Rakam Girişi
private void txtSayi_KeyPress(object sender, KeyPressEventArgs e)
{
    // Sadece rakam ve backspace izin ver
    if (!char.IsDigit(e.KeyChar) && e.KeyChar != (char)Keys.Back)
    {
        e.Handled = true; // Tuş basımını iptal et
    }
}

// Sadece harf girişi
private void txtAd_KeyPress(object sender, KeyPressEventArgs e)
{
    if (!char.IsLetter(e.KeyChar) && !char.IsWhiteSpace(e.KeyChar) 
        && e.KeyChar != (char)Keys.Back)
    {
        e.Handled = true;
    }
}
KeyDown - Özel Tuşlar
private void txtMesaj_KeyDown(object sender, KeyEventArgs e)
{
    // Enter ile gönder
    if (e.KeyCode == Keys.Enter)
    {
        btnGonder.PerformClick(); // Butona tıklamış gibi davran
        e.SuppressKeyPress = true; // Bip sesini engelle
    }
    
    // Ctrl+S ile kaydet
    if (e.Control && e.KeyCode == Keys.S)
    {
        Kaydet();
        e.SuppressKeyPress = true;
    }
    
    // Escape ile temizle
    if (e.KeyCode == Keys.Escape)
    {
        txtMesaj.Clear();
    }
}

6. Kod ile Event Bağlama

Dinamik Event Handler
// Form_Load'da event bağlama
private void Form1_Load(object sender, EventArgs e)
{
    button1.Click += Button1_Click;
    textBox1.TextChanged += TextBox1_TextChanged;
    
    // Lambda ile kısa yazım
    button2.Click += (s, args) => MessageBox.Show("Tıklandı!");
}

// Dinamik butonlar için ortak handler
private void DinamikButonlarOlustur()
{
    for (int i = 1; i <= 10; i++)
    {
        Button btn = new Button();
        btn.Text = i.ToString();
        btn.Tag = i; // Veri taşımak için Tag kullan
        btn.Click += Rakam_Click; // Aynı handler'ı bağla
        
        flowLayoutPanel1.Controls.Add(btn);
    }
}

private void Rakam_Click(object sender, EventArgs e)
{
    Button btn = (Button)sender;
    int rakam = (int)btn.Tag;
    txtSonuc.Text += rakam.ToString();
}

1. NumericUpDown

NumericUpDown, sayısal değer girişi için kullanılır. Ok tuşlarıyla veya yazarak değer girilebilir.

Özellik Açıklama
Value Mevcut değer (decimal)
Minimum En küçük değer
Maximum En büyük değer
Increment Artış/azalış miktarı
DecimalPlaces Ondalık basamak sayısı
NumericUpDown Kullanımı
// Özellikleri ayarlama
numYas.Minimum = 0;
numYas.Maximum = 120;
numYas.Value = 18;
numYas.Increment = 1;

// Fiyat için ondalıklı kullanım
numFiyat.Minimum = 0;
numFiyat.Maximum = 10000;
numFiyat.DecimalPlaces = 2;
numFiyat.Increment = 0.50M;

// Değeri okuma
int yas = (int)numYas.Value;
decimal fiyat = numFiyat.Value;

// Değer değiştiğinde
private void numAdet_ValueChanged(object sender, EventArgs e)
{
    decimal toplam = numAdet.Value * numFiyat.Value;
    lblToplam.Text = toplam.ToString("C"); // Para formatı
}

2. DateTimePicker

DateTimePicker, tarih ve saat seçimi için kullanılır. Takvim görünümü sunar.

Özellik Açıklama
Value Seçili tarih (DateTime)
MinDate Seçilebilecek en eski tarih
MaxDate Seçilebilecek en yeni tarih
Format Short, Long, Time, Custom
CustomFormat Özel format dizesi
ShowUpDown Takvim yerine ok tuşları
DateTimePicker Kullanımı
// Bugünün tarihini ayarla
dtpTarih.Value = DateTime.Now;

// Sınırları belirleme
dtpDogum.MinDate = new DateTime(1900, 1, 1);
dtpDogum.MaxDate = DateTime.Today;

// Format ayarlama
dtpTarih.Format = DateTimePickerFormat.Short; // 05.02.2024
dtpTarih.Format = DateTimePickerFormat.Long;  // 5 Şubat 2024 Pazartesi

// Özel format
dtpTarih.Format = DateTimePickerFormat.Custom;
dtpTarih.CustomFormat = "dd/MM/yyyy";

// Sadece saat için
dtpSaat.Format = DateTimePickerFormat.Custom;
dtpSaat.CustomFormat = "HH:mm";
dtpSaat.ShowUpDown = true;

// Değeri okuma
DateTime secilen = dtpTarih.Value;
lblTarih.Text = secilen.ToString("dd MMMM yyyy");

// Yaş hesaplama
int yas = DateTime.Now.Year - dtpDogum.Value.Year;
if (DateTime.Now.DayOfYear < dtpDogum.Value.DayOfYear)
    yas--;
Tarih Formatları
DateTime tarih = DateTime.Now;

tarih.ToString("dd/MM/yyyy")      // 05/02/2024
tarih.ToString("dd.MM.yyyy")      // 05.02.2024
tarih.ToString("yyyy-MM-dd")      // 2024-02-05
tarih.ToString("dd MMMM yyyy")    // 05 Şubat 2024
tarih.ToString("dddd")            // Pazartesi
tarih.ToString("HH:mm:ss")        // 14:30:00
tarih.ToString("hh:mm tt")        // 02:30 PM

3. MaskedTextBox

MaskedTextBox, belirli bir formatta veri girişi zorlar (telefon, TC no, tarih vb.).

Özellik Açıklama
Mask Giriş formatını belirleyen maske
Text Görünen metin (maskeli)
TextMaskFormat Maske/literal dahil mi?

Maske Karakterleri

Karakter Açıklama
0 Rakam (zorunlu)
9 Rakam veya boşluk (opsiyonel)
L Harf (zorunlu)
? Harf (opsiyonel)
A Alfanumerik (zorunlu)
& Herhangi karakter (zorunlu)
MaskedTextBox Örnekleri
// Telefon numarası
mtbTelefon.Mask = "(000) 000 00 00";

// TC Kimlik No
mtbTCNo.Mask = "00000000000";

// Tarih
mtbTarih.Mask = "00/00/0000";

// Saat
mtbSaat.Mask = "00:00";

// Posta Kodu
mtbPostaKodu.Mask = "00000";

// Plaka
mtbPlaka.Mask = "00 L 0000";

// Kredi Kartı
mtbKart.Mask = "0000-0000-0000-0000";

// Sadece rakamları alma
string telefon = mtbTelefon.Text.Replace("(", "")
                                 .Replace(")", "")
                                 .Replace(" ", "");

4. ProgressBar

ProgressBar, bir işlemin ilerleme durumunu görsel olarak gösterir.

%65 Tamamlandı

ProgressBar Kullanımı
// Temel ayarlar
progressBar1.Minimum = 0;
progressBar1.Maximum = 100;
progressBar1.Value = 0;

// İlerleme güncelleme
progressBar1.Value = 50;

// Artırma
progressBar1.Value += 10;

// Sonsuz animasyon (belirsiz süre)
progressBar1.Style = ProgressBarStyle.Marquee;

// Dosya kopyalama örneği
for (int i = 0; i <= 100; i++)
{
    progressBar1.Value = i;
    lblYuzde.Text = i + "%";
    Application.DoEvents(); // UI'ın güncellenmesini sağla
    Thread.Sleep(50);
}

Pratik Örnek: Kayıt Formu

Üyelik Kayıt Formu
private void btnKaydet_Click(object sender, EventArgs e)
{
    // Doğrulama
    if (string.IsNullOrWhiteSpace(txtAd.Text))
    {
        MessageBox.Show("Ad boş olamaz!");
        return;
    }
    
    if (!mtbTelefon.MaskCompleted)
    {
        MessageBox.Show("Telefon numarasını eksiksiz girin!");
        return;
    }
    
    // Yaş kontrolü
    int yas = DateTime.Now.Year - dtpDogum.Value.Year;
    if (yas < 18)
    {
        MessageBox.Show("18 yaşından küçükler kayıt olamaz!");
        return;
    }
    
    // Kayıt özeti
    string ozet = $@"
=== KAYIT BİLGİLERİ ===
Ad: {txtAd.Text}
Telefon: {mtbTelefon.Text}
Doğum Tarihi: {dtpDogum.Value:dd.MM.yyyy}
Yaş: {yas}
Adet: {numAdet.Value}
";

    MessageBox.Show(ozet, "Kayıt Başarılı");
}

1 Doğum tarihi seçildiğinde yaşı otomatik hesaplayan bir uygulama yapın. 
private void dtpDogum_ValueChanged(object sender, EventArgs e)
{
    int yas = DateTime.Now.Year - dtpDogum.Value.Year;
    if (DateTime.Now.DayOfYear < dtpDogum.Value.DayOfYear)
        yas--;
    
    lblYas.Text = "Yaşınız: " + yas;
}
2 Adet ve birim fiyat girilerek toplam tutarı hesaplayan uygulama yapın. 
private void Hesapla()
{
    decimal adet = numAdet.Value;
    decimal fiyat = numFiyat.Value;
    decimal toplam = adet * fiyat;
    
    lblToplam.Text = "Toplam: " + toplam.ToString("C");
}

private void numAdet_ValueChanged(object sender, EventArgs e) => Hesapla();
private void numFiyat_ValueChanged(object sender, EventArgs e) => Hesapla();
3 TC Kimlik No ve telefon girişi için MaskedTextBox kullanan bir form yapın. 
// Form_Load'da
mtbTCNo.Mask = "00000000000";
mtbTelefon.Mask = "(000) 000 00 00";

// Kaydet butonunda
private void btnKaydet_Click(object sender, EventArgs e)
{
    if (!mtbTCNo.MaskCompleted || !mtbTelefon.MaskCompleted)
    {
        MessageBox.Show("Tüm alanları doldurun!");
        return;
    }
    
    MessageBox.Show($"TC: {mtbTCNo.Text}\nTel: {mtbTelefon.Text}");
}

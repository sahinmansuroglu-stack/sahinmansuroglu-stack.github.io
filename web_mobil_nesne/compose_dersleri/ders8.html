Basit ViewModel Örneği
// Model
data class Todo(
    val id: Int,
    val baslik: String,
    val tamamlandi: Boolean = false
)

// UI State
data class TodoUiState(
    val todoList: List<Todo> = emptyList(),
    val yukleniyor: Boolean = false,
    val hata: String? = null
)

// ViewModel
class TodoViewModel : ViewModel() {
    private val _uiState = MutableStateFlow(TodoUiState())
    val uiState: StateFlow<TodoUiState> = _uiState.asStateFlow()
    
    private var nextId = 1
    
    fun todoEkle(baslik: String) {
        val yeniTodo = Todo(id = nextId++, baslik = baslik)
        _uiState.update { currentState ->
            currentState.copy(
                todoList = currentState.todoList + yeniTodo
            )
        }
    }
    
    fun todoSil(todoId: Int) {
        _uiState.update { currentState ->
            currentState.copy(
                todoList = currentState.todoList.filter { it.id != todoId }
            )
        }
    }
    
    fun todoTamamla(todoId: Int) {
        _uiState.update { currentState ->
            currentState.copy(
                todoList = currentState.todoList.map { todo ->
                    if (todo.id == todoId) {
                        todo.copy(tamamlandi = !todo.tamamlandi)
                    } else {
                        todo
                    }
                }
            )
        }
    }
}

// Composable (View)
@Composable
fun TodoScreen(
    viewModel: TodoViewModel = viewModel()
) {
    val uiState by viewModel.uiState.collectAsState()
    var yeniTodo by remember { mutableStateOf("") }
    
    Column(modifier = Modifier.padding(16.dp)) {
        // Yeni todo ekleme
        Row(
            modifier = Modifier.fillMaxWidth(),
            verticalAlignment = Alignment.CenterVertically
        ) {
            OutlinedTextField(
                value = yeniTodo,
                onValueChange = { yeniTodo = it },
                label = { Text("Yeni görev") },
                modifier = Modifier.weight(1f)
            )
            
            Spacer(modifier = Modifier.width(8.dp))
            
            Button(
                onClick = {
                    if (yeniTodo.isNotBlank()) {
                        viewModel.todoEkle(yeniTodo)
                        yeniTodo = ""
                    }
                }
            ) {
                Icon(Icons.Default.Add, null)
            }
        }
        
        Spacer(modifier = Modifier.height(16.dp))
        
        // Todo listesi
        LazyColumn {
            items(uiState.todoList, key = { it.id }) { todo ->
                TodoItem(
                    todo = todo,
                    onTamamla = { viewModel.todoTamamla(todo.id) },
                    onSil = { viewModel.todoSil(todo.id) }
                )
            }
        }
    }
}

@Composable
fun TodoItem(
    todo: Todo,
    onTamamla: () -> Unit,
    onSil: () -> Unit
) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(vertical = 4.dp)
    ) {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(12.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            Checkbox(
                checked = todo.tamamlandi,
                onCheckedChange = { onTamamla() }
            )
            
            Text(
                text = todo.baslik,
                modifier = Modifier.weight(1f),
                textDecoration = if (todo.tamamlandi) 
                    TextDecoration.LineThrough else TextDecoration.None
            )
            
            IconButton(onClick = onSil) {
                Icon(Icons.Default.Delete, "Sil", tint = Color.Red)
            }
        }
    }
}

Loading ve Hata Durumları
class UrunViewModel : ViewModel() {
    private val _uiState = MutableStateFlow(UrunUiState())
    val uiState: StateFlow<UrunUiState> = _uiState.asStateFlow()
    
    init {
        urunleriYukle()
    }
    
    fun urunleriYukle() {
        viewModelScope.launch {
            _uiState.update { it.copy(yukleniyor = true) }
            
            try {
                // Simüle edilmiş API çağrısı
                delay(2000)
                val urunler = listOf(
                    Urun(1, "Laptop", 15000.0),
                    Urun(2, "Telefon", 8000.0),
                    Urun(3, "Tablet", 5000.0)
                )
                
                _uiState.update { 
                    it.copy(
                        urunler = urunler,
                        yukleniyor = false
                    ) 
                }
            } catch (e: Exception) {
                _uiState.update { 
                    it.copy(
                        hata = e.message,
                        yukleniyor = false
                    ) 
                }
            }
        }
    }
    
    fun hataTemizle() {
        _uiState.update { it.copy(hata = null) }
    }
}

data class UrunUiState(
    val urunler: List<Urun> = emptyList(),
    val yukleniyor: Boolean = false,
    val hata: String? = null
)

@Composable
fun UrunScreen(viewModel: UrunViewModel = viewModel()) {
    val uiState by viewModel.uiState.collectAsState()
    
    Box(modifier = Modifier.fillMaxSize()) {
        when {
            uiState.yukleniyor -> {
                CircularProgressIndicator(
                    modifier = Modifier.align(Alignment.Center)
                )
            }
            uiState.hata != null -> {
                Column(
                    modifier = Modifier.align(Alignment.Center),
                    horizontalAlignment = Alignment.CenterHorizontally
                ) {
                    Text("Hata: ${uiState.hata}")
                    Button(onClick = { viewModel.urunleriYukle() }) {
                        Text("Tekrar Dene")
                    }
                }
            }
            else -> {
                LazyColumn(contentPadding = PaddingValues(16.dp)) {
                    items(uiState.urunler) { urun ->
                        UrunKart(urun)
                    }
                }
            }
        }
    }
}

Sealed Class ile Event Handling

One-Time Events
sealed class LoginEvent {
    data class Success(val userId: Int) : LoginEvent()
    data class Error(val message: String) : LoginEvent()
    object Loading : LoginEvent()
}

class LoginViewModel : ViewModel() {
    private val _loginState = MutableStateFlow<LoginEvent?>(null)
    val loginState: StateFlow<LoginEvent?> = _loginState.asStateFlow()
    
    var email by mutableStateOf("")
        private set
    
    var sifre by mutableStateOf("")
        private set
    
    fun emailGuncelle(yeniEmail: String) {
        email = yeniEmail
    }
    
    fun sifreGuncelle(yeniSifre: String) {
        sifre = yeniSifre
    }
    
    fun girisYap() {
        viewModelScope.launch {
            _loginState.value = LoginEvent.Loading
            
            delay(1500) // Simüle
            
            if (email == "test@test.com" && sifre == "123456") {
                _loginState.value = LoginEvent.Success(userId = 1)
            } else {
                _loginState.value = LoginEvent.Error("Geçersiz kullanıcı adı veya şifre")
            }
        }
    }
    
    fun eventTemizle() {
        _loginState.value = null
    }
}

@Composable
fun LoginScreen(
    viewModel: LoginViewModel = viewModel(),
    onLoginSuccess: (Int) -> Unit
) {
    val loginState by viewModel.loginState.collectAsState()
    val snackbarHostState = remember { SnackbarHostState() }
    
    // Event handling
    LaunchedEffect(loginState) {
        when (val state = loginState) {
            is LoginEvent.Success -> {
                onLoginSuccess(state.userId)
                viewModel.eventTemizle()
            }
            is LoginEvent.Error -> {
                snackbarHostState.showSnackbar(state.message)
                viewModel.eventTemizle()
            }
            else -> {}
        }
    }
    
    Scaffold(snackbarHost = { SnackbarHost(snackbarHostState) }) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
                .padding(16.dp),
            verticalArrangement = Arrangement.Center
        ) {
            OutlinedTextField(
                value = viewModel.email,
                onValueChange = { viewModel.emailGuncelle(it) },
                label = { Text("E-posta") },
                modifier = Modifier.fillMaxWidth()
            )
            
            Spacer(modifier = Modifier.height(8.dp))
            
            OutlinedTextField(
                value = viewModel.sifre,
                onValueChange = { viewModel.sifreGuncelle(it) },
                label = { Text("Şifre") },
                visualTransformation = PasswordVisualTransformation(),
                modifier = Modifier.fillMaxWidth()
            )
            
            Spacer(modifier = Modifier.height(16.dp))
            
            Button(
                onClick = { viewModel.girisYap() },
                modifier = Modifier.fillMaxWidth(),
                enabled = loginState != LoginEvent.Loading
            ) {
                if (loginState == LoginEvent.Loading) {
                    CircularProgressIndicator(
                        modifier = Modifier.size(24.dp),
                        color = Color.White
                    )
                } else {
                    Text("Giriş Yap")
                }
            }
        }
    }
}

Column vs LazyColumn

Fark
// ❌ Column: Tüm öğeleri aynı anda render eder
// Büyük listelerde performans sorunu!
@Composable
fun ColumnIleList(items: List<String>) {
    Column(modifier = Modifier.verticalScroll(rememberScrollState())) {
        items.forEach { item ->
            Text(item)  // Hepsi aynı anda oluşturulur
        }
    }
}

// ✅ LazyColumn: Sadece görünen öğeleri render eder
// RecyclerView benzeri performans!
@Composable
fun LazyColumnIleList(items: List<String>) {
    LazyColumn {
        items(items) { item ->
            Text(item)  // Sadece ekrandakiler oluşturulur
        }
    }
}

LazyColumn Temel Kullanım

LazyColumn Örnekleri
@Composable
fun BasitListe() {
    val isimler = listOf("Ali", "Ayşe", "Mehmet", "Fatma", "Ahmet")
    
    LazyColumn(
        modifier = Modifier.fillMaxSize(),
        contentPadding = PaddingValues(16.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        // Tek öğe
        item {
            Text(
                "Kullanıcılar",
                style = MaterialTheme.typography.headlineMedium
            )
        }
        
        // Liste öğeleri
        items(isimler) { isim ->
            KisiKarti(isim = isim)
        }
        
        // Index ile
        itemsIndexed(isimler) { index, isim ->
            Text("${index + 1}. $isim")
        }
        
        // Sabit sayıda öğe
        items(5) { index ->
            Text("Öğe $index")
        }
    }
}

@Composable
fun KisiKarti(isim: String) {
    Card(
        modifier = Modifier.fillMaxWidth()
    ) {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            Icon(
                Icons.Default.Person,
                contentDescription = null,
                modifier = Modifier.size(40.dp)
            )
            Spacer(modifier = Modifier.width(12.dp))
            Text(isim, style = MaterialTheme.typography.bodyLarge)
        }
    }
}

LazyRow

Yatay Liste
@Composable
fun YatayKategoriListesi() {
    val kategoriler = listOf(
        "🍕 Pizza", "🍔 Burger", "🍣 Sushi", 
        "🥗 Salata", "🍜 Noodle", "🌮 Taco"
    )
    
    LazyRow(
        contentPadding = PaddingValues(horizontal = 16.dp),
        horizontalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        items(kategoriler) { kategori ->
            FilterChip(
                selected = false,
                onClick = { },
                label = { Text(kategori) }
            )
        }
    }
}

@Composable
fun ResimGalerisi() {
    val resimler = (1..20).toList()
    
    LazyRow(
        contentPadding = PaddingValues(16.dp),
        horizontalArrangement = Arrangement.spacedBy(12.dp)
    ) {
        items(resimler) { index ->
            Card(
                modifier = Modifier.size(150.dp)
            ) {
                Box(
                    modifier = Modifier
                        .fillMaxSize()
                        .background(
                            Brush.linearGradient(
                                listOf(Color.Blue, Color.Cyan)
                            )
                        ),
                    contentAlignment = Alignment.Center
                ) {
                    Text(
                        "Resim $index",
                        color = Color.White
                    )
                }
            }
        }
    }
}

LazyVerticalGrid

Grid Görünümü
@Composable
fun UrunGrid() {
    val urunler = (1..20).map { "Ürün $it" }
    
    LazyVerticalGrid(
        columns = GridCells.Fixed(2),  // Sabit 2 sütun
        contentPadding = PaddingValues(16.dp),
        horizontalArrangement = Arrangement.spacedBy(12.dp),
        verticalArrangement = Arrangement.spacedBy(12.dp)
    ) {
        items(urunler) { urun ->
            UrunKarti(urun)
        }
    }
}

@Composable
fun AdaptiveGrid() {
    LazyVerticalGrid(
        columns = GridCells.Adaptive(minSize = 128.dp),  // Minimum genişlik
        contentPadding = PaddingValues(8.dp),
        horizontalArrangement = Arrangement.spacedBy(8.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        items(50) { index ->
            Card(
                modifier = Modifier
                    .fillMaxWidth()
                    .aspectRatio(1f)
            ) {
                Box(
                    contentAlignment = Alignment.Center,
                    modifier = Modifier.fillMaxSize()
                ) {
                    Text("#$index")
                }
            }
        }
    }
}

@Composable
fun UrunKarti(urun: String) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .aspectRatio(0.8f)
    ) {
        Column {
            Box(
                modifier = Modifier
                    .fillMaxWidth()
                    .weight(1f)
                    .background(Color.LightGray),
                contentAlignment = Alignment.Center
            ) {
                Icon(
                    Icons.Default.ShoppingCart,
                    contentDescription = null,
                    modifier = Modifier.size(48.dp)
                )
            }
            Text(
                text = urun,
                modifier = Modifier.padding(8.dp),
                style = MaterialTheme.typography.bodyMedium
            )
        }
    }
}

Sticky Headers

Yapışkan Başlıklar
@OptIn(ExperimentalFoundationApi::class)
@Composable
fun GrupluListe() {
    val kisiler = mapOf(
        "A" to listOf("Ali", "Ayşe", "Ahmet"),
        "B" to listOf("Burak", "Beyza"),
        "C" to listOf("Can", "Ceren", "Cem"),
        "D" to listOf("Deniz", "Dilan")
    )
    
    LazyColumn(
        modifier = Modifier.fillMaxSize()
    ) {
        kisiler.forEach { (harf, isimler) ->
            // Sticky header
            stickyHeader {
                Text(
                    text = harf,
                    modifier = Modifier
                        .fillMaxWidth()
                        .background(MaterialTheme.colorScheme.primaryContainer)
                        .padding(12.dp),
                    style = MaterialTheme.typography.titleMedium,
                    fontWeight = FontWeight.Bold
                )
            }
            
            // Grup içindeki öğeler
            items(isimler) { isim ->
                Text(
                    text = isim,
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(16.dp)
                )
                HorizontalDivider()
            }
        }
    }
}

Key Kullanımı

Performans için Key
data class Kullanici(
    val id: Int,
    val isim: String
)

@Composable
fun KeyliListe() {
    val kullanicilar = remember {
        mutableStateListOf(
            Kullanici(1, "Ali"),
            Kullanici(2, "Veli"),
            Kullanici(3, "Ayşe")
        )
    }
    
    LazyColumn {
        items(
            items = kullanicilar,
            key = { it.id }  // Benzersiz key
        ) { kullanici ->
            KullaniciSatiri(
                kullanici = kullanici,
                onSil = { kullanicilar.remove(kullanici) }
            )
        }
    }
}

@Composable
fun KullaniciSatiri(
    kullanici: Kullanici,
    onSil: () -> Unit
) {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .padding(16.dp)
            .animateItemPlacement(),  // Animasyonlu yerleşim
        verticalAlignment = Alignment.CenterVertically
    ) {
        Text(kullanici.isim, modifier = Modifier.weight(1f))
        IconButton(onClick = onSil) {
            Icon(Icons.Default.Delete, "Sil")
        }
    }
}

Column - Dikey Yerleşim

Column Örnekleri
@Composable
fun ColumnOrnekleri() {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        verticalArrangement = Arrangement.SpaceEvenly,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text("Birinci")
        Text("İkinci")
        Text("Üçüncü")
    }
}

// verticalArrangement seçenekleri:
// - Arrangement.Top (varsayılan)
// - Arrangement.Center
// - Arrangement.Bottom
// - Arrangement.SpaceBetween
// - Arrangement.SpaceAround
// - Arrangement.SpaceEvenly
// - Arrangement.spacedBy(8.dp)

// horizontalAlignment seçenekleri:
// - Alignment.Start
// - Alignment.CenterHorizontally
// - Alignment.End

Row - Yatay Yerleşim

Row Örnekleri
@Composable
fun RowOrnekleri() {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .padding(16.dp),
        horizontalArrangement = Arrangement.SpaceBetween,
        verticalAlignment = Alignment.CenterVertically
    ) {
        Icon(Icons.Default.Home, contentDescription = null)
        Text("Ana Sayfa")
        Icon(Icons.Default.ArrowForward, contentDescription = null)
    }
}

@Composable
fun KartvizitSatiri() {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .padding(12.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        // Profil resmi
        Image(
            painter = painterResource(R.drawable.avatar),
            contentDescription = null,
            modifier = Modifier
                .size(48.dp)
                .clip(CircleShape)
        )
        
        Spacer(modifier = Modifier.width(12.dp))
        
        // İsim ve bilgi (weight ile esnek genişlik)
        Column(modifier = Modifier.weight(1f)) {
            Text("Ahmet Yılmaz", fontWeight = FontWeight.Bold)
            Text("Android Geliştirici", color = Color.Gray)
        }
        
        // Sağdaki ikon
        IconButton(onClick = { }) {
            Icon(Icons.Default.MoreVert, null)
        }
    }
}

Box - Üst Üste Yerleşim

Box Örnekleri
@Composable
fun BoxOrnekleri() {
    Box(
        modifier = Modifier
            .size(200.dp)
            .background(Color.LightGray)
    ) {
        // Arka plandaki resim
        Image(
            painter = painterResource(R.drawable.background),
            contentDescription = null,
            contentScale = ContentScale.Crop,
            modifier = Modifier.fillMaxSize()
        )
        
        // Üstteki metin
        Text(
            text = "Overlay Metin",
            color = Color.White,
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .background(Color.Black.copy(alpha = 0.5f))
                .padding(8.dp)
        )
        
        // Köşedeki badge
        Badge(
            modifier = Modifier.align(Alignment.TopEnd)
        ) {
            Text("3")
        }
    }
}

@Composable
fun ProfilKarti() {
    Box(
        modifier = Modifier.fillMaxWidth(),
        contentAlignment = Alignment.Center
    ) {
        // Arka plan kartı
        Card(
            modifier = Modifier
                .fillMaxWidth()
                .padding(top = 40.dp)
        ) {
            Column(
                modifier = Modifier.padding(
                    top = 50.dp,
                    bottom = 16.dp
                ),
                horizontalAlignment = Alignment.CenterHorizontally
            ) {
                Text("Mehmet Demir", style = MaterialTheme.typography.titleLarge)
                Text("@mehmetd", color = Color.Gray)
            }
        }
        
        // Üstteki profil resmi
        Image(
            painter = painterResource(R.drawable.avatar),
            contentDescription = null,
            modifier = Modifier
                .size(80.dp)
                .clip(CircleShape)
                .border(3.dp, MaterialTheme.colorScheme.primary, CircleShape)
                .align(Alignment.TopCenter)
        )
    }
}

Modifier Zinciri

Modifier Kullanımı
@Composable
fun ModifierOrnekleri() {
    Text(
        text = "Modifier Örneği",
        modifier = Modifier
            // Boyut modifierleri
            .fillMaxWidth()           // Genişliği doldur
            .height(60.dp)            // Sabit yükseklik
            // .size(100.dp)          // Hem genişlik hem yükseklik
            // .wrapContentSize()     // İçeriğe göre boyut
            
            // Padding (iç boşluk)
            .padding(16.dp)            // Tüm kenarlar
            // .padding(horizontal = 16.dp, vertical = 8.dp)
            // .padding(start = 8.dp, end = 8.dp)
            
            // Arka plan
            .background(
                color = Color.Blue,
                shape = RoundedCornerShape(8.dp)
            )
            
            // Kenarlık
            .border(
                width = 2.dp,
                color = Color.Red,
                shape = RoundedCornerShape(8.dp)
            )
            
            // Tıklanabilirlik
            .clickable { /* onClick */ }
    )
}

@Composable
fun ModifierSirasi() {
    // Sıra önemli! Padding ve background sırası sonucu etkiler
    
    Row {
        // Önce padding, sonra background
        Box(
            modifier = Modifier
                .padding(16.dp)
                .background(Color.Red)
                .size(50.dp)
        )
        
        // Önce background, sonra padding
        Box(
            modifier = Modifier
                .background(Color.Blue)
                .padding(16.dp)
                .size(50.dp)
        )
    }
}
⚠️ Modifier Sırası Önemli!
Modifier'lar yukarıdan aşağıya sırayla uygulanır. Örneğin:
• .padding().background() → İç boşluk arka planın dışında
• .background().padding() → İç boşluk arka planın içinde

Yaygın Modifier'lar

Sık Kullanılan Modifier'lar
// Boyut
Modifier.fillMaxSize()
Modifier.fillMaxWidth()
Modifier.fillMaxHeight()
Modifier.size(100.dp)
Modifier.width(100.dp)
Modifier.height(100.dp)
Modifier.wrapContentSize()
Modifier.aspectRatio(16f/9f)

// Boşluk
Modifier.padding(16.dp)
Modifier.padding(horizontal = 16.dp)
Modifier.offset(x = 10.dp, y = 5.dp)

// Görünüm
Modifier.background(Color.Red)
Modifier.border(1.dp, Color.Black)
Modifier.clip(RoundedCornerShape(8.dp))
Modifier.shadow(4.dp)
Modifier.alpha(0.5f)

// Etkileşim
Modifier.clickable { }
Modifier.selectable(selected = true) { }
Modifier.toggleable(value = true) { }
Modifier.scrollable(state, Orientation.Vertical)

// Kaydırma
Modifier.horizontalScroll(rememberScrollState())
Modifier.verticalScroll(rememberScrollState())

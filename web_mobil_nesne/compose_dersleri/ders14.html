Data Layer

Not Entity
@Entity(tableName = "notlar")
data class Not(
    @PrimaryKey(autoGenerate = true)
    val id: Int = 0,
    val baslik: String,
    val icerik: String,
    val renk: Int = 0,  // Renk indeksi
    val favori: Boolean = false,
    val olusturmaTarihi: Long = System.currentTimeMillis(),
    val guncellemeTarihi: Long = System.currentTimeMillis()
)
NotDao
@Dao
interface NotDao {
    @Query("SELECT * FROM notlar ORDER BY guncellemeTarihi DESC")
    fun tumNotlar(): Flow<List<Not>>
    
    @Query("SELECT * FROM notlar WHERE favori = 1 ORDER BY guncellemeTarihi DESC")
    fun favoriNotlar(): Flow<List<Not>>
    
    @Query("SELECT * FROM notlar WHERE baslik LIKE '%' || :sorgu || '%' OR icerik LIKE '%' || :sorgu || '%'")
    fun araNotlar(sorgu: String): Flow<List<Not>>
    
    @Query("SELECT * FROM notlar WHERE id = :notId")
    suspend fun getirNot(notId: Int): Not?
    
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun ekle(not: Not): Long
    
    @Update
    suspend fun guncelle(not: Not)
    
    @Delete
    suspend fun sil(not: Not)
    
    @Query("UPDATE notlar SET favori = NOT favori WHERE id = :notId")
    suspend fun favoriToggle(notId: Int)
}
Repository
class NotRepository(private val dao: NotDao) {
    val tumNotlar = dao.tumNotlar()
    val favoriNotlar = dao.favoriNotlar()
    
    fun ara(sorgu: String) = dao.araNotlar(sorgu)
    
    suspend fun getir(id: Int) = dao.getirNot(id)
    suspend fun ekle(not: Not) = dao.ekle(not)
    suspend fun guncelle(not: Not) = dao.guncelle(not)
    suspend fun sil(not: Not) = dao.sil(not)
    suspend fun favoriToggle(id: Int) = dao.favoriToggle(id)
}

val notRenkleri = listOf(

    Color(0xFF80DEEA),  // Mavi

@Composable
fun NotCard(
    not: Not,
    onClick: () -> Unit,
    onFavoriClick: () -> Unit,
    onSilClick: () -> Unit,
    modifier: Modifier = Modifier
) {
    Card(
        modifier = modifier
            .fillMaxWidth()
            .clickable { onClick() },
        colors = CardDefaults.cardColors(
            containerColor = notRenkleri.getOrElse(not.renk) { Color.White }
        ),
        elevation = CardDefaults.cardElevation(defaultElevation = 2.dp)
    ) {
        Column(modifier = Modifier.padding(16.dp)) {
            Row(verticalAlignment = Alignment.CenterVertically) {
                Text(
                    text = not.baslik,
                    style = MaterialTheme.typography.titleMedium,
                    fontWeight = FontWeight.Bold,
                    maxLines = 1,
                    overflow = TextOverflow.Ellipsis,
                    modifier = Modifier.weight(1f)
                )
                
                IconButton(onClick = onFavoriClick) {
                    Icon(
                        imageVector = if (not.favori) 
                            Icons.Default.Favorite else Icons.Default.FavoriteBorder,
                        contentDescription = "Favori",
                        tint = if (not.favori) Color.Red else Color.Gray
                    )
                }
            }
            
            Spacer(modifier = Modifier.height(8.dp))
            
            Text(
                text = not.icerik,
                style = MaterialTheme.typography.bodyMedium,
                maxLines = 4,
                overflow = TextOverflow.Ellipsis,
                color = Color.DarkGray
            )
            
            Spacer(modifier = Modifier.height(12.dp))
            
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceBetween,
                verticalAlignment = Alignment.CenterVertically
            ) {
                Text(
                    text = formatTarih(not.guncellemeTarihi),
                    style = MaterialTheme.typography.bodySmall,
                    color = Color.Gray
                )
                
                IconButton(
                    onClick = onSilClick,
                    modifier = Modifier.size(24.dp)
                ) {
                    Icon(
                        Icons.Default.Delete,
                        contentDescription = "Sil",
                        tint = Color.Gray,
                        modifier = Modifier.size(18.dp)
                    )
                }
            }
        }
    }
}

fun formatTarih(timestamp: Long): String {
    val sdf = SimpleDateFormat("dd MMM yyyy, HH:mm", Locale("tr"))
    return sdf.format(Date(timestamp))
}

Home Screen

HomeViewModel
data class HomeUiState(
    val notlar: List<Not> = emptyList(),
    val aramaSorgusu: String = "",
    val sadecesFavoriler: Boolean = false,
    val yukleniyor: Boolean = false
)

class HomeViewModel(
    private val repository: NotRepository
) : ViewModel() {
    
    private val _aramaSorgusu = MutableStateFlow("")
    private val _sadeceFavoriler = MutableStateFlow(false)
    
    val uiState: StateFlow<HomeUiState> = combine(
        _aramaSorgusu,
        _sadeceFavoriler,
        _aramaSorgusu.flatMapLatest { sorgu ->
            if (sorgu.isEmpty()) repository.tumNotlar
            else repository.ara(sorgu)
        }
    ) { sorgu, favoriler, notlar ->
        HomeUiState(
            notlar = if (favoriler) notlar.filter { it.favori } else notlar,
            aramaSorgusu = sorgu,
            sadecesFavoriler = favoriler
        )
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = HomeUiState(yukleniyor = true)
    )
    
    fun aramaGuncelle(sorgu: String) {
        _aramaSorgusu.value = sorgu
    }
    
    fun favoriFiltresiToggle() {
        _sadeceFavoriler.value = !_sadeceFavoriler.value
    }
    
    fun favoriToggle(notId: Int) {
        viewModelScope.launch {
            repository.favoriToggle(notId)
        }
    }
    
    fun notSil(not: Not) {
        viewModelScope.launch {
            repository.sil(not)
        }
    }
}
HomeScreen
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun HomeScreen(
    viewModel: HomeViewModel,
    onNotTikla: (Int) -> Unit,
    onYeniNot: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    var silinecekNot by remember { mutableStateOf<Not?>(null) }
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("📝 Notlarım") },
                actions = {
                    IconButton(onClick = { viewModel.favoriFiltresiToggle() }) {
                        Icon(
                            if (uiState.sadecesFavoriler) Icons.Default.Favorite 
                            else Icons.Default.FavoriteBorder,
                            contentDescription = "Favoriler",
                            tint = if (uiState.sadecesFavoriler) Color.Red else Color.Gray
                        )
                    }
                }
            )
        },
        floatingActionButton = {
            ExtendedFloatingActionButton(
                onClick = onYeniNot,
                icon = { Icon(Icons.Default.Add, null) },
                text = { Text("Yeni Not") }
            )
        }
    ) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
        ) {
            // Arama çubuğu
            OutlinedTextField(
                value = uiState.aramaSorgusu,
                onValueChange = { viewModel.aramaGuncelle(it) },
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp),
                placeholder = { Text("Notlarda ara...") },
                leadingIcon = { Icon(Icons.Default.Search, null) },
                trailingIcon = {
                    if (uiState.aramaSorgusu.isNotEmpty()) {
                        IconButton(onClick = { viewModel.aramaGuncelle("") }) {
                            Icon(Icons.Default.Clear, "Temizle")
                        }
                    }
                },
                singleLine = true,
                shape = RoundedCornerShape(12.dp)
            )
            
            // Not listesi
            if (uiState.notlar.isEmpty()) {
                EmptyState(
                    message = if (uiState.aramaSorgusu.isNotEmpty()) 
                        "Sonuç bulunamadı" else "Henüz not yok",
                    icon = Icons.Default.Note
                )
            } else {
                LazyVerticalStaggeredGrid(
                    columns = StaggeredGridCells.Fixed(2),
                    contentPadding = PaddingValues(16.dp),
                    horizontalArrangement = Arrangement.spacedBy(12.dp),
                    verticalItemSpacing = 12.dp
                ) {
                    items(uiState.notlar, key = { it.id }) { not ->
                        NotCard(
                            not = not,
                            onClick = { onNotTikla(not.id) },
                            onFavoriClick = { viewModel.favoriToggle(not.id) },
                            onSilClick = { silinecekNot = not },
                            modifier = Modifier.animateItemPlacement()
                        )
                    }
                }
            }
        }
    }
    
    // Silme onay dialog
    silinecekNot?.let { not ->
        AlertDialog(
            onDismissRequest = { silinecekNot = null },
            title = { Text("Notu Sil") },
            text = { Text("\"${not.baslik}\" notunu silmek istediğinize emin misiniz?") },
            confirmButton = {
                TextButton(
                    onClick = {
                        viewModel.notSil(not)
                        silinecekNot = null
                    }
                ) {
                    Text("Sil", color = Color.Red)
                }
            },
            dismissButton = {
                TextButton(onClick = { silinecekNot = null }) {
                    Text("İptal")
                }
            }
        )
    }
}

Detail Screen

DetailScreen (Ekleme/Düzenleme)
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun DetailScreen(
    notId: Int?,  // null ise yeni not
    viewModel: DetailViewModel,
    onGeriDon: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    
    // Not yükle (düzenleme modunda)
    LaunchedEffect(notId) {
        notId?.let { viewModel.notYukle(it) }
    }
    
    // Kaydetme başarılı olunca geri dön
    LaunchedEffect(uiState.kaydedildi) {
        if (uiState.kaydedildi) {
            onGeriDon()
        }
    }
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = { 
                    Text(if (notId == null) "Yeni Not" else "Notu Düzenle") 
                },
                navigationIcon = {
                    IconButton(onClick = onGeriDon) {
                        Icon(Icons.AutoMirrored.Filled.ArrowBack, "Geri")
                    }
                },
                actions = {
                    IconButton(
                        onClick = { viewModel.kaydet() },
                        enabled = uiState.baslik.isNotBlank()
                    ) {
                        Icon(Icons.Default.Check, "Kaydet")
                    }
                }
            )
        }
    ) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
                .padding(16.dp)
        ) {
            // Başlık
            OutlinedTextField(
                value = uiState.baslik,
                onValueChange = { viewModel.baslikGuncelle(it) },
                modifier = Modifier.fillMaxWidth(),
                placeholder = { Text("Başlık") },
                textStyle = MaterialTheme.typography.titleLarge,
                singleLine = true
            )
            
            Spacer(modifier = Modifier.height(16.dp))
            
            // Renk seçici
            Text("Renk Seçin:", style = MaterialTheme.typography.labelMedium)
            
            Spacer(modifier = Modifier.height(8.dp))
            
            LazyRow(
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                itemsIndexed(notRenkleri) { index, renk ->
                    Box(
                        modifier = Modifier
                            .size(40.dp)
                            .clip(CircleShape)
                            .background(renk)
                            .border(
                                width = if (uiState.secilenRenk == index) 3.dp else 1.dp,
                                color = if (uiState.secilenRenk == index) 
                                    MaterialTheme.colorScheme.primary else Color.Gray,
                                shape = CircleShape
                            )
                            .clickable { viewModel.renkSec(index) }
                    )
                }
            }
            
            Spacer(modifier = Modifier.height(16.dp))
            
            // İçerik
            OutlinedTextField(
                value = uiState.icerik,
                onValueChange = { viewModel.icerikGuncelle(it) },
                modifier = Modifier
                    .fillMaxWidth()
                    .weight(1f),
                placeholder = { Text("Not içeriği...") },
                textStyle = MaterialTheme.typography.bodyLarge
            )
        }
    }
}

Navigation

AppNavigation
@Composable
fun AppNavigation(
    repository: NotRepository
) {
    val navController = rememberNavController()
    
    NavHost(
        navController = navController,
        startDestination = "home"
    ) {
        composable("home") {
            val viewModel: HomeViewModel = viewModel(
                factory = HomeViewModelFactory(repository)
            )
            
            HomeScreen(
                viewModel = viewModel,
                onNotTikla = { notId ->
                    navController.navigate("detail/$notId")
                },
                onYeniNot = {
                    navController.navigate("detail/new")
                }
            )
        }
        
        composable(
            route = "detail/{notId}",
            arguments = listOf(
                navArgument("notId") { type = NavType.StringType }
            )
        ) { backStackEntry ->
            val notIdStr = backStackEntry.arguments?.getString("notId")
            val notId = if (notIdStr == "new") null else notIdStr?.toIntOrNull()
            
            val viewModel: DetailViewModel = viewModel(
                factory = DetailViewModelFactory(repository)
            )
            
            DetailScreen(
                notId = notId,
                viewModel = viewModel,
                onGeriDon = { navController.popBackStack() }
            )
        }
    }
}

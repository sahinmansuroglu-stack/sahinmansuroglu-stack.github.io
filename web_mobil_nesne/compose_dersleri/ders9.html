@Entity(tableName = "notlar")
data class Not(

DAO (Data Access Object)

DAO Interface
@Dao
interface NotDao {
    // Tüm notları getir (Flow ile reactive)
    @Query("SELECT * FROM notlar ORDER BY olusturma_tarihi DESC")
    fun tumNotlar(): Flow<List<Not>>
    
    // ID ile tek not getir
    @Query("SELECT * FROM notlar WHERE id = :notId")
    suspend fun notGetir(notId: Int): Not?
    
    // Favori notları getir
    @Query("SELECT * FROM notlar WHERE favori = 1")
    fun favoriNotlar(): Flow<List<Not>>
    
    // Arama
    @Query("SELECT * FROM notlar WHERE baslik LIKE '%' || :aramaMetni || '%'")
    fun notAra(aramaMetni: String): Flow<List<Not>>
    
    // Ekleme
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun notEkle(not: Not): Long
    
    // Güncelleme
    @Update
    suspend fun notGuncelle(not: Not)
    
    // Silme
    @Delete
    suspend fun notSil(not: Not)
    
    // ID ile silme
    @Query("DELETE FROM notlar WHERE id = :notId")
    suspend fun notSilById(notId: Int)
    
    // Favori toggle
    @Query("UPDATE notlar SET favori = NOT favori WHERE id = :notId")
    suspend fun favoriToggle(notId: Int)
    
    // Tüm notları sil
    @Query("DELETE FROM notlar")
    suspend fun tumNotlariSil()
}

Database

Database Sınıfı
@Database(
    entities = [Not::class],
    version = 1,
    exportSchema = false
)
abstract class NotDatabase : RoomDatabase() {
    abstract fun notDao(): NotDao
    
    companion object {
        @Volatile
        private var INSTANCE: NotDatabase? = null
        
        fun getDatabase(context: Context): NotDatabase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    NotDatabase::class.java,
                    "not_database"
                ).build()
                INSTANCE = instance
                instance
            }
        }
    }
}

Repository

Repository Pattern
class NotRepository(private val notDao: NotDao) {
    val tumNotlar: Flow<List<Not>> = notDao.tumNotlar()
    val favoriNotlar: Flow<List<Not>> = notDao.favoriNotlar()
    
    fun notAra(aramaMetni: String): Flow<List<Not>> {
        return notDao.notAra(aramaMetni)
    }
    
    suspend fun notGetir(notId: Int): Not? {
        return notDao.notGetir(notId)
    }
    
    suspend fun notEkle(not: Not): Long {
        return notDao.notEkle(not)
    }
    
    suspend fun notGuncelle(not: Not) {
        notDao.notGuncelle(not)
    }
    
    suspend fun notSil(not: Not) {
        notDao.notSil(not)
    }
    
    suspend fun favoriToggle(notId: Int) {
        notDao.favoriToggle(notId)
    }
}

ViewModel ile Kullanım

NotViewModel
class NotViewModel(
    private val repository: NotRepository
) : ViewModel() {
    
    val tumNotlar: StateFlow<List<Not>> = repository.tumNotlar
        .stateIn(
            scope = viewModelScope,
            started = SharingStarted.WhileSubscribed(5000),
            initialValue = emptyList()
        )
    
    private val _aramaSorgusu = MutableStateFlow("")
    
    val filtrelenmisNotlar: StateFlow<List<Not>> = _aramaSorgusu
        .flatMapLatest { sorgu ->
            if (sorgu.isEmpty()) {
                repository.tumNotlar
            } else {
                repository.notAra(sorgu)
            }
        }
        .stateIn(
            scope = viewModelScope,
            started = SharingStarted.WhileSubscribed(5000),
            initialValue = emptyList()
        )
    
    fun aramaGuncelle(sorgu: String) {
        _aramaSorgusu.value = sorgu
    }
    
    fun notEkle(baslik: String, icerik: String) {
        viewModelScope.launch {
            val yeniNot = Not(baslik = baslik, icerik = icerik)
            repository.notEkle(yeniNot)
        }
    }
    
    fun notSil(not: Not) {
        viewModelScope.launch {
            repository.notSil(not)
        }
    }
    
    fun favoriToggle(notId: Int) {
        viewModelScope.launch {
            repository.favoriToggle(notId)
        }
    }
}

// ViewModel Factory
class NotViewModelFactory(
    private val repository: NotRepository
) : ViewModelProvider.Factory {
    override fun <T : ViewModel> create(modelClass: Class<T>): T {
        if (modelClass.isAssignableFrom(NotViewModel::class.java)) {
            @Suppress("UNCHECKED_CAST")
            return NotViewModel(repository) as T
        }
        throw IllegalArgumentException("Unknown ViewModel class")
    }
}

UI Kullanımı

Not Uygulaması Ekranı
@Composable
fun NotListesiScreen(
    viewModel: NotViewModel
) {
    val notlar by viewModel.filtrelenmisNotlar.collectAsState()
    var aramaSorgusu by remember { mutableStateOf("") }
    var dialogAcik by remember { mutableStateOf(false) }
    
    Scaffold(
        topBar = {
            SearchBar(
                query = aramaSorgusu,
                onQueryChange = { 
                    aramaSorgusu = it
                    viewModel.aramaGuncelle(it)
                }
            )
        },
        floatingActionButton = {
            FloatingActionButton(onClick = { dialogAcik = true }) {
                Icon(Icons.Default.Add, "Not Ekle")
            }
        }
    ) { padding ->
        LazyColumn(
            modifier = Modifier.padding(padding),
            contentPadding = PaddingValues(16.dp),
            verticalArrangement = Arrangement.spacedBy(8.dp)
        ) {
            items(notlar, key = { it.id }) { not ->
                NotKart(
                    not = not,
                    onSil = { viewModel.notSil(not) },
                    onFavori = { viewModel.favoriToggle(not.id) }
                )
            }
        }
    }
    
    if (dialogAcik) {
        NotEkleDialog(
            onKapat = { dialogAcik = false },
            onKaydet = { baslik, icerik ->
                viewModel.notEkle(baslik, icerik)
                dialogAcik = false
            }
        )
    }
}

@Composable
fun NotKart(
    not: Not,
    onSil: () -> Unit,
    onFavori: () -> Unit
) {
    Card(modifier = Modifier.fillMaxWidth()) {
        Column(modifier = Modifier.padding(16.dp)) {
            Row(verticalAlignment = Alignment.CenterVertically) {
                Text(
                    text = not.baslik,
                    style = MaterialTheme.typography.titleMedium,
                    modifier = Modifier.weight(1f)
                )
                
                IconButton(onClick = onFavori) {
                    Icon(
                        imageVector = if (not.favori) 
                            Icons.Default.Favorite else Icons.Default.FavoriteBorder,
                        contentDescription = "Favori",
                        tint = if (not.favori) Color.Red else Color.Gray
                    )
                }
                
                IconButton(onClick = onSil) {
                    Icon(
                        Icons.Default.Delete,
                        contentDescription = "Sil",
                        tint = Color.Red
                    )
                }
            }
            
            Spacer(modifier = Modifier.height(8.dp))
            
            Text(
                text = not.icerik,
                style = MaterialTheme.typography.bodyMedium,
                maxLines = 3,
                overflow = TextOverflow.Ellipsis
            )
        }
    }
}

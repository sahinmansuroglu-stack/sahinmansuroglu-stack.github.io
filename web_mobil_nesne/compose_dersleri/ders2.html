Text Composable

Text Örnekleri
@Composable
fun TextOrnekleri() {
    Column(modifier = Modifier.padding(16.dp)) {
        // Basit text
        Text(text = "Merhaba Compose!")
        
        // Özelleştirilmiş text
        Text(
            text = "Büyük Başlık",
            fontSize = 28.sp,
            fontWeight = FontWeight.Bold,
            color = Color.Blue
        )
        
        // Material Typography kullanımı
        Text(
            text = "Material Başlık",
            style = MaterialTheme.typography.headlineMedium
        )
        
        // Çok satırlı text
        Text(
            text = "Bu uzun bir metin örneği. Birden fazla satıra yayılabilir.",
            maxLines = 2,
            overflow = TextOverflow.Ellipsis
        )
        
        // Annotated String (Zengin metin)
        Text(
            buildAnnotatedString {
                withStyle(SpanStyle(color = Color.Red)) {
                    append("Kırmızı")
                }
                append(" ve ")
                withStyle(SpanStyle(fontWeight = FontWeight.Bold)) {
                    append("Kalın")
                }
                append(" metin")
            }
        )
    }
}

Button Türleri

Farklı Button Çeşitleri
@Composable
fun ButtonOrnekleri() {
    Column(
        modifier = Modifier.padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        // Standart Button
        Button(onClick = { /* tıklama işlemi */ }) {
            Text("Standart Buton")
        }
        
        // Filled Tonal Button
        FilledTonalButton(onClick = { }) {
            Text("Tonal Buton")
        }
        
        // Outlined Button
        OutlinedButton(onClick = { }) {
            Text("Outlined Buton")
        }
        
        // Text Button
        TextButton(onClick = { }) {
            Text("Text Buton")
        }
        
        // Elevated Button
        ElevatedButton(onClick = { }) {
            Text("Elevated Buton")
        }
        
        // Icon ile Button
        Button(onClick = { }) {
            Icon(
                imageVector = Icons.Default.Favorite,
                contentDescription = null,
                modifier = Modifier.size(18.dp)
            )
            Spacer(Modifier.width(8.dp))
            Text("Beğen")
        }
        
        // Floating Action Button
        FloatingActionButton(onClick = { }) {
            Icon(Icons.Default.Add, contentDescription = "Ekle")
        }
        
        // Extended FAB
        ExtendedFloatingActionButton(
            onClick = { },
            icon = { Icon(Icons.Default.Edit, null) },
            text = { Text("Düzenle") }
        )
    }
}

Image ve Icon

Görsel Kullanımı
@Composable
fun ImageOrnekleri() {
    Column(
        modifier = Modifier.padding(16.dp),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        // Drawable'dan resim
        Image(
            painter = painterResource(id = R.drawable.ic_launcher_foreground),
            contentDescription = "Logo",
            modifier = Modifier.size(100.dp)
        )
        
        // Yuvarlak resim
        Image(
            painter = painterResource(id = R.drawable.profile),
            contentDescription = "Profil",
            modifier = Modifier
                .size(80.dp)
                .clip(CircleShape)
                .border(2.dp, Color.Gray, CircleShape),
            contentScale = ContentScale.Crop
        )
        
        // Material Icon
        Icon(
            imageVector = Icons.Default.Person,
            contentDescription = "Kullanıcı",
            modifier = Modifier.size(48.dp),
            tint = MaterialTheme.colorScheme.primary
        )
        
        // Outlined Icon
        Icon(
            imageVector = Icons.Outlined.Settings,
            contentDescription = "Ayarlar",
            tint = Color.Gray
        )
    }
}

TextField

Metin Girişi
@Composable
fun TextFieldOrnekleri() {
    var text by remember { mutableStateOf("") }
    var password by remember { mutableStateOf("") }
    
    Column(
        modifier = Modifier.padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        // Basit TextField
        TextField(
            value = text,
            onValueChange = { text = it },
            label = { Text("İsim") },
            placeholder = { Text("Adınızı girin") }
        )
        
        // Outlined TextField
        OutlinedTextField(
            value = text,
            onValueChange = { text = it },
            label = { Text("E-posta") },
            leadingIcon = { Icon(Icons.Default.Email, null) },
            singleLine = true
        )
        
        // Şifre alanı
        OutlinedTextField(
            value = password,
            onValueChange = { password = it },
            label = { Text("Şifre") },
            leadingIcon = { Icon(Icons.Default.Lock, null) },
            visualTransformation = PasswordVisualTransformation(),
            keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Password)
        )
        
        // Hata durumlu TextField
        OutlinedTextField(
            value = text,
            onValueChange = { text = it },
            label = { Text("Kullanıcı Adı") },
            isError = text.length < 3,
            supportingText = {
                if (text.length < 3) {
                    Text("En az 3 karakter giriniz")
                }
            }
        )
    }
}

Diğer Temel Bileşenler

Checkbox, Switch, Slider
@Composable
fun DigerBilesenler() {
    var checked by remember { mutableStateOf(false) }
    var switched by remember { mutableStateOf(false) }
    var sliderValue by remember { mutableFloatStateOf(0.5f) }
    
    Column(modifier = Modifier.padding(16.dp)) {
        // Checkbox
        Row(verticalAlignment = Alignment.CenterVertically) {
            Checkbox(
                checked = checked,
                onCheckedChange = { checked = it }
            )
            Text("Şartları kabul ediyorum")
        }
        
        // Switch
        Row(
            verticalAlignment = Alignment.CenterVertically,
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Bildirimler", modifier = Modifier.weight(1f))
            Switch(
                checked = switched,
                onCheckedChange = { switched = it }
            )
        }
        
        // Slider
        Text("Ses: ${(sliderValue * 100).toInt()}%")
        Slider(
            value = sliderValue,
            onValueChange = { sliderValue = it },
            valueRange = 0f..1f
        )
        
        // CircularProgressIndicator
        CircularProgressIndicator()
        
        // LinearProgressIndicator
        LinearProgressIndicator(
            progress = { sliderValue },
            modifier = Modifier.fillMaxWidth()
        )
    }
}

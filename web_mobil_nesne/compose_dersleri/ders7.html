NavHost ve NavController
// Route tanımları
object Routes {
    const val HOME = "home"
    const val DETAIL = "detail"
    const val PROFILE = "profile"
    const val SETTINGS = "settings"
}

@Composable
fun AppNavigation() {
    val navController = rememberNavController()
    
    NavHost(
        navController = navController,
        startDestination = Routes.HOME
    ) {
        composable(Routes.HOME) {
            AnaSayfa(
                onDetayaGit = { navController.navigate(Routes.DETAIL) },
                onProfil = { navController.navigate(Routes.PROFILE) }
            )
        }
        
        composable(Routes.DETAIL) {
            DetaySayfasi(
                onGeriDon = { navController.popBackStack() }
            )
        }
        
        composable(Routes.PROFILE) {
            ProfilSayfasi(navController)
        }
        
        composable(Routes.SETTINGS) {
            AyarlarSayfasi(navController)
        }
    }
}

@Composable
fun AnaSayfa(
    onDetayaGit: () -> Unit,
    onProfil: () -> Unit
) {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        horizontalAlignment = Alignment.CenterHorizontally,
        verticalArrangement = Arrangement.Center
    ) {
        Text("Ana Sayfa", style = MaterialTheme.typography.headlineLarge)
        
        Spacer(modifier = Modifier.height(16.dp))
        
        Button(onClick = onDetayaGit) {
            Text("Detay Sayfasına Git")
        }
        
        Button(onClick = onProfil) {
            Text("Profil Sayfasına Git")
        }
    }
}

@Composable
fun DetaySayfasi(onGeriDon: () -> Unit) {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp)
    ) {
        IconButton(onClick = onGeriDon) {
            Icon(Icons.AutoMirrored.Filled.ArrowBack, "Geri")
        }
        
        Text("Detay Sayfası", style = MaterialTheme.typography.headlineLarge)
    }
}

Parametre ile Navigation

Veri Aktarımı
@Composable
fun ParametreliNavigation() {
    val navController = rememberNavController()
    
    NavHost(
        navController = navController,
        startDestination = "urunler"
    ) {
        // Ürün listesi
        composable("urunler") {
            UrunListesi(
                onUrunTikla = { urunId ->
                    navController.navigate("urun/$urunId")
                }
            )
        }
        
        // Ürün detayı (zorunlu parametre)
        composable(
            route = "urun/{urunId}",
            arguments = listOf(
                navArgument("urunId") { type = NavType.IntType }
            )
        ) { backStackEntry ->
            val urunId = backStackEntry.arguments?.getInt("urunId") ?: 0
            UrunDetay(
                urunId = urunId,
                onGeriDon = { navController.popBackStack() }
            )
        }
        
        // Opsiyonel parametre
        composable(
            route = "arama?query={query}",
            arguments = listOf(
                navArgument("query") {
                    type = NavType.StringType
                    defaultValue = ""
                    nullable = true
                }
            )
        ) { backStackEntry ->
            val query = backStackEntry.arguments?.getString("query")
            AramaSayfasi(query = query ?: "")
        }
    }
}

@Composable
fun UrunListesi(onUrunTikla: (Int) -> Unit) {
    val urunler = (1..20).map { "Ürün $it" }
    
    LazyColumn(contentPadding = PaddingValues(16.dp)) {
        itemsIndexed(urunler) { index, urun ->
            Card(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(vertical = 4.dp)
                    .clickable { onUrunTikla(index + 1) }
            ) {
                Text(
                    text = urun,
                    modifier = Modifier.padding(16.dp)
                )
            }
        }
    }
}

@Composable
fun UrunDetay(urunId: Int, onGeriDon: () -> Unit) {
    Column(modifier = Modifier.padding(16.dp)) {
        IconButton(onClick = onGeriDon) {
            Icon(Icons.AutoMirrored.Filled.ArrowBack, "Geri")
        }
        
        Text(
            "Ürün Detayı",
            style = MaterialTheme.typography.headlineLarge
        )
        Text(
            "Ürün ID: $urunId",
            style = MaterialTheme.typography.titleLarge
        )
    }
}

Bottom Navigation

Alt Navigasyon Çubuğu
sealed class BottomNavItem(
    val route: String,
    val title: String,
    val icon: ImageVector
) {
    object Home : BottomNavItem("home", "Ana Sayfa", Icons.Default.Home)
    object Search : BottomNavItem("search", "Ara", Icons.Default.Search)
    object Favorites : BottomNavItem("favorites", "Favoriler", Icons.Default.Favorite)
    object Profile : BottomNavItem("profile", "Profil", Icons.Default.Person)
}

@Composable
fun MainScreen() {
    val navController = rememberNavController()
    val items = listOf(
        BottomNavItem.Home,
        BottomNavItem.Search,
        BottomNavItem.Favorites,
        BottomNavItem.Profile
    )
    
    Scaffold(
        bottomBar = {
            NavigationBar {
                val navBackStackEntry by navController.currentBackStackEntryAsState()
                val currentRoute = navBackStackEntry?.destination?.route
                
                items.forEach { item ->
                    NavigationBarItem(
                        icon = { Icon(item.icon, contentDescription = item.title) },
                        label = { Text(item.title) },
                        selected = currentRoute == item.route,
                        onClick = {
                            navController.navigate(item.route) {
                                // Pop up to the start destination
                                popUpTo(navController.graph.findStartDestination().id) {
                                    saveState = true
                                }
                                // Avoid multiple copies
                                launchSingleTop = true
                                // Restore state
                                restoreState = true
                            }
                        }
                    )
                }
            }
        }
    ) { paddingValues ->
        NavHost(
            navController = navController,
            startDestination = BottomNavItem.Home.route,
            modifier = Modifier.padding(paddingValues)
        ) {
            composable(BottomNavItem.Home.route) { HomePage() }
            composable(BottomNavItem.Search.route) { SearchPage() }
            composable(BottomNavItem.Favorites.route) { FavoritesPage() }
            composable(BottomNavItem.Profile.route) { ProfilePage() }
        }
    }
}

@Composable

Side Effect Nedir?

Side effect, Composable fonksiyonun kapsamı dışında gerçekleşen işlemlerdir. Örneğin: API çağrıları, analitik olayları, timer'lar vb.

⚠️ Önemli Kurallar:
• Composable'lar herhangi bir zamanda yeniden çalışabilir (recomposition)
• Side effect'ler kontrollü bir şekilde yönetilmeli
• Compose kendi yaşam döngüsü API'lerini sağlar

LaunchedEffect

Coroutine ile Side Effect
@Composable
fun LaunchedEffectOrnekleri() {
    var mesaj by remember { mutableStateOf("") }
    
    // Sadece bir kez çalışır (composition'da)
    LaunchedEffect(Unit) {
        // Bu kod sadece composable ilk kez compose edildiğinde çalışır
        delay(2000)
        mesaj = "2 saniye sonra yüklendi!"
    }
    
    Text(mesaj.ifEmpty { "Yükleniyor..." })
}

@Composable
fun KeyIleLaunchedEffect(userId: Int) {
    var kullanici by remember { mutableStateOf<String?>(null) }
    
    // userId değiştiğinde yeniden çalışır
    LaunchedEffect(userId) {
        kullanici = null  // Reset
        delay(1000)  // Simulated API call
        kullanici = "Kullanıcı $userId"
    }
    
    Text(kullanici ?: "Kullanıcı yükleniyor...")
}

@Composable
fun TimerOrnek() {
    var saniye by remember { mutableIntStateOf(0) }
    var calisiyor by remember { mutableStateOf(true) }
    
    // calisiyor true olduğu sürece timer çalışır
    // calisiyor false olunca coroutine iptal edilir
    LaunchedEffect(calisiyor) {
        while (calisiyor) {
            delay(1000)
            saniye++
        }
    }
    
    Column(horizontalAlignment = Alignment.CenterHorizontally) {
        Text(
            text = "$saniye saniye",
            fontSize = 32.sp,
            fontWeight = FontWeight.Bold
        )
        
        Spacer(modifier = Modifier.height(16.dp))
        
        Button(onClick = { calisiyor = !calisiyor }) {
            Text(if (calisiyor) "Durdur" else "Başlat")
        }
    }
}

@Composable
fun SnackbarOrnek() {
    val snackbarHostState = remember { SnackbarHostState() }
    var showSnackbar by remember { mutableStateOf(false) }
    
    // One-time event handling
    LaunchedEffect(showSnackbar) {
        if (showSnackbar) {
            snackbarHostState.showSnackbar(
                message = "İşlem tamamlandı!",
                duration = SnackbarDuration.Short
            )
            showSnackbar = false
        }
    }
    
    Scaffold(
        snackbarHost = { SnackbarHost(snackbarHostState) }
    ) { padding ->
        Box(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding),
            contentAlignment = Alignment.Center
        ) {
            Button(onClick = { showSnackbar = true }) {
                Text("Snackbar Göster")
            }
        }
    }
}

rememberCoroutineScope

Event Handler'da Coroutine
@Composable
fun CoroutineScopeOrnek() {
    val scope = rememberCoroutineScope()
    var mesaj by remember { mutableStateOf("") }
    var yukleniyor by remember { mutableStateOf(false) }
    
    Column(
        modifier = Modifier.padding(16.dp),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Button(
            onClick = {
                // Event handler'da coroutine başlat
                scope.launch {
                    yukleniyor = true
                    delay(2000)  // Simulated work
                    mesaj = "İşlem tamamlandı!"
                    yukleniyor = false
                }
            },
            enabled = !yukleniyor
        ) {
            if (yukleniyor) {
                CircularProgressIndicator(
                    modifier = Modifier.size(20.dp),
                    color = Color.White,
                    strokeWidth = 2.dp
                )
            } else {
                Text("İşlem Başlat")
            }
        }
        
        Spacer(modifier = Modifier.height(16.dp))
        
        Text(mesaj)
    }
}

DisposableEffect

Cleanup Gerektiren Durumlar
@Composable
fun DisposableEffectOrnek(lifecycleOwner: LifecycleOwner) {
    DisposableEffect(lifecycleOwner) {
        val observer = LifecycleEventObserver { _, event ->
            when (event) {
                Lifecycle.Event.ON_START -> {
                    Log.d("Lifecycle", "Ekran görünür oldu")
                }
                Lifecycle.Event.ON_STOP -> {
                    Log.d("Lifecycle", "Ekran gizlendi")
                }
                else -> {}
            }
        }
        
        lifecycleOwner.lifecycle.addObserver(observer)
        
        // Cleanup: Composable kaldırıldığında çağrılır
        onDispose {
            lifecycleOwner.lifecycle.removeObserver(observer)
            Log.d("Lifecycle", "Observer kaldırıldı")
        }
    }
    
    Text("Lifecycle Observer Aktif")
}

    }
}

SideEffect

Non-Compose Kod İle Senkronizasyon
@Composable
fun SideEffectOrnek(onColorChange: (Color) -> Unit) {
    var color by remember { mutableStateOf(Color.Red) }
    
    // Her başarılı recomposition sonrası çalışır
    SideEffect {
        // Non-compose kod ile state senkronizasyonu
        onColorChange(color)
        Log.d("SideEffect", "Renk değişti: $color")
    }
    
    Column(modifier = Modifier.padding(16.dp)) {
        Box(
            modifier = Modifier
                .size(100.dp)
                .background(color)
        )
        
        Spacer(modifier = Modifier.height(16.dp))
        
        Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
            Button(onClick = { color = Color.Red }) { Text("Kırmızı") }
            Button(onClick = { color = Color.Green }) { Text("Yeşil") }
            Button(onClick = { color = Color.Blue }) { Text("Mavi") }
        }
    }
}

produceState

Asenkron Veri Üretimi
@Composable
fun ProduceStateOrnek(url: String) {
    val imageState by produceState<ImageLoadState>(
        initialValue = ImageLoadState.Loading,
        key1 = url
    ) {
        value = ImageLoadState.Loading
        
        try {
            // Simulated image loading
            delay(2000)
            value = ImageLoadState.Success(url)
        } catch (e: Exception) {
            value = ImageLoadState.Error(e.message ?: "Hata")
        }
    }
    
    when (val state = imageState) {
        is ImageLoadState.Loading -> {
            CircularProgressIndicator()
        }
        is ImageLoadState.Success -> {
            Text("Resim yüklendi: ${state.url}")
        }
        is ImageLoadState.Error -> {
            Text("Hata: ${state.message}", color = Color.Red)
        }
    }
}

sealed class ImageLoadState {
    object Loading : ImageLoadState()
    data class Success(val url: String) : ImageLoadState()
    data class Error(val message: String) : ImageLoadState()
}

derivedStateOf

Hesaplanmış State
@Composable
fun DerivedStateOrnek() {
    var firstName by remember { mutableStateOf("") }
    var lastName by remember { mutableStateOf("") }
    
    // Sadece firstName veya lastName değiştiğinde hesaplanır
    val fullName by remember {
        derivedStateOf { 
            "$firstName $lastName".trim()
        }
    }
    
    // Form geçerliliği
    val isValid by remember {
        derivedStateOf {
            firstName.length >= 2 && lastName.length >= 2
        }
    }
    
    Column(modifier = Modifier.padding(16.dp)) {
        OutlinedTextField(
            value = firstName,
            onValueChange = { firstName = it },
            label = { Text("Ad") }
        )
        
        Spacer(modifier = Modifier.height(8.dp))
        
        OutlinedTextField(
            value = lastName,
            onValueChange = { lastName = it },
            label = { Text("Soyad") }
        )
        
        Spacer(modifier = Modifier.height(16.dp))
        
        Text("Tam İsim: $fullName")
        
        Spacer(modifier = Modifier.height(8.dp))
        
        Button(
            onClick = { },
            enabled = isValid
        ) {
            Text("Kaydet")
        }
    }
}
